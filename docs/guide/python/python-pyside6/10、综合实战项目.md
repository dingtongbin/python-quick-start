### 完整代码

```python
from PySide6.QtWidgets import (
    QApplication, QMainWindow, QWidget, QVBoxLayout,
    QHBoxLayout, QTextEdit, QPushButton, QSplitter,
    QLabel, QFileDialog, QMessageBox, QToolBar
)
from PySide6.QtCore import Qt
import markdown
import sys

class MarkdownEditor(QMainWindow):
    """Markdown 编辑器"""
    
    def __init__(self):
        super().__init__()
        self.setWindowTitle("📝 Markdown 编辑器")
        self.setGeometry(100, 100, 1200, 800)
        
        self.current_file = None
        
        # 中央部件 - 分割器
        splitter = QSplitter(Qt.Horizontal)
        
        # 编辑区
        self.editor = QTextEdit()
        self.editor.setPlaceholderText("在此输入 Markdown 文本...")
        self.editor.setFontPointSize(12)
        splitter.addWidget(self.editor)
        
        # 预览区
        self.preview = QTextEdit()
        self.preview.setReadOnly(True)
        splitter.addWidget(self.preview)
        
        # 设置比例
        splitter.setStretchFactor(0, 1)
        splitter.setStretchFactor(1, 1)
        
        self.setCentralWidget(splitter)
        
        # 创建菜单和工具栏
        self.create_menu()
        self.create_toolbar()
        
        # 实时预览
        self.editor.textChanged.connect(self.update_preview)
        
        self.statusBar().showMessage("就绪")
    
    def create_menu(self):
        """创建菜单栏"""
        menubar = self.menuBar()
        
        # 文件菜单
        file_menu = menubar.addMenu("文件(&F)")
        
        new_action = file_menu.addAction("新建(&N)")
        new_action.setShortcut("Ctrl+N")
        new_action.triggered.connect(self.new_file)
        
        open_action = file_menu.addAction("打开(&O)")
        open_action.setShortcut("Ctrl+O")
        open_action.triggered.connect(self.open_file)
        
        save_action = file_menu.addAction("保存(&S)")
        save_action.setShortcut("Ctrl+S")
        save_action.triggered.connect(self.save_file)
        
        file_menu.addSeparator()
        
        export_action = file_menu.addAction("导出 HTML")
        export_action.triggered.connect(self.export_html)
        
        file_menu.addSeparator()
        
        exit_action = file_menu.addAction("退出(&X)")
        exit_action.setShortcut("Ctrl+Q")
        exit_action.triggered.connect(self.close)
        
        # 帮助菜单
        help_menu = menubar.addMenu("帮助(&H)")
        about_action = help_menu.addAction("关于")
        about_action.triggered.connect(self.show_about)
    
    def create_toolbar(self):
        """创建工具栏"""
        toolbar = QToolBar("主工具栏")
        self.addToolBar(toolbar)
        
        toolbar.addAction("📄 新建", self.new_file)
        toolbar.addAction("📂 打开", self.open_file)
        toolbar.addAction("💾 保存", self.save_file)
        toolbar.addSeparator()
        toolbar.addAction("🌐 导出HTML", self.export_html)
    
    def update_preview(self):
        """更新预览"""
        text = self.editor.toPlainText()
        
        # 转换为 HTML
        html = markdown.markdown(text, extensions=['extra', 'codehilite'])
        
        # 添加样式
        styled_html = f"""
        <!DOCTYPE html>
        <html>
        <head>
            <meta charset="utf-8">
            <style>
                body {{
                    font-family: "Microsoft YaHei", Arial, sans-serif;
                    padding: 20px;
                    line-height: 1.6;
                    color: #333;
                }}
                h1, h2, h3 {{
                    color: #2196F3;
                    margin-top: 20px;
                }}
                code {{
                    background-color: #f4f4f4;
                    padding: 2px 5px;
                    border-radius: 3px;
                    font-family: "Consolas", monospace;
                }}
                pre {{
                    background-color: #f4f4f4;
                    padding: 15px;
                    border-radius: 5px;
                    overflow-x: auto;
                }}
                blockquote {{
                    border-left: 4px solid #ddd;
                    padding-left: 15px;
                    color: #666;
                    margin: 10px 0;
                }}
                table {{
                    border-collapse: collapse;
                    width: 100%;
                }}
                th, td {{
                    border: 1px solid #ddd;
                    padding: 8px;
                    text-align: left;
                }}
                th {{
                    background-color: #f4f4f4;
                }}
                a {{
                    color: #2196F3;
                    text-decoration: none;
                }}
                a:hover {{
                    text-decoration: underline;
                }}
            </style>
        </head>
        <body>
        {html}
        </body>
        </html>
        """
        
        self.preview.setHtml(styled_html)
    
    def new_file(self):
        """新建文件"""
        if self.maybe_save():
            self.editor.clear()
            self.current_file = None
            self.setWindowTitle("Markdown 编辑器 - 未命名")
            self.statusBar().showMessage("新建文件")
    
    def open_file(self):
        """打开文件"""
        if not self.maybe_save():
            return
        
        file_path, _ = QFileDialog.getOpenFileName(
            self,
            "打开文件",
            "",
            "Markdown 文件 (*.md);;所有文件 (*)"
        )
        
        if file_path:
            try:
                with open(file_path, 'r', encoding='utf-8') as f:
                    content = f.read()
                
                self.editor.setPlainText(content)
                self.current_file = file_path
                self.setWindowTitle(f"Markdown 编辑器 - {file_path}")
                self.statusBar().showMessage(f"已打开: {file_path}")
            
            except Exception as e:
                QMessageBox.critical(self, "错误", f"无法打开文件:\n{str(e)}")
    
    def save_file(self):
        """保存文件"""
        if not self.current_file:
            self.save_as()
            return
        
        try:
            content = self.editor.toPlainText()
            with open(self.current_file, 'w', encoding='utf-8') as f:
                f.write(content)
            
            self.statusBar().showMessage(f"已保存: {self.current_file}")
        
        except Exception as e:
            QMessageBox.critical(self, "错误", f"无法保存文件:\n{str(e)}")
    
    def save_as(self):
        """另存为"""
        file_path, _ = QFileDialog.getSaveFileName(
            self,
            "保存文件",
            "",
            "Markdown 文件 (*.md);;所有文件 (*)"
        )
        
        if file_path:
            if not file_path.endswith('.md'):
                file_path += '.md'
            
            self.current_file = file_path
            self.save_file()
    
    def export_html(self):
        """导出为 HTML"""
        file_path, _ = QFileDialog.getSaveFileName(
            self,
            "导出 HTML",
            "",
            "HTML 文件 (*.html);;所有文件 (*)"
        )
        
        if file_path:
            if not file_path.endswith('.html'):
                file_path += '.html'
            
            try:
                text = self.editor.toPlainText()
                html = markdown.markdown(text, extensions=['extra', 'codehilite'])
                
                full_html = f"""<!DOCTYPE html>
<html lang="zh-CN">
<head>
    <meta charset="utf-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Exported Markdown</title>
    <style>
        body {{
            font-family: "Microsoft YaHei", Arial, sans-serif;
            max-width: 800px;
            margin: 0 auto;
            padding: 20px;
            line-height: 1.6;
        }}
    </style>
</head>
<body>
{html}
</body>
</html>"""
                
                with open(file_path, 'w', encoding='utf-8') as f:
                    f.write(full_html)
                
                QMessageBox.information(
                    self,
                    "成功",
                    f"已导出到:\n{file_path}"
                )
            
            except Exception as e:
                QMessageBox.critical(self, "错误", f"导出失败:\n{str(e)}")
    
    def maybe_save(self):
        """检查是否需要保存"""
        if self.editor.document().isModified():
            reply = QMessageBox.question(
                self,
                "确认",
                "文档已修改，是否保存？",
                QMessageBox.Save | QMessageBox.Discard | QMessageBox.Cancel
            )
            
            if reply == QMessageBox.Save:
                self.save_file()
                return True
            elif reply == QMessageBox.Cancel:
                return False
        
        return True
    
    def show_about(self):
        """显示关于对话框"""
        QMessageBox.about(
            self,
            "关于",
            "<h2>Markdown 编辑器</h2>"
            "<p>一个简单易用的 Markdown 编辑器</p>"
            "<p>支持实时预览、文件管理、HTML 导出</p>"
        )

if __name__ == "__main__":
    app = QApplication(sys.argv)
    
    # 设置全局样式
    app.setStyleSheet("""
        QMainWindow {
            background-color: #f5f5f5;
        }
        QSplitter::handle {
            background-color: #ddd;
        }
        QToolBar {
            background-color: white;
            border-bottom: 1px solid #ddd;
            spacing: 5px;
        }
        QToolBar QAction {
            padding: 5px;
        }
    """)
    
    editor = MarkdownEditor()
    editor.show()
    sys.exit(app.exec())
```

### 核心代码

```python
import json
from pathlib import Path
from datetime import datetime
from PySide6.QtWidgets import (
    QApplication, QMainWindow, QWidget, QVBoxLayout,
    QHBoxLayout, QLineEdit, QPushButton, QListWidget,
    QListWidgetItem, QMessageBox, QLabel, QComboBox,
    QDateEdit, QCheckBox, QGroupBox, QFormLayout
)
from PySide6.QtCore import Qt, QDate

class TodoItem:
    """待办事项数据模型"""
    
    def __init__(self, text, category="默认", priority="中", due_date=None):
        self.text = text
        self.category = category
        self.priority = priority
        self.due_date = due_date
        self.done = False
        self.created_at = datetime.now().isoformat()
    
    def to_dict(self):
        return {
            'text': self.text,
            'category': self.category,
            'priority': self.priority,
            'due_date': self.due_date,
            'done': self.done,
            'created_at': self.created_at
        }
    
    @staticmethod
    def from_dict(data):
        item = TodoItem(data['text'], data['category'], data['priority'])
        item.due_date = data.get('due_date')
        item.done = data.get('done', False)
        item.created_at = data.get('created_at')
        return item

class EnhancedTodoApp(QMainWindow):
    def __init__(self):
        super().__init__()
        self.setWindowTitle("📝 待办事项 - 增强版")
        self.setGeometry(100, 100, 700, 800)
        
        self.todos = []
        self.data_file = Path("todos_enhanced.json")
        
        self.init_ui()
        self.load_todos()
    
    def init_ui(self):
        central_widget = QWidget()
        self.setCentralWidget(central_widget)
        main_layout = QVBoxLayout()
        
        # 标题
        title = QLabel("待办事项管理器")
        title.setAlignment(Qt.AlignCenter)
        title.setStyleSheet("""
            font-size: 24px;
            font-weight: bold;
            padding: 15px;
            color: #2196F3;
            background-color: white;
            border-radius: 5px;
        """)
        main_layout.addWidget(title)
        
        # 添加任务表单
        form_group = QGroupBox("添加新任务")
        form_layout = QFormLayout()
        
        self.todo_input = QLineEdit()
        self.todo_input.setPlaceholderText("任务内容...")
        form_layout.addRow("内容:", self.todo_input)
        
        self.category_combo = QComboBox()
        self.category_combo.addItems(["工作", "学习", "生活", "其他"])
        form_layout.addRow("分类:", self.category_combo)
        
        self.priority_combo = QComboBox()
        self.priority_combo.addItems(["高", "中", "低"])
        form_layout.addRow("优先级:", self.priority_combo)
        
        self.due_date_edit = QDateEdit()
        self.due_date_edit.setCalendarPopup(True)
        self.due_date_edit.setDate(QDate.currentDate().addDays(7))
        form_layout.addRow("截止日期:", self.due_date_edit)
        
        form_group.setLayout(form_layout)
        main_layout.addWidget(form_group)
        
        # 添加按钮
        add_btn = QPushButton("➕ 添加任务")
        add_btn.clicked.connect(self.add_todo)
        add_btn.setStyleSheet("""
            QPushButton {
                background-color: #4CAF50;
                color: white;
                padding: 10px;
                border-radius: 5px;
                font-weight: bold;
            }
            QPushButton:hover {
                background-color: #45a049;
            }
        """)
        main_layout.addWidget(add_btn)
        
        # 过滤选项
        filter_layout = QHBoxLayout()
        filter_label = QLabel("筛选:")
        self.filter_combo = QComboBox()
        self.filter_combo.addItems(["全部", "未完成", "已完成", "高优先级"])
        self.filter_combo.currentTextChanged.connect(self.apply_filter)
        filter_layout.addWidget(filter_label)
        filter_layout.addWidget(self.filter_combo)
        filter_layout.addStretch()
        main_layout.addLayout(filter_layout)
        
        # 任务列表
        self.todo_list = QListWidget()
        self.todo_list.itemDoubleClicked.connect(self.toggle_complete)
        main_layout.addWidget(self.todo_list)
        
        # 统计信息
        self.stats_label = QLabel("")
        self.stats_label.setStyleSheet("""
            padding: 10px;
            background-color: white;
            border-radius: 5px;
            font-size: 12px;
        """)
        main_layout.addWidget(self.stats_label)
        
        central_widget.setLayout(main_layout)
    
    def add_todo(self):
        text = self.todo_input.text().strip()
        if not text:
            QMessageBox.warning(self, "提示", "请输入任务内容！")
            return
        
        todo = TodoItem(
            text=text,
            category=self.category_combo.currentText(),
            priority=self.priority_combo.currentText(),
            due_date=self.due_date_edit.date().toString("yyyy-MM-dd")
        )
        
        self.todos.append(todo)
        self.refresh_list()
        self.todo_input.clear()
        self.save_todos()
    
    def toggle_complete(self, item):
        row = self.todo_list.row(item)
        if 0 <= row < len(self.todos):
            self.todos[row].done = not self.todos[row].done
            self.refresh_list()
            self.save_todos()
    
    def apply_filter(self, filter_text):
        self.refresh_list(filter_text)
    
    def refresh_list(self, filter_text="全部"):
        self.todo_list.clear()
        
        filtered_todos = []
        for todo in self.todos:
            if filter_text == "全部":
                filtered_todos.append(todo)
            elif filter_text == "未完成" and not todo.done:
                filtered_todos.append(todo)
            elif filter_text == "已完成" and todo.done:
                filtered_todos.append(todo)
            elif filter_text == "高优先级" and todo.priority == "高":
                filtered_todos.append(todo)
        
        for todo in filtered_todos:
            item = QListWidgetItem()
            
            # 设置显示文本
            status = "✓" if todo.done else "○"
            priority_icon = {"高": "🔴", "中": "🟡", "低": "🟢"}[todo.priority]
            display_text = f"{status} {priority_icon} {todo.text}"
            
            item.setText(display_text)
            
            # 设置样式
            if todo.done:
                item.setForeground(Qt.gray)
                font = item.font()
                font.setStrikeOut(True)
                item.setFont(font)
            
            # 存储数据
            item.setData(Qt.UserRole, todo)
            
            self.todo_list.addItem(item)
        
        self.update_stats()
    
    def update_stats(self):
        total = len(self.todos)
        done = sum(1 for t in self.todos if t.done)
        pending = total - done
        high_priority = sum(1 for t in self.todos if t.priority == "高" and not t.done)
        
        stats_text = f"""
        📊 统计信息：
        总任务: {total} | 已完成: {done} | 待完成: {pending}
        高优先级待办: {high_priority}
        """
        self.stats_label.setText(stats_text)
    
    def save_todos(self):
        data = [todo.to_dict() for todo in self.todos]
        with open(self.data_file, 'w', encoding='utf-8') as f:
            json.dump(data, f, ensure_ascii=False, indent=2)
    
    def load_todos(self):
        if self.data_file.exists():
            with open(self.data_file, 'r', encoding='utf-8') as f:
                data = json.load(f)
            self.todos = [TodoItem.from_dict(d) for d in data]
            self.refresh_list()

if __name__ == "__main__":
    app = QApplication([])
    window = EnhancedTodoApp()
    window.show()
    app.exec()
```
