## 12.2 MVC 模式在 Qt 中的应用

### Model（模型）

```python
class TodoModel:
    """数据模型"""
    
    def __init__(self):
        self.todos = []
        self._observers = []
    
    def add_observer(self, callback):
        """注册观察者"""
        self._observers.append(callback)
    
    def notify(self):
        """通知观察者"""
        for callback in self._observers:
            callback()
    
    def add_todo(self, text):
        """添加任务"""
        todo = {'text': text, 'done': False}
        self.todos.append(todo)
        self.notify()
    
    def remove_todo(self, index):
        """删除任务"""
        if 0 <= index < len(self.todos):
            self.todos.pop(index)
            self.notify()
    
    def toggle_todo(self, index):
        """切换完成状态"""
        if 0 <= index < len(self.todos):
            self.todos[index]['done'] = not self.todos[index]['done']
            self.notify()
    
    def get_all(self):
        """获取所有任务"""
        return self.todos.copy()
```

### View（视图）

```python
class TodoView(QMainWindow):
    """视图层"""
    
    def __init__(self):
        super().__init__()
        self.setWindowTitle("待办事项")
        
        # 创建组件
        self.list_widget = QListWidget()
        self.input_field = QLineEdit()
        self.add_button = QPushButton("添加")
        
        # 布局
        layout = QVBoxLayout()
        layout.addWidget(self.input_field)
        layout.addWidget(self.add_button)
        layout.addWidget(self.list_widget)
        
        central_widget = QWidget()
        central_widget.setLayout(layout)
        self.setCentralWidget(central_widget)
    
    def display_todos(self, todos):
        """显示任务列表"""
        self.list_widget.clear()
        for todo in todos:
            status = "✓" if todo['done'] else "○"
            item = QListWidgetItem(f"{status} {todo['text']}")
            self.list_widget.addItem(item)
    
    def get_input(self):
        """获取输入"""
        return self.input_field.text()
    
    def clear_input(self):
        """清空输入"""
        self.input_field.clear()
```

### Controller（控制器）

```python
class TodoController:
    """控制器"""
    
    def __init__(self, model, view):
        self.model = model
        self.view = view
        
        # 连接信号
        self.view.add_button.clicked.connect(self.add_todo)
        self.view.list_widget.itemDoubleClicked.connect(self.toggle_todo)
        
        # 注册观察者
        self.model.add_observer(self.update_view)
    
    def add_todo(self):
        """添加任务"""
        text = self.view.get_input()
        if text:
            self.model.add_todo(text)
            self.view.clear_input()
    
    def toggle_todo(self, item):
        """切换任务状态"""
        index = self.view.list_widget.row(item)
        self.model.toggle_todo(index)
    
    def update_view(self):
        """更新视图"""
        todos = self.model.get_all()
        self.view.display_todos(todos)
    
    def run(self):
        """运行应用"""
        self.update_view()
        self.view.show()
```

### 使用

```python
if __name__ == "__main__":
    app = QApplication([])
    
    model = TodoModel()
    view = TodoView()
    controller = TodoController(model, view)
    
    controller.run()
    
    app.exec()
```

## 12.3 内存管理

### 常见内存泄漏

#### 1. 未断开的信号连接

```python
# ❌ 可能导致内存泄漏
class Widget(QWidget):
    def __init__(self):
        super().__init__()
        self.data = LargeObject()
        
        # 如果 widget 被销毁，但信号仍然连接
        some_signal.connect(self.handle_data)

# ✅ 及时断开
def closeEvent(self, event):
    some_signal.disconnect(self.handle_data)
    super().closeEvent(event)
```

#### 2. 循环引用

```python
# ❌ 循环引用
class Parent:
    def __init__(self):
        self.child = Child(self)

class Child:
    def __init__(self, parent):
        self.parent = parent  # 循环引用

# ✅ 使用弱引用
import weakref

class Child:
    def __init__(self, parent):
        self.parent_ref = weakref.ref(parent)
```

#### 3. 未及时清理的资源

```python
# ✅ 使用上下文管理器
class DatabaseConnection:
    def __enter__(self):
        self.conn = sqlite3.connect('db.db')
        return self.conn
    
    def __exit__(self, exc_type, exc_val, exc_tb):
        self.conn.close()

# 使用
with DatabaseConnection() as conn:
    # 自动关闭
    pass
```

## 12.4 性能优化技巧

### 1. 避免在 UI 线程执行耗时操作

```python
# ❌ 阻塞 UI
def on_button_clicked(self):
    result = heavy_computation()  # 耗时操作
    self.label.setText(str(result))

# ✅ 使用线程
def on_button_clicked(self):
    self.worker = WorkerThread()
    self.worker.result.connect(self.update_ui)
    self.worker.start()
```

### 2. 批量更新 UI

```python

    list_widget.addItem(item)  # 每次都重绘

# ✅ 批量更新
list_widget.setUpdatesEnabled(False)
for item in large_list:
    list_widget.addItem(item)
list_widget.setUpdatesEnabled(True)
```

### 3. 使用虚拟滚动

```python
# 对于大量数据，使用 QListView + QAbstractListModel
class LargeListModel(QAbstractListModel):
    def __init__(self, data):
        super().__init__()
        self._data = data
    
    def rowCount(self, parent=None):
        return len(self._data)
    
    def data(self, index, role):
        if role == Qt.DisplayRole:
            return self._data[index.row()]
```

## 12.5 常见陷阱与解决方案

### 陷阱 1：忘记调用父类构造函数

```python
# ❌ 错误
class MyWidget(QWidget):
    def __init__(self):
        pass  # 忘记调用 super().__init__()

# ✅ 正确
class MyWidget(QWidget):
    def __init__(self):
        super().__init__()
```

### 陷阱 2：在循环中创建大量组件

```python
# ❌ 性能差
for i in range(10000):
    label = QLabel(str(i))
    layout.addWidget(label)

# ✅ 使用虚拟列表
list_view = QListView()
model = QStringListModel(large_data_list)
list_view.setModel(model)
```

### 陷阱 3：阻塞事件循环

```python
# ❌ 阻塞
import time
time.sleep(5)  # UI 卡死 5 秒

# ✅ 使用定时器
QTimer.singleShot(5000, self.do_something)
```

### 陷阱 4：线程安全问题

```python
# ❌ 直接在线程中更新 UI
class Worker(QThread):
    def run(self):
        main_window.label.setText("Done")  # 危险！

# ✅ 使用信号
class Worker(QThread):
    finished = Signal(str)
    
    def run(self):
        self.finished.emit("Done")

worker.finished.connect(main_window.label.setText)
```

### 2. 异常处理

```python
import traceback
import sys

def exception_hook(exc_type, exc_value, exc_traceback):
    """全局异常处理"""
    if issubclass(exc_type, KeyboardInterrupt):
        sys.__excepthook__(exc_type, exc_value, exc_traceback)
        return
    
    logger.critical("未捕获的异常", exc_info=(exc_type, exc_value, exc_traceback))
    
    # 显示错误对话框
    error_msg = ''.join(traceback.format_exception(exc_type, exc_value, exc_traceback))
    QMessageBox.critical(None, "错误", f"发生错误:\n{error_msg}")

# 安装全局异常处理
sys.excepthook = exception_hook
```

def profile_function(func, *args):

    stats = pstats.Stats(profiler)

def calculate_total(items: list[dict]) -> float:
    """
    计算项目总价
    
    Args:
        items: 项目列表，每个项目包含 'price' 和 'quantity'
    
    Returns:
        总价
    
    Raises:
        ValueError: 如果价格为负数

    total = 0
    for item in items:
        if item['price'] < 0:
            raise ValueError("价格不能为负数")
        total += item['price'] * item['quantity']
    
    return total
