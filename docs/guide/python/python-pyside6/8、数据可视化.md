```python
from PySide6.QtWidgets import QApplication, QWidget, QVBoxLayout
from matplotlib.backends.backend_qt5agg import FigureCanvasQTAgg as FigureCanvas
from matplotlib.figure import Figure
import numpy as np

class MplCanvas(FigureCanvas):
    """Matplotlib 画布组件"""
    
    def __init__(self, parent=None, width=5, height=4, dpi=100):
        # 创建图形
        fig = Figure(figsize=(width, height), dpi=dpi)
        self.axes = fig.add_subplot(111)
        
        # 初始化父类
        super().__init__(fig)

class BasicPlot(QWidget):
    def __init__(self):
        super().__init__()
        self.setWindowTitle("Matplotlib 基本图表")
        self.setGeometry(100, 100, 800, 600)
        
        # 创建画布
        self.canvas = MplCanvas(self, width=8, height=6)
        
        # 绘制数据
        x = np.linspace(0, 10, 100)
        y = np.sin(x)
        
        self.canvas.axes.plot(x, y, 'b-', linewidth=2, label='sin(x)')
        self.canvas.axes.set_title('正弦波', fontsize=16)
        self.canvas.axes.set_xlabel('X轴', fontsize=12)
        self.canvas.axes.set_ylabel('Y轴', fontsize=12)
        self.canvas.axes.legend()
        self.canvas.axes.grid(True, alpha=0.3)
        
        # 刷新画布
        self.canvas.draw()
        
        # 布局
        layout = QVBoxLayout()
        layout.addWidget(self.canvas)
        self.setLayout(layout)

if __name__ == "__main__":
    app = QApplication([])
    window = BasicPlot()
    window.show()
    app.exec()
```

```python
    def create_line_plot(self):
        """折线图"""

        y1 = np.sin(x)
        y2 = np.cos(x)
        
        canvas.axes.plot(x, y1, 'r-', label='sin(x)')
        canvas.axes.plot(x, y2, 'b--', label='cos(x)')
        canvas.axes.set_title('多条折线')
        canvas.axes.legend()
        canvas.axes.grid(True)
        canvas.draw()
        
        layout = QVBoxLayout()
        layout.addWidget(canvas)
        widget.setLayout(layout)
        return widget
    
    def create_bar_chart(self):
        """柱状图"""
        widget = QWidget()
        canvas = MplCanvas(widget, width=8, height=5)
        
        categories = ['Python', 'Java', 'C++', 'JavaScript', 'Go']

        colors = ['#FF6B6B', '#4ECDC4', '#45B7D1', '#FFA07A', '#98D8C8']

        canvas.axes.set_ylabel('评分')

        for bar, value in zip(bars, values):

            canvas.axes.text(bar.get_x() + bar.get_width()/2., height,
                           f'{value}', ha='center', va='bottom')
        
        canvas.draw()
        
        layout = QVBoxLayout()
        layout.addWidget(canvas)
        widget.setLayout(layout)
        return widget

        scatter = canvas.axes.scatter(x, y, c=colors, s=sizes, alpha=0.6)

        canvas.axes.set_xlabel('X')
        canvas.axes.set_ylabel('Y')
        plt.colorbar(scatter)

        layout.addWidget(canvas)

    def create_pie_chart(self):
```

## 8.3 实时数据更新

### 动态更新图表

```python
from PySide6.QtCore import QTimer

class RealtimePlot(QWidget):
    def __init__(self):
        super().__init__()
        self.setWindowTitle("实时数据图表")
        self.setGeometry(100, 100, 800, 600)
        
        self.canvas = MplCanvas(self, width=8, height=6)
        
        # 数据存储
        self.x_data = []
        self.y_data = []
        self.max_points = 100
        
        # 初始化图表
        self.line, = self.canvas.axes.plot([], [], 'r-', linewidth=2)
        self.canvas.axes.set_ylim(-1.5, 1.5)
        self.canvas.axes.set_xlim(0, self.max_points)
        self.canvas.axes.set_title('实时数据')
        self.canvas.axes.grid(True)
        
        layout = QVBoxLayout()
        layout.addWidget(self.canvas)
        self.setLayout(layout)
        
        # 定时器
        self.timer = QTimer()
        self.timer.timeout.connect(self.update_plot)
        self.timer.start(100)  # 100ms 更新一次
    
    def update_plot(self):
        """更新图表"""
        # 添加新数据
        self.x_data.append(len(self.x_data))
        self.y_data.append(np.sin(len(self.x_data) * 0.1) + np.random.randn() * 0.1)
        
        # 限制数据点数
        if len(self.x_data) > self.max_points:
            self.x_data = self.x_data[-self.max_points:]
            self.y_data = self.y_data[-self.max_points:]
        
        # 更新线条
        self.line.set_data(self.x_data, self.y_data)
        
        # 调整视图
        self.canvas.axes.set_xlim(0, len(self.x_data))
        
        # 重绘
        self.canvas.draw()
```

```python
import psutil

class CPUMonitor(QWidget):
    def __init__(self):
        super().__init__()
        self.setWindowTitle("CPU 使用率监控")
        
        self.canvas = MplCanvas(self, width=10, height=4)
        self.cpu_data = []
        
        self.line, = self.canvas.axes.plot([], [], 'g-', linewidth=2)
        self.canvas.axes.set_ylim(0, 100)
        self.canvas.axes.set_title('CPU 使用率')
        self.canvas.axes.set_ylabel('使用率 (%)')
        self.canvas.axes.grid(True)
        
        layout = QVBoxLayout()
        layout.addWidget(self.canvas)
        self.setLayout(layout)
        
        self.timer = QTimer()
        self.timer.timeout.connect(self.update_cpu)
        self.timer.start(1000)
    
    def update_cpu(self):
        cpu_percent = psutil.cpu_percent()
        self.cpu_data.append(cpu_percent)
        
        if len(self.cpu_data) > 60:
            self.cpu_data = self.cpu_data[-60:]
        
        x_data = list(range(len(self.cpu_data)))
        self.line.set_data(x_data, self.cpu_data)
        self.canvas.axes.set_xlim(0, len(x_data))
        self.canvas.draw()
```

## 8.4 图表交互

### 鼠标点击事件

```python
from matplotlib.backend_bases import MouseButton

class InteractivePlot(QWidget):
    def __init__(self):
        super().__init__()
        
        self.canvas = MplCanvas(self, width=8, height=6)
        
        # 生成数据
        x = np.linspace(0, 10, 100)
        y = np.sin(x)
        
        self.canvas.axes.plot(x, y, 'b-')
        self.canvas.axes.set_title('点击图表查看坐标')
        self.canvas.axes.grid(True)
        
        # 绑定点击事件
        self.canvas.mpl_connect('button_press_event', self.on_click)
        
        # 显示坐标的标签
        self.coord_label = QLabel("点击图表查看坐标")
        
        layout = QVBoxLayout()
        layout.addWidget(self.canvas)
        layout.addWidget(self.coord_label)
        self.setLayout(layout)
    
    def on_click(self, event):
        """处理点击事件"""
        if event.inaxes is not None:
            x, y = event.xdata, event.ydata
            self.coord_label.setText(f"坐标: ({x:.2f}, {y:.2f})")
            
            # 在图上标记
            self.canvas.axes.plot(x, y, 'ro', markersize=10)
            self.canvas.draw()
```

## 综合示例：股票行情看板

```python
from PySide6.QtWidgets import (
    QApplication, QMainWindow, QWidget, QVBoxLayout,
    QHBoxLayout, QLabel, QComboBox, QPushButton
)
from PySide6.QtCore import QTimer, Qt
import numpy as np

class StockDashboard(QMainWindow):
    def __init__(self):
        super().__init__()
        self.setWindowTitle("📈 股票行情看板")
        self.setGeometry(100, 100, 1000, 700)
        
        central_widget = QWidget()
        self.setCentralWidget(central_widget)
        main_layout = QVBoxLayout()
        
        # 标题栏
        header = QHBoxLayout()
        title = QLabel("股票行情")
        title.setStyleSheet("font-size: 20px; font-weight: bold;")
        header.addWidget(title)
        
        self.stock_combo = QComboBox()
        self.stock_combo.addItems(["AAPL", "GOOGL", "MSFT", "AMZN"])
        header.addWidget(self.stock_combo)
        
        refresh_btn = QPushButton("刷新")
        refresh_btn.clicked.connect(self.refresh_data)
        header.addWidget(refresh_btn)
        
        main_layout.addLayout(header)
        
        # 价格信息
        info_layout = QHBoxLayout()
        self.price_label = QLabel("价格: --")
        self.price_label.setStyleSheet("font-size: 24px; color: #2196F3;")
        info_layout.addWidget(self.price_label)
        
        self.change_label = QLabel("涨跌: --")
        info_layout.addWidget(self.change_label)
        
        main_layout.addLayout(info_layout)
        
        # 图表
        self.canvas = MplCanvas(self, width=10, height=5)
        main_layout.addWidget(self.canvas)
        
        central_widget.setLayout(main_layout)
        
        # 初始化
        self.price_history = []
        self.update_stock_data()
        
        # 自动刷新
        self.timer = QTimer()
        self.timer.timeout.connect(self.update_stock_data)
        self.timer.start(5000)
    
    def update_stock_data(self):
        """更新股票数据"""
        # 模拟股票价格
        base_price = 150
        new_price = base_price + np.random.randn() * 5
        self.price_history.append(new_price)
        
        if len(self.price_history) > 50:
            self.price_history = self.price_history[-50:]
        
        # 更新标签
        self.price_label.setText(f"价格: ${new_price:.2f}")
        
        change = new_price - base_price
        change_percent = (change / base_price) * 100
        color = "green" if change >= 0 else "red"
        sign = "+" if change >= 0 else ""
        self.change_label.setText(f"涨跌: {sign}{change:.2f} ({sign}{change_percent:.2f}%)")
        self.change_label.setStyleSheet(f"font-size: 18px; color: {color};")
        
        # 更新图表
        self.canvas.axes.clear()
        self.canvas.axes.plot(self.price_history, 'b-', linewidth=2)
        self.canvas.axes.set_title(f'{self.stock_combo.currentText()} 价格走势')
        self.canvas.axes.set_xlabel('时间')
        self.canvas.axes.set_ylabel('价格 ($)')
        self.canvas.axes.grid(True)
        self.canvas.draw()
    
    def refresh_data(self):
        self.price_history = []
        self.update_stock_data()

if __name__ == "__main__":
    app = QApplication([])
    window = StockDashboard()
    window.show()
    app.exec()
```
