# 第五章：事件处理

本章将深入学习 PySide6 的事件处理机制，包括鼠标事件、键盘事件、定时器事件等。事件处理是 GUI 编程的核心，让应用程序能够响应用户交互。

---

## 5.1 事件系统概述

### 什么是事件？

**事件**是用户或系统产生的动作信号，如点击鼠标、按下键盘、窗口大小改变等。PySide6 通过事件系统将这些信号传递给相应的组件进行处理。

### 事件处理流程

```
用户操作 → 系统产生事件 → Qt 事件循环 → 分发到对应组件 → 调用事件处理器
```

### 事件处理方式

**方式 1：重写事件处理方法**

```python
class CustomWidget(QWidget):
    def mousePressEvent(self, event):
        """处理鼠标按下事件"""
        print(f"鼠标按下: {event.pos()}")
```

**方式 2：使用信号与槽**

```python
button.clicked.connect(self.on_button_clicked)
```

**方式 3：安装事件过滤器**

```python
widget.installEventFilter(self)

def eventFilter(self, obj, event):
    if event.type() == QEvent.MouseButtonPress:
        print("鼠标按下")
    return False
```

```python
from PySide6.QtWidgets import QApplication, QWidget, QLabel, QVBoxLayout
from PySide6.QtCore import Qt
from PySide6.QtGui import QMouseEvent, QPainter, QColor

class MouseEventWidget(QWidget):
    def __init__(self):
        super().__init__()
        self.setWindowTitle("鼠标事件示例")
        self.setGeometry(100, 100, 600, 400)
        
        # 用于显示鼠标位置
        self.label = QLabel("移动鼠标查看坐标")
        self.label.setAlignment(Qt.AlignCenter)
        self.label.setStyleSheet("""
            font-size: 16px;
            padding: 20px;
            background-color: #f0f0f0;
        """)
        
        layout = QVBoxLayout()
        layout.addWidget(self.label)
        self.setLayout(layout)
        
        # 记录鼠标状态
        self.setMouseTracking(True)  # 即使不按按钮也追踪鼠标
    
    def mousePressEvent(self, event: QMouseEvent):
        """鼠标按下"""
        if event.button() == Qt.LeftButton:
            self.label.setText(f"左键按下: ({event.x()}, {event.y()})")
            self.label.setStyleSheet("""
                font-size: 16px;
                padding: 20px;
                background-color: #FFCDD2;
            """)
        elif event.button() == Qt.RightButton:
            self.label.setText(f"右键按下: ({event.x()}, {event.y()})")
            self.label.setStyleSheet("""
                font-size: 16px;
                padding: 20px;
                background-color: #C8E6C9;
            """)
        elif event.button() == Qt.MiddleButton:
            self.label.setText(f"中键按下: ({event.x()}, {event.y()})")
    
    def mouseReleaseEvent(self, event: QMouseEvent):
        """鼠标释放"""
        self.label.setText(f"鼠标释放: ({event.x()}, {event.y()})")
        self.label.setStyleSheet("""
            font-size: 16px;
            padding: 20px;
            background-color: #f0f0f0;
        """)
    
    def mouseMoveEvent(self, event: QMouseEvent):
        """鼠标移动"""
        self.label.setText(f"鼠标位置: ({event.x()}, {event.y()})")
    
    def mouseDoubleClickEvent(self, event: QMouseEvent):
        """双击"""
        self.label.setText(f"双击: ({event.x()}, {event.y()})")
        self.label.setStyleSheet("""
            font-size: 16px;
            padding: 20px;
            background-color: #BBDEFB;
        """)
    
    def enterEvent(self, event):
        """鼠标进入"""
        self.label.setText("鼠标进入窗口")
    
    def leaveEvent(self, event):
        """鼠标离开"""
        self.label.setText("鼠标离开窗口")

if __name__ == "__main__":
    app = QApplication([])
    window = MouseEventWidget()
    window.show()
    app.exec()
```

### 鼠标拖拽绘图

```python
class DrawingWidget(QWidget):
    """鼠标拖拽绘图"""
    
    def __init__(self):
        super().__init__()
        self.setWindowTitle("鼠标绘图")
        self.setGeometry(100, 100, 800, 600)
        
        self.drawing = False
        self.last_point = None
        self.lines = []  # 存储线条
        
        self.setMinimumSize(800, 600)
    
    def mousePressEvent(self, event: QMouseEvent):
        """开始绘图"""
        if event.button() == Qt.LeftButton:
            self.drawing = True
            self.last_point = event.pos()
    
    def mouseMoveEvent(self, event: QMouseEvent):
        """绘制线条"""
        if self.drawing and self.last_point:
            # 添加线条
            self.lines.append((self.last_point, event.pos()))
            self.last_point = event.pos()
            self.update()  # 触发重绘
    
    def mouseReleaseEvent(self, event: QMouseEvent):
        """结束绘图"""
        if event.button() == Qt.LeftButton:
            self.drawing = False
            self.last_point = None
    
    def paintEvent(self, event):
        """绘制所有线条"""
        painter = QPainter(self)
        painter.setRenderHint(QPainter.Antialiasing)
        
        pen = QColor("#2196F3")
        painter.setPen(pen)
        
        for start, end in self.lines:
            painter.drawLine(start, end)
        
        painter.end()
```

```python
from PySide6.QtWidgets import QApplication, QWidget, QLabel, QVBoxLayout
from PySide6.QtGui import QKeyEvent
from PySide6.QtCore import Qt

class KeyboardEventWidget(QWidget):
    def __init__(self):
        super().__init__()
        self.setWindowTitle("键盘事件示例")
        self.setGeometry(100, 100, 400, 300)
        
        self.label = QLabel("按下键盘按键")
        self.label.setAlignment(Qt.AlignCenter)
        self.label.setStyleSheet("""
            font-size: 18px;
            padding: 30px;
            background-color: #f0f0f0;
        """)
        
        layout = QVBoxLayout()
        layout.addWidget(self.label)
        self.setLayout(layout)
        
        # 设置焦点
        self.setFocusPolicy(Qt.StrongFocus)
        self.setFocus()
    
    def keyPressEvent(self, event: QKeyEvent):
        """按键按下"""
        key = event.key()
        
        # 特殊按键
        if key == Qt.Key_Escape:
            self.label.setText("按下了 ESC 键")
            self.close()
        elif key == Qt.Key_Return or key == Qt.Key_Enter:
            self.label.setText("按下了回车键")
        elif key == Qt.Key_Space:
            self.label.setText("按下了空格键")
        elif key == Qt.Key_Tab:
            self.label.setText("按下了 Tab 键")
        elif key == Qt.Key_Backspace:
            self.label.setText("按下了退格键")
        elif key == Qt.Key_Delete:
            self.label.setText("按下了 Delete 键")
        
        # 方向键
        elif key == Qt.Key_Up:
            self.label.setText("↑ 上")
        elif key == Qt.Key_Down:
            self.label.setText("↓ 下")
        elif key == Qt.Key_Left:
            self.label.setText("← 左")
        elif key == Qt.Key_Right:
            self.label.setText("→ 右")
        
        # 功能键
        elif key == Qt.Key_F1:
            self.label.setText("F1 - 帮助")
        elif key == Qt.Key_F5:
            self.label.setText("F5 - 刷新")
        
        # 字母和数字
        else:
            text = event.text()
            if text:
                self.label.setText(f"按下了: {text}")
            
            # 显示按键码
            modifiers = event.modifiers()
            mod_text = []
            if modifiers & Qt.ControlModifier:
                mod_text.append("Ctrl")
            if modifiers & Qt.ShiftModifier:
                mod_text.append("Shift")
            if modifiers & Qt.AltModifier:
                mod_text.append("Alt")
            
            if mod_text:
                self.label.setText(f"{' + '.join(mod_text)} + {text}")
    
    def keyReleaseEvent(self, event: QKeyEvent):
        """按键释放"""
        key = event.key()
        if key == Qt.Key_Escape:
            self.label.setText("ESC 键已释放")

if __name__ == "__main__":
    app = QApplication([])
    window = KeyboardEventWidget()
    window.show()
    app.exec()
```

```python
class GameControlWidget(QWidget):
    """简单的游戏控制"""
    
    def __init__(self):
        super().__init__()
        self.setWindowTitle("游戏控制")
        self.setGeometry(100, 100, 400, 400)
        
        self.player_x = 180
        self.player_y = 180
        self.speed = 10
        
        self.setFocusPolicy(Qt.StrongFocus)
        self.setFocus()
    
    def keyPressEvent(self, event: QKeyEvent):
        """控制玩家移动"""
        if event.key() == Qt.Key_W or event.key() == Qt.Key_Up:
            self.player_y = max(0, self.player_y - self.speed)
        elif event.key() == Qt.Key_S or event.key() == Qt.Key_Down:
            self.player_y = min(360, self.player_y + self.speed)
        elif event.key() == Qt.Key_A or event.key() == Qt.Key_Left:
            self.player_x = max(0, self.player_x - self.speed)
        elif event.key() == Qt.Key_D or event.key() == Qt.Key_Right:
            self.player_x = min(360, self.player_x + self.speed)
        
        self.update()
    
    def paintEvent(self, event):
        """绘制玩家"""
        painter = QPainter(self)
        painter.setRenderHint(QPainter.Antialiasing)
        
        # 绘制背景
        painter.fillRect(self.rect(), QColor("#2c3e50"))
        
        # 绘制玩家
```

```python
from PySide6.QtWidgets import QListWidget, QListWidgetItem
from PySide6.QtCore import Qt

class DragDropListWidget(QListWidget):
    """支持拖放的列表"""
    
    def __init__(self):
        super().__init__()
        self.setAcceptDrops(True)
        self.setDragEnabled(True)
        self.setDragDropMode(QAbstractItemView.InternalMove)
        
        # 添加示例数据
        for i in range(10):
            item = QListWidgetItem(f"项目 {i+1}")
            self.addItem(item)
    
    def dragEnterEvent(self, event):
        """拖动进入"""
        if event.mimeData().hasText():
            event.acceptProposedAction()
    
    def dropEvent(self, event):
        """放下"""
        if event.mimeData().hasText():
            text = event.mimeData().text()
            item = QListWidgetItem(text)
            self.addItem(item)
            event.acceptProposedAction()
```

```python
class FileDropWidget(QWidget):

        self.setGeometry(100, 100, 500, 400)

        self.label.setStyleSheet("""
            border: 3px dashed #ccc;
            padding: 50px;
            font-size: 18px;
            color: #666;
        """)
        
        layout = QVBoxLayout()
        layout.addWidget(self.label)
        self.setLayout(layout)
    
    def dragEnterEvent(self, event):
        """拖动进入"""
        if event.mimeData().hasUrls():
            event.acceptProposedAction()
            self.label.setStyleSheet("""
                border: 3px dashed #4CAF50;
                padding: 50px;
                font-size: 18px;
                color: #4CAF50;
                background-color: #E8F5E9;
            """)
        else:
            event.ignore()
    
    def dragLeaveEvent(self, event):
        """拖动离开"""
        self.label.setStyleSheet("""
            border: 3px dashed #ccc;
            padding: 50px;
            font-size: 18px;
            color: #666;
        """)
    
    def dropEvent(self, event):
        """放下文件"""
        urls = event.mimeData().urls()
        
        file_paths = []
        for url in urls:
            file_path = url.toLocalFile()
            file_paths.append(file_path)
        
        if file_paths:
            message = f"拖放了 {len(file_paths)} 个文件:\n\n"
            for path in file_paths[:5]:  # 只显示前5个
                message += f"• {path}\n"
            
            if len(file_paths) > 5:
                message += f"... 还有 {len(file_paths) - 5} 个文件"
            
            self.label.setText(message)
            self.label.setStyleSheet("""
                border: 3px solid #4CAF50;
                padding: 20px;
                font-size: 14px;
                color: #333;
                background-color: white;
            """)
        
        event.acceptProposedAction()
```

### 基本事件过滤器

```python
from PySide6.QtCore import QEvent

class EventFilterExample(QWidget):
    def __init__(self):
        super().__init__()
        self.setWindowTitle("事件过滤器示例")
        
        layout = QVBoxLayout()
        
        # 创建按钮
        self.button1 = QPushButton("按钮1")
        self.button2 = QPushButton("按钮2")
        self.button3 = QPushButton("按钮3")
        
        layout.addWidget(self.button1)
        layout.addWidget(self.button2)
        layout.addWidget(self.button3)
        
        self.setLayout(layout)
        
        # 安装事件过滤器
        self.button1.installEventFilter(self)
        self.button2.installEventFilter(self)
        self.button3.installEventFilter(self)
    
    def eventFilter(self, obj, event):
        """事件过滤器"""
        if event.type() == QEvent.MouseButtonPress:
            if obj == self.button1:
                print("按钮1被点击")
            elif obj == self.button2:
                print("按钮2被点击")
            elif obj == self.button3:
                print("按钮3被点击")
        
        # 返回 False 让事件继续传播
        # 返回 True 阻止事件传播
        return False
```

### 过滤特定事件

```python
def eventFilter(self, obj, event):
    """只过滤特定事件"""
    
    # 过滤键盘事件
    if event.type() == QEvent.KeyPress:
        key_event = QKeyEvent(event)
        if key_event.key() == Qt.Key_Escape:
            print("ESC 键被拦截")
            return True  # 阻止事件
    
    # 过滤鼠标滚轮
    if event.type() == QEvent.Wheel:
        print("鼠标滚轮事件")
        return True  # 禁用滚轮
    
    return False
```

```python
        save_btn = QPushButton("保存")
        save_btn.clicked.connect(self.save_image)
        layout.addWidget(save_btn)
        
        layout.addStretch()
        
        return layout
    
    def choose_color(self):
        """选择颜色"""
        color = QColorDialog.getColor(self.current_color, self)
        if color.isValid():
            self.current_color = color
            self.color_preview.setStyleSheet(
                f"background-color: {color.name()}; min-width: 30px;"
            )
    
    def change_brush_size(self, size):
        """改变笔刷大小"""
        self.brush_size = size
    
    def clear_canvas(self):
        """清空画布"""
        reply = QMessageBox.question(
            self, "确认", "确定要清空画布吗？",
            QMessageBox.Yes | QMessageBox.No
        )
        
        if reply == QMessageBox.Yes:
            self.pixmap.fill(Qt.white)
            self.canvas_label.setPixmap(self.pixmap)
    
    def save_image(self):
        """保存图片"""
        from PySide6.QtWidgets import QFileDialog
        
        file_path, _ = QFileDialog.getSaveFileName(
            self, "保存图片", "", "PNG 图片 (*.png)"
        )
        
        if file_path:
            if not file_path.endswith('.png'):
                file_path += '.png'
            self.pixmap.save(file_path)
            QMessageBox.information(self, "成功", "图片已保存！")
    
    def eventFilter(self, obj, event):
        """事件过滤器 - 处理画布事件"""
        if obj == self.canvas_label:
            if event.type() == QEvent.MouseButtonPress:
                if event.button() == Qt.LeftButton:
                    self.drawing = True
                    self.last_point = event.pos()
            
            elif event.type() == QEvent.MouseMove:
                if self.drawing and self.last_point:
                    painter = QPainter(self.pixmap)
                    pen = QPen(self.current_color, self.brush_size, 
                              Qt.SolidLine, Qt.RoundCap, Qt.RoundJoin)
                    painter.setPen(pen)
                    painter.drawLine(self.last_point, event.pos())
                    painter.end()
                    
                    self.last_point = event.pos()
                    self.canvas_label.setPixmap(self.pixmap)
            
            elif event.type() == QEvent.MouseButtonRelease:
                if event.button() == Qt.LeftButton:
                    self.drawing = False
                    self.last_point = None
        
        return super().eventFilter(obj, event)
```
