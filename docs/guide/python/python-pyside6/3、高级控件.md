## 3.1 QListWidget - 列表控件

### QListWidget 简介

**QListWidget** 是一个基于项（item-based）的列表控件，用于显示项目列表。每个项目可以是文本、图标或两者的组合。

**特点：**
- 支持单选和多选模式
- 可以添加图标和自定义数据
- 支持拖放操作
- 内置排序功能

### 基本用法

```python
from PySide6.QtWidgets import (
    QApplication, QWidget, QListWidget, 
    QListWidgetItem, QVBoxLayout
)
from PySide6.QtCore import Qt
from PySide6.QtGui import QIcon

class BasicListWidget(QWidget):
    def __init__(self):
        super().__init__()
        self.setWindowTitle("列表控件示例")
        self.setGeometry(100, 100, 400, 500)
        
        layout = QVBoxLayout()
        
        # 创建列表控件
        list_widget = QListWidget()
        
        # 方法1：直接添加文本
        list_widget.addItem("项目1")
        list_widget.addItem("项目2")
        list_widget.addItem("项目3")
        
        # 方法2：批量添加
        items = ["苹果", "香蕉", "橙子", "葡萄", "西瓜"]
        list_widget.addItems(items)
        
        # 方法3：使用 QListWidgetItem（更灵活）
        item = QListWidgetItem()
        item.setText("自定义项目")
        # item.setIcon(QIcon("icon.png"))
        item.setData(Qt.UserRole, 123)  # 存储自定义数据
        list_widget.addItem(item)
        
        layout.addWidget(list_widget)
        self.setLayout(layout)

if __name__ == "__main__":
    app = QApplication([])
    window = BasicListWidget()
    window.show()
    app.exec()
```

```python
class ListWidgetSignals(QWidget):
    def __init__(self):
        super().__init__()
        layout = QVBoxLayout()
        
        self.list_widget = QListWidget()
        self.list_widget.addItems([
            "Python", "Java", "C++", 
            "JavaScript", "Go", "Rust"
        ])
        
        # 信号连接
        self.list_widget.itemClicked.connect(self.on_item_clicked)
        self.list_widget.itemDoubleClicked.connect(self.on_item_double_clicked)
        self.list_widget.currentItemChanged.connect(self.on_current_changed)
        self.list_widget.itemSelectionChanged.connect(self.on_selection_changed)
        
        layout.addWidget(self.list_widget)
        
        # 显示信息的标签
        self.info_label = QLabel("选择一个项目")
        layout.addWidget(self.info_label)
        
        self.setLayout(layout)
    
    def on_item_clicked(self, item):
        """单击事件"""
        self.info_label.setText(f"单击: {item.text()}")
    
    def on_item_double_clicked(self, item):
        """双击事件"""
        self.info_label.setText(f"双击: {item.text()}")
    
    def on_current_changed(self, current, previous):
        """当前项改变"""
        if current:
            self.info_label.setText(
                f"当前: {current.text()}, "
                f"之前: {previous.text() if previous else '无'}"
            )
    
    def on_selection_changed(self):
        """选择改变"""
        selected = self.list_widget.selectedItems()
        texts = [item.text() for item in selected]
        self.info_label.setText(f"选中: {', '.join(texts)}")
```

### 完整示例：文件浏览器列表

```python
from pathlib import Path

class FileBrowserList(QWidget):
    def __init__(self):
        super().__init__()
        self.setWindowTitle("文件浏览器")
        self.setGeometry(100, 100, 500, 600)
        
        layout = QVBoxLayout()
        
        # 路径输入
        from PySide6.QtWidgets import QLineEdit, QPushButton, QHBoxLayout
        path_layout = QHBoxLayout()
        self.path_input = QLineEdit()
        self.path_input.setText(str(Path.home()))
        browse_btn = QPushButton("浏览")
        browse_btn.clicked.connect(self.browse_directory)
        path_layout.addWidget(self.path_input)
        path_layout.addWidget(browse_btn)
        layout.addLayout(path_layout)
        
        # 文件列表
        self.file_list = QListWidget()
        self.file_list.itemDoubleClicked.connect(self.open_item)
        layout.addWidget(self.file_list)
        
        # 状态栏
        self.status_label = QLabel("")
        layout.addWidget(self.status_label)
        
        self.setLayout(layout)
        
        # 加载初始目录
        self.load_directory(Path.home())
    
    def load_directory(self, path):
        """加载目录内容"""
        self.file_list.clear()
        
        try:
            path = Path(path)
            items = list(path.iterdir())
            
            # 文件夹优先
            dirs = sorted([p for p in items if p.is_dir()])
            files = sorted([p for p in items if p.is_file()])
            
            for dir_path in dirs:
                item = QListWidgetItem(f"📁 {dir_path.name}")
                item.setData(Qt.UserRole, str(dir_path))
                self.file_list.addItem(item)
            
            for file_path in files:
                item = QListWidgetItem(f"📄 {file_path.name}")
                item.setData(Qt.UserRole, str(file_path))
                self.file_list.addItem(item)
            
            self.status_label.setText(f"共 {len(items)} 个项目")
            
        except Exception as e:
            self.status_label.setText(f"错误: {str(e)}")
    
    def browse_directory(self):
        """浏览目录"""
        from PySide6.QtWidgets import QFileDialog
        directory = QFileDialog.getExistingDirectory(
            self, "选择目录", str(Path.home())
        )
        if directory:
            self.path_input.setText(directory)
            self.load_directory(directory)
    
    def open_item(self, item):
        """打开项目"""
        path = Path(item.data(Qt.UserRole))
        
        if path.is_dir():
            self.path_input.setText(str(path))
            self.load_directory(path)
        else:
            import subprocess
            subprocess.Popen(['start', str(path)], shell=True)

if __name__ == "__main__":
    app = QApplication([])
    window = FileBrowserList()
    window.show()
    app.exec()
```

## 3.2 QTableWidget - 表格控件

### QTableWidget 简介

**QTableWidget** 是一个基于项的表格控件，用于显示和编辑二维数据表。

**特点：**
- 行列结构清晰
- 支持单元格编辑
- 可以设置表头
- 支持排序和筛选

### 基本用法

```python
from PySide6.QtWidgets import (
    QApplication, QWidget, QTableWidget, 
    QTableWidgetItem, QVBoxLayout
)

class BasicTableWidget(QWidget):
    def __init__(self):
        super().__init__()
        self.setWindowTitle("表格控件示例")
        self.setGeometry(100, 100, 800, 500)
        
        layout = QVBoxLayout()
        
        # 创建表格
        table = QTableWidget()
        
        # 设置行列数
        table.setRowCount(5)
        table.setColumnCount(4)
        
        # 设置表头
        table.setHorizontalHeaderLabels(["姓名", "年龄", "城市", "职业"])
        table.setVerticalHeaderLabels(["行1", "行2", "行3", "行4", "行5"])
        
        # 填充数据
        data = [
            ["张三", "25", "北京", "工程师"],
            ["李四", "30", "上海", "设计师"],
            ["王五", "28", "广州", "产品经理"],
            ["赵六", "35", "深圳", "架构师"],
            ["孙七", "27", "杭州", "数据分析师"]
        ]
        
        for row, row_data in enumerate(data):
            for col, cell_data in enumerate(row_data):
                item = QTableWidgetItem(cell_data)
                table.setItem(row, col, item)
        
        layout.addWidget(table)
        self.setLayout(layout)

if __name__ == "__main__":
    app = QApplication([])
    window = BasicTableWidget()
    window.show()
    app.exec()
```

class TableWidgetSignals(QWidget):
    def __init__(self):
        super().__init__()
        layout = QVBoxLayout()
        
        self.table = QTableWidget(5, 3)
        self.table.setHorizontalHeaderLabels(["列1", "列2", "列3"])
        
        # 填充示例数据
        for row in range(5):
            for col in range(3):
                self.table.setItem(row, col, QTableWidgetItem(f"{row},{col}"))
        
        # 信号连接
        self.table.cellClicked.connect(self.on_cell_clicked)
        self.table.cellDoubleClicked.connect(self.on_cell_double_clicked)
        self.table.currentCellChanged.connect(self.on_current_cell_changed)
        self.table.itemChanged.connect(self.on_item_changed)
        
        layout.addWidget(self.table)
        
        self.info_label = QLabel("点击单元格")
        layout.addWidget(self.info_label)
        
        self.setLayout(layout)
    
    def on_cell_clicked(self, row, column):
        """单元格点击"""
        item = self.table.item(row, column)
        text = item.text() if item else ""
        self.info_label.setText(f"点击: 行{row}, 列{column}, 值:{text}")
    
    def on_cell_double_clicked(self, row, column):
        """单元格双击"""
        self.info_label.setText(f"双击: 行{row}, 列{column}")
    
    def on_current_cell_changed(self, row, column, previousRow, previousColumn):
        """当前单元格改变"""
        self.info_label.setText(
            f"当前: ({row}, {column}), "
            f"之前: ({previousRow}, {previousColumn})"
        )
    
    def on_item_changed(self, item):
        """单元格内容改变"""
        row = self.table.row(item)
        col = self.table.column(item)
        self.info_label.setText(f"修改: 行{row}, 列{col}, 新值:{item.text()}")

### 完整示例：学生成绩管理系统

```python
from PySide6.QtWidgets import (
    QApplication, QMainWindow, QWidget, QVBoxLayout,
    QHBoxLayout, QTableWidget, QTableWidgetItem,
    QPushButton, QLabel, QLineEdit, QMessageBox,

from PySide6.QtCore import Qt

class StudentGradeSystem(QMainWindow):
    def __init__(self):
        super().__init__()
        self.setWindowTitle("学生成绩管理系统")
        self.setGeometry(100, 100, 900, 600)
        
        central_widget = QWidget()
        self.setCentralWidget(central_widget)
        main_layout = QVBoxLayout()
        
        # 标题
        title = QLabel("📊 学生成绩管理")
        title.setAlignment(Qt.AlignCenter)
        title.setStyleSheet("font-size: 20px; font-weight: bold; padding: 10px;")
        main_layout.addWidget(title)
        
        # 操作按钮
        btn_layout = QHBoxLayout()
        
        add_btn = QPushButton("➕ 添加学生")
        add_btn.clicked.connect(self.add_student)
        btn_layout.addWidget(add_btn)
        
        delete_btn = QPushButton("➖ 删除选中")
        delete_btn.clicked.connect(self.delete_student)
        btn_layout.addWidget(delete_btn)
        
        calc_avg_btn = QPushButton("📈 计算平均分")
        calc_avg_btn.clicked.connect(self.calculate_average)
        btn_layout.addWidget(calc_avg_btn)
        
        btn_layout.addStretch()
        main_layout.addLayout(btn_layout)
        
        # 表格
        self.table = QTableWidget()
        self.table.setColumnCount(5)
        self.table.setHorizontalHeaderLabels([
            "学号", "姓名", "语文", "数学", "英语"
        ])
        
        # 设置列宽
        header = self.table.horizontalHeader()
        header.setSectionResizeMode(0, QHeaderView.Stretch)
        header.setSectionResizeMode(1, QHeaderView.Stretch)
        header.setSectionResizeMode(2, QHeaderView.Stretch)
        header.setSectionResizeMode(3, QHeaderView.Stretch)
        header.setSectionResizeMode(4, QHeaderView.Stretch)
        
        # 启用排序
        self.table.setSortingEnabled(True)
        
        # 整行选择
        self.table.setSelectionBehavior(QTableWidget.SelectRows)
        
        main_layout.addWidget(self.table)
        
        # 统计信息
        self.stats_label = QLabel("")
        self.stats_label.setStyleSheet("padding: 10px; font-size: 12px;")
        main_layout.addWidget(self.stats_label)
        
        central_widget.setLayout(main_layout)
        
        # 初始化示例数据
        self.init_sample_data()
    
    def init_sample_data(self):
        """初始化示例数据"""
        students = [
            ("001", "张三", "85", "92", "78"),
            ("002", "李四", "90", "88", "95"),
            ("003", "王五", "76", "85", "82"),
            ("004", "赵六", "92", "95", "90"),
            ("005", "孙七", "88", "79", "86"),
        ]
        
        self.table.setRowCount(len(students))
        
        for row, student in enumerate(students):
            for col, value in enumerate(student):
                item = QTableWidgetItem(value)
                if col >= 2:  # 成绩列居中对齐
                    item.setTextAlignment(Qt.AlignCenter)
                self.table.setItem(row, col, item)
        
        self.update_stats()
    
    def add_student(self):
        """添加学生"""
        row = self.table.rowCount()
        self.table.insertRow(row)
        
        # 添加空行
        for col in range(5):
            self.table.setItem(row, col, QTableWidgetItem(""))
        
        # 聚焦到第一列
        self.table.setCurrentCell(row, 0)
        self.table.editItem(self.table.item(row, 0))
    
    def delete_student(self):
        """删除选中的学生"""
        selected_rows = set()
        
        for item in self.table.selectedItems():
            selected_rows.add(item.row())
        
        if not selected_rows:
            QMessageBox.warning(self, "提示", "请先选择要删除的行！")
            return
        
        # 从后往前删除（避免索引变化）
        for row in sorted(selected_rows, reverse=True):
            self.table.removeRow(row)
        
        self.update_stats()
        QMessageBox.information(self, "成功", f"已删除 {len(selected_rows)} 条记录")
    
    def calculate_average(self):
        """计算平均分"""
        if self.table.rowCount() == 0:
            QMessageBox.warning(self, "提示", "没有数据！")
            return
        
        totals = [0, 0, 0]  # 语文、数学、英语总分
        count = self.table.rowCount()
        
        for row in range(count):
            for col in range(2, 5):  # 成绩列
                item = self.table.item(row, col)
                if item:
                    try:
                        score = float(item.text())
                        totals[col - 2] += score
                    except ValueError:
                        pass
        
        averages = [total / count if count > 0 else 0 for total in totals]
        
        message = f"""
平均分统计：

语文: {averages[0]:.2f}
数学: {averages[1]:.2f}
英语: {averages[2]:.2f}

总人数: {count}
        """
        
        QMessageBox.information(self, "平均分统计", message)
    
    def update_stats(self):
        """更新统计信息"""
        count = self.table.rowCount()
        self.stats_label.setText(f"总人数: {count}")

if __name__ == "__main__":
    app = QApplication([])
    
    app.setStyleSheet("""
        QMainWindow {
            background-color: #f5f5f5;
        }
        QPushButton {
            padding: 8px 15px;
            background-color: #4CAF50;
            color: white;
            border: none;
            border-radius: 4px;
        }
        QPushButton:hover {
            background-color: #45a049;
        }
    """)
    
    window = StudentGradeSystem()
    window.show()
    app.exec()

    def previous_page(self):
        current = self.stacked.currentIndex()
        if current > 0:
            self.stacked.setCurrentIndex(current - 1)
    
    def next_page(self):
        current = self.stacked.currentIndex()
        if current < self.stacked.count() - 1:
            self.stacked.setCurrentIndex(current + 1)
```

from PySide6.QtWidgets import (

class DataManager(QMainWindow):

    def __init__(self):
        super().__init__()

        self.setGeometry(100, 100, 1000, 700)
        
        # 创建中央部件
        central_widget = QWidget()

        tab_widget.addTab(table_tab, "📋 表格视图")
        
        # 树形视图标签页
        tree_tab = self.create_tree_view()
        tab_widget.addTab(tree_tab, "🌳 树形视图")
        
        # 列表视图标签页
        list_tab = self.create_list_view()
        tab_widget.addTab(list_tab, "📝 列表视图")
        
        main_layout.addWidget(tab_widget)
        
        central_widget.setLayout(main_layout)
    
    def create_table_view(self):
        """创建表格视图"""

        layout = QVBoxLayout()

        delete_btn = QPushButton("➖ 删除")

            table.setItem(i, 3, QTableWidgetItem("活跃" if i % 3 != 0 else " inactive"))

        QTreeWidgetItem(docs, ["笔记.txt", "10KB", "2024-01-02"])

        layout.addWidget(list_widget)
        widget.setLayout(layout)
        return widget

if __name__ == "__main__":
