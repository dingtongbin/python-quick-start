## 6.1 Qt 样式表（QSS）基础

### 什么是 QSS？

**Qt 样式表（QSS）** 是一种类似于 CSS 的样式语言，用于定制 Qt 组件的外观。它支持选择器、属性、伪状态等概念。

### 基本语法

```python
# 设置全局样式
app.setStyleSheet("""
    QWidget {
        background-color: #f0f0f0;
        font-family: "Microsoft YaHei";
        font-size: 12px;
    }
    
    QPushButton {
        background-color: #4CAF50;
        color: white;
        padding: 8px 16px;
        border: none;
        border-radius: 4px;
    }
    
    QPushButton:hover {
        background-color: #45a049;
    }
""")

# 设置单个组件样式
button.setStyleSheet("""
    QPushButton {
        background-color: blue;
        color: white;
    }
""")
```

### 选择器类型

```python
"""
/* 通用选择器 - 所有 QWidget */
QWidget {
    background-color: white;
}

/* 类选择器 - 特定类型的组件 */
QPushButton {
    background-color: green;
}

/* ID 选择器 - 通过对象名 */
#myButton {
    background-color: red;
}

/* 属性选择器 */
QPushButton[flat="true"] {
    background-color: transparent;
}

/* 后代选择器 */
QDialog QPushButton {
    background-color: yellow;
}

/* 子选择器 */
QToolBar > QPushButton {
    background-color: orange;
}

/* 伪状态 */
QPushButton:hover {
    background-color: lightgreen;
}

QPushButton:pressed {
    background-color: darkgreen;
}

QPushButton:disabled {
    background-color: gray;
}
"""
```

### 主题切换器

```python
class ThemeSwitcher:
    """主题切换器"""
    
    THEMES = {
        'dark': DARK_THEME,
        'light': LIGHT_THEME,
    }
    
    @staticmethod
    def apply_theme(app, theme_name='light'):
        """应用主题"""
        if theme_name in ThemeSwitcher.THEMES:
            app.setStyleSheet(ThemeSwitcher.THEMES[theme_name])
            print(f"已切换到 {theme_name} 主题")
        else:
            print(f"未知主题: {theme_name}")

# 使用
class MainWindow(QMainWindow):
    def __init__(self):
        super().__init__()
        
        # 主题切换按钮
        dark_btn = QPushButton("深色主题")
        dark_btn.clicked.connect(lambda: ThemeSwitcher.apply_theme(app, 'dark'))
        
        light_btn = QPushButton("浅色主题")
        light_btn.clicked.connect(lambda: ThemeSwitcher.apply_theme(app, 'light'))
```

## 6.5 图标和资源文件

### 使用图标

```python
from PySide6.QtGui import QIcon

# 从文件加载
button.setIcon(QIcon("icon.png"))

# 设置图标大小
button.setIconSize(QSize(24, 24))

# 不同状态的图标
button.setIcon(QIcon.fromTheme("document-save"))
```

### 资源文件（QRC）

```xml
<!-- resources.qrc -->
<RCC>
    <qresource prefix="/icons">
        <file>save.png</file>
        <file>open.png</file>
        <file>exit.png</file>
    </qresource>
    <qresource prefix="/styles">
        <file>dark.css</file>
        <file>light.css</file>
    </qresource>
</RCC>
```

```bash
# 编译资源文件
pyside6-rcc resources.qrc -o resources_rc.py
```

```python
# 在代码中使用
from resources_rc import *

button.setIcon(QIcon(":/icons/save.png"))

# 加载样式文件
with open(":/styles/dark.css", "r") as f:
    app.setStyleSheet(f.read())
```

## 综合示例：主题管理器应用

```python
from PySide6.QtWidgets import (
    QApplication, QMainWindow, QWidget, QVBoxLayout,
    QHBoxLayout, QPushButton, QLabel, QComboBox,
    QGroupBox, QFormLayout
)
from PySide6.QtCore import Qt

class ThemeManagerApp(QMainWindow):
    """主题管理器应用"""
    
    def __init__(self):
        super().__init__()
        self.setWindowTitle("🎨 主题管理器")
        self.setGeometry(100, 100, 600, 500)
        
        central_widget = QWidget()
        self.setCentralWidget(central_widget)
        main_layout = QVBoxLayout()
        
        # 标题
        title = QLabel("主题管理器")
        title.setAlignment(Qt.AlignCenter)
        title.setStyleSheet("""
            font-size: 24px;
            font-weight: bold;
            padding: 20px;
            color: #2196F3;
        """)
        main_layout.addWidget(title)
        
        # 主题选择
        theme_group = QGroupBox("选择主题")
        theme_layout = QFormLayout()
        
        self.theme_combo = QComboBox()
        self.theme_combo.addItems(["浅色主题", "深色主题", "蓝色主题", "绿色主题"])
        self.theme_combo.currentTextChanged.connect(self.change_theme)
        theme_layout.addRow("主题:", self.theme_combo)
        
        theme_group.setLayout(theme_layout)
        main_layout.addWidget(theme_group)
        
        # 预览区域
        preview_group = QGroupBox("样式预览")
        preview_layout = QVBoxLayout()
        
        # 按钮预览
        btn_layout = QHBoxLayout()
        primary_btn = QPushButton("主要按钮")
        secondary_btn = QPushButton("次要按钮")
        danger_btn = QPushButton("危险按钮")
        btn_layout.addWidget(primary_btn)
        btn_layout.addWidget(secondary_btn)
        btn_layout.addWidget(danger_btn)
        preview_layout.addLayout(btn_layout)
        
        # 输入框预览
        input1 = QLineEdit()
        input1.setPlaceholderText("输入框预览")
        preview_layout.addWidget(input1)
        
        # 下拉框预览
        combo = QComboBox()
        combo.addItems(["选项1", "选项2", "选项3"])
        preview_layout.addWidget(combo)
        
        preview_group.setLayout(preview_layout)
        main_layout.addWidget(preview_group)
        
        main_layout.addStretch()
        central_widget.setLayout(main_layout)
        
        # 应用默认主题
        self.change_theme("浅色主题")
    
    def change_theme(self, theme_name):
        """切换主题"""
        themes = {
            "浅色主题": LIGHT_THEME,
            "深色主题": DARK_THEME,
            "蓝色主题": BLUE_THEME,
            "绿色主题": GREEN_THEME,
        }
        
        if theme_name in themes:
            app.setStyleSheet(themes[theme_name])

if __name__ == "__main__":
    app = QApplication([])
    window = ThemeManagerApp()
    window.show()
    app.exec()
```
