## 9.1 SQLite 数据库基础

### 为什么选择 SQLite？

✅ **无需服务器** - 单文件数据库
✅ **零配置** - 开箱即用
✅ **跨平台** - Windows、macOS、Linux
✅ **Python 内置** - sqlite3 模块
✅ **适合桌面应用** - 轻量级、高效

### 基本操作

```python
import sqlite3

class DatabaseHelper:
    def __init__(self, db_name="app.db"):
        self.db_name = db_name
        self.conn = None
    
    def connect(self):
        """连接数据库"""
        self.conn = sqlite3.connect(self.db_name)
        self.conn.row_factory = sqlite3.Row  # 字典式访问
        return self.conn
    
    def close(self):
        """关闭连接"""
        if self.conn:
            self.conn.close()
    
    def create_table(self):
        """创建表"""
        cursor = self.conn.cursor()
        cursor.execute("""
            CREATE TABLE IF NOT EXISTS users (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                name TEXT NOT NULL,
                age INTEGER,
                email TEXT UNIQUE,
                created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
            )
        """)
        self.conn.commit()
    
    def insert_user(self, name, age, email):
        """插入用户"""
        cursor = self.conn.cursor()
        cursor.execute(
            "INSERT INTO users (name, age, email) VALUES (?, ?, ?)",
            (name, age, email)
        )
        self.conn.commit()
        return cursor.lastrowid
    
    def get_all_users(self):
        """获取所有用户"""
        cursor = self.conn.cursor()
        cursor.execute("SELECT * FROM users ORDER BY id DESC")
        return cursor.fetchall()
    
    def get_user_by_id(self, user_id):
        """根据ID获取用户"""
        cursor = self.conn.cursor()
        cursor.execute("SELECT * FROM users WHERE id = ?", (user_id,))
        return cursor.fetchone()
    
    def update_user(self, user_id, name, age, email):
        """更新用户"""
        cursor = self.conn.cursor()
        cursor.execute(
            "UPDATE users SET name=?, age=?, email=? WHERE id=?",
            (name, age, email, user_id)
        )
        self.conn.commit()
    
    def delete_user(self, user_id):
        """删除用户"""
        cursor = self.conn.cursor()
        cursor.execute("DELETE FROM users WHERE id = ?", (user_id,))
        self.conn.commit()

# 使用示例
db = DatabaseHelper()
db.connect()
db.create_table()

# 插入数据
db.insert_user("张三", 25, "zhangsan@example.com")
db.insert_user("李四", 30, "lisi@example.com")

# 查询数据
users = db.get_all_users()
for user in users:
    print(f"{user['name']} - {user['email']}")

db.close()
```

## 9.2 Qt SQL 模块

### 使用 QSqlDatabase

```python
from PySide6.QtSql import QSqlDatabase, QSqlQuery

class QtDatabase:
    def __init__(self, db_name="app.db"):
        self.db = QSqlDatabase.addDatabase("QSQLITE")
        self.db.setDatabaseName(db_name)
    
    def open(self):
        """打开数据库"""
        if not self.db.open():
            print(f"数据库错误: {self.db.lastError().text()}")
            return False
        return True
    
    def close(self):
        """关闭数据库"""
        self.db.close()
    
    def execute_query(self, sql, params=None):
        """执行查询"""
        query = QSqlQuery()
        
        if params:
            query.prepare(sql)
            for key, value in params.items():
                query.bindValue(key, value)
        
        if not query.exec():
            print(f"查询错误: {query.lastError().text()}")
            return False
        
        return query

# 使用
qt_db = QtDatabase()
if qt_db.open():
    # 创建表
    qt_db.execute_query("""
        CREATE TABLE IF NOT EXISTS products (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            name TEXT,
            price REAL,
            stock INTEGER
        )
    """)
    
    # 插入数据
    qt_db.execute_query(
        "INSERT INTO products (name, price, stock) VALUES (:name, :price, :stock)",
        {':name': 'iPhone', ':price': 7999, ':stock': 100}
    )
    
    qt_db.close()
```

## 9.3 QSqlTableModel - 表格模型

### 基本用法

```python
from PySide6.QtWidgets import QApplication, QWidget, QVBoxLayout, QTableView
from PySide6.QtSql import QSqlTableModel, QSqlDatabase

class TableModelExample(QWidget):
    def __init__(self):
        super().__init__()
        self.setWindowTitle("QSqlTableModel 示例")
        
        # 初始化数据库
        db = QSqlDatabase.addDatabase("QSQLITE")
        db.setDatabaseName("products.db")
        db.open()
        
        # 创建模型
        self.model = QSqlTableModel()
        self.model.setTable("products")
        self.model.setEditStrategy(QSqlTableModel.OnFieldChange)
        self.model.select()
        
        # 设置表头
        self.model.setHeaderData(0, Qt.Horizontal, "ID")
        self.model.setHeaderData(1, Qt.Horizontal, "名称")
        self.model.setHeaderData(2, Qt.Horizontal, "价格")
        self.model.setHeaderData(3, Qt.Horizontal, "库存")
        
        # 创建视图
        self.view = QTableView()
        self.view.setModel(self.model)
        
        # 隐藏 ID 列
        # self.view.hideColumn(0)
        
        layout = QVBoxLayout()
        layout.addWidget(self.view)
        self.setLayout(layout)
```

### CRUD 操作

```python
class ProductManager(QWidget):
    def __init__(self):
        super().__init__()
        
        # 初始化模型
        self.model = QSqlTableModel()
        self.model.setTable("products")
        self.model.select()
        
        # UI
        layout = QVBoxLayout()
        
        # 表格视图
        self.view = QTableView()
        self.view.setModel(self.model)
        layout.addWidget(self.view)
        
        # 按钮
        btn_layout = QHBoxLayout()
        
        add_btn = QPushButton("添加")
        add_btn.clicked.connect(self.add_product)
        btn_layout.addWidget(add_btn)
        
        delete_btn = QPushButton("删除选中")
        delete_btn.clicked.connect(self.delete_selected)
        btn_layout.addWidget(delete_btn)
        
        save_btn = QPushButton("保存")
        save_btn.clicked.connect(self.save_changes)
        btn_layout.addWidget(save_btn)
        
        layout.addLayout(btn_layout)
        self.setLayout(layout)
    
    def add_product(self):
        """添加产品"""
        row = self.model.rowCount()
        self.model.insertRow(row)
        
        # 设置默认值
        self.model.setData(self.model.index(row, 1), "新产品")
        self.model.setData(self.model.index(row, 2), 0.0)
        self.model.setData(self.model.index(row, 3), 0)
    
    def delete_selected(self):
        """删除选中行"""
        indexes = self.view.selectionModel().selectedRows()
        
        if not indexes:
            QMessageBox.warning(self, "提示", "请先选择要删除的行")
            return
        
        reply = QMessageBox.question(
            self, "确认",
            f"确定要删除 {len(indexes)} 条记录吗？",
            QMessageBox.Yes | QMessageBox.No
        )
        
        if reply == QMessageBox.Yes:
            for index in sorted(indexes, reverse=True):
                self.model.removeRow(index.row())
            
            self.model.submitAll()
    
    def save_changes(self):
        """保存更改"""
        if self.model.submitAll():
            QMessageBox.information(self, "成功", "保存成功")
        else:
            QMessageBox.critical(self, "错误", "保存失败")
```

## 9.4 完整示例：联系人管理系统

```python
from PySide6.QtWidgets import (
    QApplication, QMainWindow, QWidget, QVBoxLayout,
    QHBoxLayout, QTableView, QPushButton, QLineEdit,
    QLabel, QMessageBox, QHeaderView, QFormLayout,
    QDialog, QDialogButtonBox
)
from PySide6.QtSql import QSqlDatabase, QSqlTableModel, QSqlQuery
from PySide6.QtCore import Qt
import sys

class ContactDialog(QDialog):
    """联系人编辑对话框"""
    
    def __init__(self, parent=None, contact_data=None):
        super().__init__(parent)
        self.setWindowTitle("编辑联系人")
        self.contact_data = contact_data or {}
        
        layout = QFormLayout()
        
        self.name_input = QLineEdit()
        self.name_input.setText(self.contact_data.get('name', ''))
        layout.addRow("姓名:", self.name_input)
        
        self.phone_input = QLineEdit()
        self.phone_input.setText(self.contact_data.get('phone', ''))
        layout.addRow("电话:", self.phone_input)
        
        self.email_input = QLineEdit()
        self.email_input.setText(self.contact_data.get('email', ''))
        layout.addRow("邮箱:", self.email_input)
        
        self.address_input = QLineEdit()
        self.address_input.setText(self.contact_data.get('address', ''))
        layout.addRow("地址:", self.address_input)
        
        buttons = QDialogButtonBox(
            QDialogButtonBox.Ok | QDialogButtonBox.Cancel
        )
        buttons.accepted.connect(self.accept)
        buttons.rejected.connect(self.reject)
        layout.addRow(buttons)
        
        self.setLayout(layout)
    
    def get_data(self):
        return {
            'name': self.name_input.text(),
            'phone': self.phone_input.text(),
            'email': self.email_input.text(),
            'address': self.address_input.text()
        }

class ContactManager(QMainWindow):
    def __init__(self):
        super().__init__()
        self.setWindowTitle("📇 联系人管理系统")
        self.setGeometry(100, 100, 900, 600)
        
        # 初始化数据库
        self.init_database()
        
        # 初始化模型
        self.model = QSqlTableModel()
        self.model.setTable("contacts")
        self.model.setEditStrategy(QSqlTableModel.OnManualSubmit)
        self.model.select()
        
        # 设置表头
        self.model.setHeaderData(0, Qt.Horizontal, "ID")
        self.model.setHeaderData(1, Qt.Horizontal, "姓名")
        self.model.setHeaderData(2, Qt.Horizontal, "电话")
        self.model.setHeaderData(3, Qt.Horizontal, "邮箱")
        self.model.setHeaderData(4, Qt.Horizontal, "地址")
        
        central_widget = QWidget()
        self.setCentralWidget(central_widget)
        main_layout = QVBoxLayout()
        
        # 搜索栏
        search_layout = QHBoxLayout()
        search_label = QLabel("搜索:")
        self.search_input = QLineEdit()
        self.search_input.setPlaceholderText("输入姓名或电话...")
        self.search_input.textChanged.connect(self.search_contacts)
        
        search_layout.addWidget(search_label)
        search_layout.addWidget(self.search_input)
        search_layout.addStretch()
        main_layout.addLayout(search_layout)
        
        # 表格
        self.view = QTableView()
        self.view.setModel(self.model)
        self.view.setSelectionBehavior(QTableView.SelectRows)
        self.view.horizontalHeader().setSectionResizeMode(QHeaderView.Stretch)
        self.view.hideColumn(0)  # 隐藏ID列
        main_layout.addWidget(self.view)
        
        # 按钮栏
        btn_layout = QHBoxLayout()
        
        add_btn = QPushButton("➕ 添加联系人")
        add_btn.clicked.connect(self.add_contact)
        btn_layout.addWidget(add_btn)
        
        edit_btn = QPushButton("✏️ 编辑")
        edit_btn.clicked.connect(self.edit_contact)
        btn_layout.addWidget(edit_btn)
        
        delete_btn = QPushButton("🗑️ 删除")
        delete_btn.clicked.connect(self.delete_contact)
        btn_layout.addWidget(delete_btn)
        
        btn_layout.addStretch()
        
        save_btn = QPushButton("💾 保存")
        save_btn.clicked.connect(self.save_changes)
        btn_layout.addWidget(save_btn)
        
        revert_btn = QPushButton("↩️ 撤销")
        revert_btn.clicked.connect(self.revert_changes)
        btn_layout.addWidget(revert_btn)
        
        main_layout.addLayout(btn_layout)
        
        central_widget.setLayout(main_layout)
        
        # 状态栏
        self.update_status()
    
    def init_database(self):
        """初始化数据库"""
        db = QSqlDatabase.addDatabase("QSQLITE")
        db.setDatabaseName("contacts.db")
        
        if not db.open():
            QMessageBox.critical(self, "错误", "无法打开数据库")
            return
        
        # 创建表
        query = QSqlQuery()
        query.exec("""
            CREATE TABLE IF NOT EXISTS contacts (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                name TEXT NOT NULL,
                phone TEXT,
                email TEXT,
                address TEXT
            )
        """)
    
    def add_contact(self):
        """添加联系人"""
        dialog = ContactDialog(self)
        
        if dialog.exec() == QDialog.Accepted:
            data = dialog.get_data()
            
            if not data['name']:
                QMessageBox.warning(self, "错误", "姓名不能为空")
                return
            
            row = self.model.rowCount()
            self.model.insertRow(row)
            self.model.setData(self.model.index(row, 1), data['name'])
            self.model.setData(self.model.index(row, 2), data['phone'])
            self.model.setData(self.model.index(row, 3), data['email'])
            self.model.setData(self.model.index(row, 4), data['address'])
            
            self.update_status()
    
    def edit_contact(self):
        """编辑联系人"""
        indexes = self.view.selectionModel().selectedRows()
        
        if not indexes:
            QMessageBox.warning(self, "提示", "请先选择要编辑的联系人")
            return
        
        row = indexes[0].row()
        
        # 获取当前数据
        contact_data = {
            'name': self.model.data(self.model.index(row, 1)),
            'phone': self.model.data(self.model.index(row, 2)),
            'email': self.model.data(self.model.index(row, 3)),
            'address': self.model.data(self.model.index(row, 4))
        }
        
        dialog = ContactDialog(self, contact_data)
        
        if dialog.exec() == QDialog.Accepted:
            data = dialog.get_data()
            
            self.model.setData(self.model.index(row, 1), data['name'])
            self.model.setData(self.model.index(row, 2), data['phone'])
            self.model.setData(self.model.index(row, 3), data['email'])
            self.model.setData(self.model.index(row, 4), data['address'])
    
    def delete_contact(self):
        """删除联系人"""
        indexes = self.view.selectionModel().selectedRows()
        
        if not indexes:
            QMessageBox.warning(self, "提示", "请先选择要删除的联系人")
            return
        
        reply = QMessageBox.question(
            self, "确认",
            f"确定要删除 {len(indexes)} 个联系人吗？",
            QMessageBox.Yes | QMessageBox.No
        )
        
        if reply == QMessageBox.Yes:
            for index in sorted(indexes, reverse=True):
                self.model.removeRow(index.row())
            
            self.update_status()
    
    def save_changes(self):
        """保存更改"""
        if self.model.submitAll():
            QMessageBox.information(self, "成功", "保存成功")
            self.update_status()
        else:
            QMessageBox.critical(self, "错误", "保存失败")
            self.model.revertAll()
    
    def revert_changes(self):
        """撤销更改"""
        self.model.revertAll()
    
    def search_contacts(self, text):
        """搜索联系人"""
        if not text:
            self.model.setFilter("")
            return
        
        filter_str = f"name LIKE '%{text}%' OR phone LIKE '%{text}%'"
        self.model.setFilter(filter_str)
    
    def update_status(self):
        """更新状态栏"""
        count = self.model.rowCount()
        self.statusBar().showMessage(f"共 {count} 个联系人")

if __name__ == "__main__":
    app = QApplication(sys.argv)
    
    app.setStyleSheet("""
        QMainWindow {
            background-color: #f5f5f5;
        }
        QPushButton {
            padding: 8px 15px;
            background-color: #4CAF50;
            color: white;
            border: none;
            border-radius: 4px;
        }
        QPushButton:hover {
            background-color: #45a049;
        }
        QTableView {
            background-color: white;
            gridline-color: #eee;
        }
        QHeaderView::section {
            background-color: #f5f5f5;
            padding: 8px;
            border: 1px solid #ddd;
        }
    """)
    
    window = ContactManager()
    window.show()
    sys.exit(app.exec())
```
