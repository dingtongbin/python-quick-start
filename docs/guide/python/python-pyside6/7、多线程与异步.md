# 第七章：多线程与异步

本章将深入学习 PySide6 的多线程编程，包括 QThread、线程间通信、进度条更新等。多线程是保持 GUI 响应流畅的关键技术。

---

## 7.1 为什么需要多线程？

### 单线程的问题

在 GUI 应用中，所有界面更新都在主线程（UI 线程）执行。如果在主线程执行耗时操作，会导致：

❌ **界面卡死** - 无法响应用户操作
❌ **窗口未响应** - 操作系统可能提示程序无响应
❌ **用户体验差** - 无法显示进度或取消操作

```python
# ❌ 错误示例 - 在主线程执行耗时操作
def on_button_clicked(self):
    # 模拟耗时任务（5秒）
    import time
    for i in range(50):
        time.sleep(0.1)  # 阻塞主线程！
        print(f"处理中... {i+1}%")
    
    # 这行代码要5秒后才执行
    self.label.setText("完成")
```

### 多线程的优势

✅ **界面保持响应** - 用户可以继续操作
✅ **可以显示进度** - 实时更新进度条
✅ **可以取消任务** - 用户随时中断
✅ **更好的用户体验** - 流畅的交互

## 7.2 QThread 基础

### 继承 QThread 方式

```python
from PySide6.QtCore import QThread, Signal
from PySide6.QtWidgets import (
    QApplication, QWidget, QVBoxLayout, 
    QPushButton, QLabel, QProgressBar
)
import time

class WorkerThread(QThread):
    """工作线程 - 继承 QThread"""
    
    # 定义信号
    progress = Signal(int)      # 进度信号
    result = Signal(str)        # 结果信号
    error = Signal(str)         # 错误信号
    
    def __init__(self, task_duration=5):
        super().__init__()
        self.task_duration = task_duration
        self._is_running = True
    
    def run(self):
        """线程执行的任务"""
        try:
            steps = 100
            for i in range(steps):
                if not self._is_running:
                    break
                
                # 模拟耗时任务
                time.sleep(self.task_duration / steps)
                
                # 发送进度信号
                self.progress.emit(i + 1)
            
            if self._is_running:
                self.result.emit("任务完成！")
        
        except Exception as e:
            self.error.emit(str(e))
    
    def stop(self):
        """停止任务"""
        self._is_running = False

class ThreadExample(QWidget):
    def __init__(self):
        super().__init__()
        self.setWindowTitle("QThread 示例")
        self.setGeometry(100, 100, 500, 300)
        
        layout = QVBoxLayout()
        
        # 状态标签
        self.status_label = QLabel("准备就绪")
        self.status_label.setStyleSheet("font-size: 16px; padding: 10px;")
        layout.addWidget(self.status_label)
        
        # 进度条
        self.progress_bar = QProgressBar()
        self.progress_bar.setRange(0, 100)
        layout.addWidget(self.progress_bar)
        
        # 按钮布局
        btn_layout = QHBoxLayout()
        
        self.start_btn = QPushButton("开始任务")
        self.start_btn.clicked.connect(self.start_task)
        btn_layout.addWidget(self.start_btn)
        
        self.stop_btn = QPushButton("停止")
        self.stop_btn.clicked.connect(self.stop_task)
        self.stop_btn.setEnabled(False)
        btn_layout.addWidget(self.stop_btn)
        
        layout.addLayout(btn_layout)
        
        self.setLayout(layout)
        
        self.worker = None
    
    def start_task(self):
        """启动任务"""
        self.worker = WorkerThread(task_duration=5)
        
        # 连接信号
        self.worker.progress.connect(self.update_progress)
        self.worker.result.connect(self.task_completed)
        self.worker.error.connect(self.task_error)
        self.worker.finished.connect(self.thread_finished)
        
        # 启动线程
        self.worker.start()
        
        # 更新 UI
        self.start_btn.setEnabled(False)
        self.stop_btn.setEnabled(True)
        self.status_label.setText("任务进行中...")
    
    def stop_task(self):
        """停止任务"""
        if self.worker:
            self.worker.stop()
            self.status_label.setText("正在停止...")
    
    def update_progress(self, value):
        """更新进度"""
        self.progress_bar.setValue(value)
        self.status_label.setText(f"进度: {value}%")
    
    def task_completed(self, message):
        """任务完成"""
        self.status_label.setText(message)
        QMessageBox.information(self, "成功", message)
    
    def task_error(self, error_msg):
        """任务错误"""
        self.status_label.setText(f"错误: {error_msg}")
        QMessageBox.critical(self, "错误", error_msg)
    
    def thread_finished(self):
        """线程结束"""
        self.start_btn.setEnabled(True)
        self.stop_btn.setEnabled(False)
        self.progress_bar.setValue(0)

if __name__ == "__main__":
    app = QApplication([])
    window = ThreadExample()
    window.show()
    app.exec()
```

## 7.3 moveToThread 模式（推荐）

这种方式更灵活，推荐使用！

```python
from PySide6.QtCore import QObject, QThread, Signal

class Worker(QObject):
    """工作对象 - 继承 QObject"""
    
    progress = Signal(int)
    finished = Signal(str)
    error = Signal(str)
    
    def __init__(self, task_id=None):
        super().__init__()
        self.task_id = task_id
        self._is_running = True
    
    def do_work(self):
        """执行任务"""
        try:
            for i in range(100):
                if not self._is_running:
                    self.finished.emit("任务已取消")
                    return
                
                # 模拟任务
                import time
                time.sleep(0.05)
                
                self.progress.emit(i + 1)
            
            self.finished.emit(f"任务 {self.task_id} 完成")
        
        except Exception as e:
            self.error.emit(str(e))
    
    def stop(self):
        self._is_running = False

# 使用示例
class MainWindow(QWidget):
    def __init__(self):
        super().__init__()
        
        layout = QVBoxLayout()
        
        self.label = QLabel("准备就绪")
        layout.addWidget(self.label)
        
        start_btn = QPushButton("开始")
        start_btn.clicked.connect(self.start_worker)
        layout.addWidget(start_btn)
        
        self.setLayout(layout)
        
        self.worker = None
        self.thread = None
    
    def start_worker(self):
        # 创建工作对象和线程
        self.worker = Worker(task_id="Task-001")
        self.thread = QThread()
        
        # 将工作对象移动到线程
        self.worker.moveToThread(self.thread)
        
        # 连接信号
        self.thread.started.connect(self.worker.do_work)
        self.worker.progress.connect(self.update_progress)
        self.worker.finished.connect(self.task_finished)
        self.worker.error.connect(self.task_error)
        
        # 清理
        self.worker.finished.connect(self.thread.quit)
        self.worker.finished.connect(self.worker.deleteLater)
        self.thread.finished.connect(self.thread.deleteLater)
        
        # 启动线程
        self.thread.start()
        self.label.setText("任务启动...")
    
    def update_progress(self, value):
        self.label.setText(f"进度: {value}%")
    
    def task_finished(self, message):
        self.label.setText(message)
    
    def task_error(self, error_msg):
        self.label.setText(f"错误: {error_msg}")
```

## 7.4 线程间通信

### 使用信号槽通信

```python
class DataProcessor(QObject):
    """数据处理器"""
    
    progress = Signal(int)
    data_ready = Signal(list)
    
    def process_data(self, data_list):
        """处理数据"""
        results = []
        total = len(data_list)
        
        for i, data in enumerate(data_list):
            # 处理数据
            result = self.heavy_computation(data)
            results.append(result)
            
            # 发送进度
            self.progress.emit(int((i + 1) / total * 100))
        
        # 发送结果
        self.data_ready.emit(results)
    
    def heavy_computation(self, data):
        """耗时计算"""
        import time
        time.sleep(0.1)
        return data * 2

# 主线程接收结果
class MainWidget(QWidget):
    def __init__(self):
        super().__init__()
        
        self.processor = DataProcessor()
        self.thread = QThread()
        self.processor.moveToThread(self.thread)
        
        # 连接信号
        self.thread.started.connect(
            lambda: self.processor.process_data([1, 2, 3, 4, 5])
        )
        self.processor.data_ready.connect(self.handle_results)
        self.processor.progress.connect(self.update_progress)
        
        self.processor.finished.connect(self.thread.quit)
        
        self.thread.start()
    
    def handle_results(self, results):
        """处理结果（在主线程执行）"""
        print(f"收到结果: {results}")
        # 更新 UI
        self.label.setText(f"结果: {results}")
```

### 使用队列通信

```python
from queue import Queue

class QueueWorker(QObject):
    def __init__(self):
        super().__init__()
        self.queue = Queue()
    
    def add_task(self, task):
        """添加任务"""
        self.queue.put(task)
    
    def run(self):
        """处理任务队列"""
        while True:
            task = self.queue.get()
            if task is None:
                break
            
            # 处理任务
            result = self.process(task)
            self.result_signal.emit(result)
```

## 7.5 实战：文件下载管理器

```python
from PySide6.QtWidgets import (
    QApplication, QMainWindow, QWidget, QVBoxLayout,
    QHBoxLayout, QPushButton, QProgressBar, QLabel,
    QListWidget, QListWidgetItem, QMessageBox
)
from PySide6.QtCore import QThread, Signal, Qt
import time
import random

class DownloadTask(QThread):
    """下载任务线程"""
    
    progress = Signal(int)
    finished = Signal(str)
    speed = Signal(float)
    
    def __init__(self, filename, size_mb):
        super().__init__()
        self.filename = filename
        self.size_mb = size_mb
        self._is_running = True
    
    def run(self):
        """模拟下载"""
        downloaded = 0
        total = self.size_mb * 1024 * 1024  # 转换为字节
        
        while downloaded < total and self._is_running:
            # 模拟下载速度
            chunk_size = random.randint(1024 * 100, 1024 * 500)
            downloaded += chunk_size
            
            # 计算进度
            percent = int(downloaded / total * 100)
            self.progress.emit(percent)
            
            # 计算速度（MB/s）
            speed = chunk_size / 1024 / 1024 * 10
            self.speed.emit(speed)
            
            time.sleep(0.1)
        
        if self._is_running:
            self.finished.emit(f"{self.filename} 下载完成")
    
    def stop(self):
        self._is_running = False

class DownloadItem(QWidget):
    """下载项组件"""
    
    def __init__(self, filename, size_mb):
        super().__init__()
        
        layout = QVBoxLayout()
        
        # 文件名
        name_label = QLabel(filename)
        name_label.setStyleSheet("font-weight: bold;")
        layout.addWidget(name_label)
        
        # 进度条
        self.progress_bar = QProgressBar()
        layout.addWidget(self.progress_bar)
        
        # 速度和状态
        info_layout = QHBoxLayout()
        self.speed_label = QLabel("速度: 0 MB/s")
        self.status_label = QLabel("等待中...")
        info_layout.addWidget(self.speed_label)
        info_layout.addWidget(self.status_label)
        layout.addLayout(info_layout)
        
        self.setLayout(layout)
        
        # 启动下载
        self.task = DownloadTask(filename, size_mb)
        self.task.progress.connect(self.progress_bar.setValue)
        self.task.speed.connect(self.update_speed)
        self.task.finished.connect(self.download_complete)
        self.task.start()
    
    def update_speed(self, speed):
        self.speed_label.setText(f"速度: {speed:.2f} MB/s")
    
    def download_complete(self, message):
        self.status_label.setText("✓ 完成")
        self.status_label.setStyleSheet("color: green;")
        QMessageBox.information(self, "完成", message)

class DownloadManager(QMainWindow):
    """下载管理器"""
    
    def __init__(self):
        super().__init__()
        self.setWindowTitle("📥 下载管理器")
        self.setGeometry(100, 100, 600, 500)
        
        central_widget = QWidget()
        self.setCentralWidget(central_widget)
        main_layout = QVBoxLayout()
        
        # 标题
        title = QLabel("下载管理器")
        title.setAlignment(Qt.AlignCenter)
        title.setStyleSheet("font-size: 20px; font-weight: bold; padding: 10px;")
        main_layout.addWidget(title)
        
        # 添加下载按钮
        add_btn = QPushButton("➕ 添加下载任务")
        add_btn.clicked.connect(self.add_download)
        main_layout.addWidget(add_btn)
        
        # 下载列表
        self.download_list = QListWidget()
        main_layout.addWidget(self.download_list)
        
        central_widget.setLayout(main_layout)
    
    def add_download(self):
        """添加下载任务"""
        from PySide6.QtWidgets import QInputDialog
        
        filename, ok = QInputDialog.getText(
            self, "添加下载", "输入文件名:"
        )
        
        if ok and filename:
            size, ok = QInputDialog.getInt(
                self, "文件大小", "输入文件大小(MB):",
                10, 1, 1000
            )
            
            if ok:
                # 创建下载项
                item = QListWidgetItem()
                item.setSizeHint(DownloadItem(filename, size).sizeHint())
                
                download_item = DownloadItem(filename, size)
                self.download_list.addItem(item)
                self.download_list.setItemWidget(item, download_item)

if __name__ == "__main__":
    app = QApplication([])
    
    app.setStyleSheet("""
        QMainWindow {
            background-color: #f5f5f5;
        }
        QPushButton {
            padding: 10px;
            background-color: #2196F3;
            color: white;
            border: none;
            border-radius: 4px;
        }
        QPushButton:hover {
            background-color: #1976D2;
        }
    """)
    
    window = DownloadManager()
    window.show()
    app.exec()
```
