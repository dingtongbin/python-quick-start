## 2.1 QWidget - 所有组件的基类

### QWidget 简介

**QWidget** 是 PySide6 中所有用户界面对象的基类。它是用户界面的基本构建块，可以接收鼠标、键盘和其他系统事件，并在屏幕上绘制自己。

**特点：**
- 所有可视组件都继承自 QWidget
- 可以作为容器包含其他组件
- 支持样式表自定义外观
- 提供几何管理（位置、大小）

### 基本用法

```python
from PySide6.QtWidgets import QApplication, QWidget
import sys

class BasicWidget(QWidget):
    def __init__(self):
        super().__init__()
        
        # 设置窗口标题
        self.setWindowTitle("QWidget 示例")
        
        # 设置窗口位置和大小 (x, y, width, height)
        self.setGeometry(100, 100, 400, 300)
        
        # 设置最小/最大尺寸
        self.setMinimumSize(200, 150)
        self.setMaximumSize(800, 600)
        
        # 设置背景颜色（通过样式表）
        self.setStyleSheet("background-color: #f0f0f0;")

if __name__ == "__main__":
    app = QApplication(sys.argv)
    widget = BasicWidget()
    widget.show()
    sys.exit(app.exec())
```

### 自定义 QWidget

```python
from PySide6.QtWidgets import QWidget
from PySide6.QtGui import QPainter, QColor, QPen
from PySide6.QtCore import Qt

class CustomWidget(QWidget):
    """自定义绘制组件"""
    
    def __init__(self):
        super().__init__()
        self.setMinimumSize(400, 300)
        self.setWindowTitle("自定义绘制")
    
    def paintEvent(self, event):
        """绘制事件"""
        painter = QPainter(self)
        painter.setRenderHint(QPainter.Antialiasing)
        
        # 绘制矩形
        pen = QPen(QColor("#2196F3"), 2)
        painter.setPen(pen)
        painter.drawRect(50, 50, 100, 80)
        
        # 绘制圆形
        painter.setBrush(QColor("#4CAF50"))
        painter.drawEllipse(200, 50, 80, 80)
        
        # 绘制文本
        painter.setPen(QColor("#333"))
        painter.drawText(50, 200, "Hello, Custom Widget!")
        
        painter.end()

if __name__ == "__main__":
    app = QApplication([])
    widget = CustomWidget()
    widget.show()
    app.exec()
```

## 2.2 QMainWindow - 主窗口框架

### QMainWindow 简介

**QMainWindow** 是一个提供了完整应用程序窗口框架的顶级窗口，包括：
- 菜单栏（Menu Bar）
- 工具栏（Tool Bar）
- 状态栏（Status Bar）
- 中央部件（Central Widget）
- 停靠窗口（Dock Widgets）

### 基本结构

```python
from PySide6.QtWidgets import (
    QApplication, 
    QMainWindow, 
    QLabel,
    QMenuBar,
    QToolBar,
    QStatusBar
)
import sys

class MainWindow(QMainWindow):
    def __init__(self):
        super().__init__()
        
        # 设置窗口属性
        self.setWindowTitle("QMainWindow 示例")
        self.setGeometry(100, 100, 800, 600)
        
        # 1. 创建中央部件
        central_widget = QLabel("这是中央部件")
        central_widget.setAlignment(Qt.AlignCenter)
        self.setCentralWidget(central_widget)
        
        # 2. 创建菜单栏
        self.create_menu_bar()
        
        # 3. 创建工具栏
        self.create_tool_bar()
        
        # 4. 创建状态栏
        self.create_status_bar()
    
    def create_menu_bar(self):
        """创建菜单栏"""
        menubar = self.menuBar()
        
        # 文件菜单
        file_menu = menubar.addMenu("文件(&F)")
        
        # 添加动作
        new_action = file_menu.addAction("新建(&N)")
        new_action.setShortcut("Ctrl+N")
        new_action.triggered.connect(self.on_new)
        
        open_action = file_menu.addAction("打开(&O)")
        open_action.setShortcut("Ctrl+O")
        
        file_menu.addSeparator()  # 分隔线
        
        exit_action = file_menu.addAction("退出(&X)")
        exit_action.setShortcut("Ctrl+Q")
        exit_action.triggered.connect(self.close)
        
        # 编辑菜单
        edit_menu = menubar.addMenu("编辑(&E)")
        edit_menu.addAction("复制(&C)")
        edit_menu.addAction("粘贴(&V)")
    
    def create_tool_bar(self):
        """创建工具栏"""
        toolbar = QToolBar("主工具栏")
        self.addToolBar(toolbar)
        
        # 添加工具按钮
        toolbar.addAction("新建")
        toolbar.addAction("打开")
        toolbar.addSeparator()
        toolbar.addAction("保存")
    
    def create_status_bar(self):
        """创建状态栏"""
        statusbar = QStatusBar()
        self.setStatusBar(statusbar)
        statusbar.showMessage("就绪", 3000)  # 显示3秒
    
    def on_new(self):
        """新建文件"""
        self.statusBar().showMessage("创建新文件")

if __name__ == "__main__":
    app = QApplication(sys.argv)
    window = MainWindow()
    window.show()
    sys.exit(app.exec())
```

## 2.3 布局管理器

布局管理器用于自动管理组件的位置和大小，使界面能够适应不同窗口大小。

### QVBoxLayout - 垂直布局

```python
from PySide6.QtWidgets import (
    QApplication, QWidget, QPushButton, 
    QLabel, QVBoxLayout
)

class VerticalLayoutExample(QWidget):
    def __init__(self):
        super().__init__()
        self.setWindowTitle("垂直布局示例")
        
        # 创建垂直布局
        layout = QVBoxLayout()
        
        # 添加组件
        layout.addWidget(QLabel("标签1"))
        layout.addWidget(QPushButton("按钮1"))
        layout.addWidget(QPushButton("按钮2"))
        layout.addWidget(QPushButton("按钮3"))
        
        # 添加弹性空间（推到顶部或底部）
        layout.addStretch()  # 在底部添加弹性空间
        
        # 设置布局
        self.setLayout(layout)

if __name__ == "__main__":
    app = QApplication([])
    window = VerticalLayoutExample()
    window.show()
    app.exec()
```

**效果：**
```
┌─────────────┐
│ 标签1       │
│ 按钮1       │
│ 按钮2       │
│ 按钮3       │
│             │  ← 弹性空间
│             │
└─────────────┘
```

### 布局嵌套

实际应用中经常需要嵌套使用多种布局：

```python
class NestedLayoutExample(QWidget):
    def __init__(self):
        super().__init__()
        self.setWindowTitle("嵌套布局示例")
        
        # 主垂直布局
        main_layout = QVBoxLayout()
        
        # 顶部水平布局
        top_layout = QHBoxLayout()
        top_layout.addWidget(QLabel("用户名:"))
        top_layout.addWidget(QLineEdit())
        main_layout.addLayout(top_layout)
        
        # 中间网格布局
        grid_layout = QGridLayout()
        for i in range(3):
            for j in range(3):
                btn = QPushButton(f"Btn {i},{j}")
                grid_layout.addWidget(btn, i, j)
        main_layout.addLayout(grid_layout)
        
        # 底部按钮
        bottom_layout = QHBoxLayout()
        bottom_layout.addStretch()  # 左侧弹性空间
        bottom_layout.addWidget(QPushButton("取消"))
        bottom_layout.addWidget(QPushButton("确定"))
        main_layout.addLayout(bottom_layout)
        
        self.setLayout(main_layout)
```

        input7 = QLineEdit()
        input7.setMaxLength(20)

        input8 = QLineEdit()
        input8.setClearButtonEnabled(True)
        layout.addRow("可清除:", input8)

        input9 = QLineEdit()
        input9.textChanged.connect(self.on_text_changed)
        input9.returnPressed.connect(self.on_return_pressed)
        input9.editingFinished.connect(self.on_editing_finished)
        layout.addRow("信号测试:", input9)
        
        self.setLayout(layout)
    
    def on_text_changed(self, text):
        print(f"文本改变: {text}")
    
    def on_return_pressed(self):
        print("回车键按下")
    
    def on_editing_finished(self):
        print("编辑完成")

## 综合示例：用户注册表单

```python
from PySide6.QtWidgets import (
    QApplication, QMainWindow, QWidget, QVBoxLayout,
    QFormLayout, QLineEdit, QComboBox, QRadioButton,
    QCheckBox, QPushButton, QLabel, QButtonGroup,
    QMessageBox
)
from PySide6.QtCore import Qt

class RegistrationForm(QMainWindow):
    def __init__(self):
        super().__init__()
        self.setWindowTitle("用户注册")
        self.setGeometry(100, 100, 400, 500)
        
        central_widget = QWidget()
        self.setCentralWidget(central_widget)
        main_layout = QVBoxLayout()
        
        # 标题
        title = QLabel("📝 用户注册")
        title.setAlignment(Qt.AlignCenter)
        title.setStyleSheet("font-size: 20px; font-weight: bold; padding: 10px;")
        main_layout.addWidget(title)
        
        # 表单区域
        form_widget = QWidget()
        form_layout = QFormLayout()
        
        # 用户名
        self.username_input = QLineEdit()
        self.username_input.setPlaceholderText("请输入用户名")
        form_layout.addRow("用户名:", self.username_input)
        
        # 邮箱
        self.email_input = QLineEdit()
        self.email_input.setPlaceholderText("example@email.com")
        form_layout.addRow("邮箱:", self.email_input)
        
        # 密码
        self.password_input = QLineEdit()
        self.password_input.setEchoMode(QLineEdit.Password)
        form_layout.addRow("密码:", self.password_input)
        
        # 性别
        gender_widget = QWidget()
        gender_layout = QVBoxLayout()
        self.radio_male = QRadioButton("男")
        self.radio_female = QRadioButton("女")
        self.radio_male.setChecked(True)
        gender_layout.addWidget(self.radio_male)
        gender_layout.addWidget(self.radio_female)
        gender_widget.setLayout(gender_layout)
        form_layout.addRow("性别:", gender_widget)
        
        # 城市
        self.city_combo = QComboBox()
        self.city_combo.addItems(["北京", "上海", "广州", "深圳", "杭州"])
        form_layout.addRow("城市:", self.city_combo)
        
        # 兴趣
        interests_widget = QWidget()
        interests_layout = QVBoxLayout()
        self.cb_reading = QCheckBox("阅读")
        self.cb_music = QCheckBox("音乐")
        self.cb_sports = QCheckBox("运动")
        interests_layout.addWidget(self.cb_reading)
        interests_layout.addWidget(self.cb_music)
        interests_layout.addWidget(self.cb_sports)
        interests_widget.setLayout(interests_layout)
        form_layout.addRow("兴趣:", interests_widget)
        
        form_widget.setLayout(form_layout)
        main_layout.addWidget(form_widget)
        
        # 按钮
        btn_layout = QVBoxLayout()
        self.submit_btn = QPushButton("注册")
        self.submit_btn.setStyleSheet("""
            QPushButton {
                background-color: #4CAF50;
                color: white;
                padding: 10px;
                border-radius: 5px;
                font-size: 14px;
            }
            QPushButton:hover {
                background-color: #45a049;
            }
        """)
        self.submit_btn.clicked.connect(self.submit_form)
        btn_layout.addWidget(self.submit_btn)
        
        main_layout.addLayout(btn_layout)
        
        central_widget.setLayout(main_layout)
    
    def submit_form(self):
        """提交表单"""
        username = self.username_input.text()
        email = self.email_input.text()
        password = self.password_input.text()
        
        # 简单验证
        if not username or not email or not password:
            QMessageBox.warning(self, "错误", "请填写所有必填字段！")
            return
        
        # 获取性别
        gender = "男" if self.radio_male.isChecked() else "女"
        
        # 获取城市
        city = self.city_combo.currentText()
        
        # 获取兴趣
        interests = []
        if self.cb_reading.isChecked():
            interests.append("阅读")
        if self.cb_music.isChecked():
            interests.append("音乐")
        if self.cb_sports.isChecked():
            interests.append("运动")
        
        # 显示结果
        message = f"""
注册成功！

用户名: {username}
邮箱: {email}
性别: {gender}
城市: {city}
兴趣: {', '.join(interests) if interests else '无'}
        """
        
        QMessageBox.information(self, "成功", message)
        print(message)

if __name__ == "__main__":
    app = QApplication([])
    
    # 全局样式
    app.setStyleSheet("""
        QMainWindow {

        }

            border-radius: 4px;
        }
        QLineEdit:focus {
            border-color: #4CAF50;
        }
    """)
    
    window = RegistrationForm()
    window.show()
    app.exec()
```
