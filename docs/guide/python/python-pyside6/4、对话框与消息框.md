本章将深入学习 PySide6 中的各种对话框和消息框组件。对话框是与用户进行交互的重要方式，用于显示信息、获取输入、选择文件等。

## 4.1 QMessageBox - 消息对话框

### QMessageBox 简介

**QMessageBox** 是用于显示简短消息的标准对话框，支持多种类型：信息、警告、错误、问题确认等。

### 基本用法

```python
from PySide6.QtWidgets import (
    QApplication, QWidget, QPushButton, 
    QVBoxLayout, QMessageBox
)

class MessageBoxExample(QWidget):
    def __init__(self):
        super().__init__()
        self.setWindowTitle("消息对话框示例")
        self.setGeometry(100, 100, 400, 300)
        
        layout = QVBoxLayout()
        
        # 信息框按钮
        info_btn = QPushButton("显示信息框")
        info_btn.clicked.connect(self.show_info)
        layout.addWidget(info_btn)
        
        # 警告框按钮
        warning_btn = QPushButton("显示警告框")
        warning_btn.clicked.connect(self.show_warning)
        layout.addWidget(warning_btn)
        
        # 错误框按钮
        error_btn = QPushButton("显示错误框")
        error_btn.clicked.connect(self.show_error)
        layout.addWidget(error_btn)
        
        # 问题框按钮
        question_btn = QPushButton("显示问题框")
        question_btn.clicked.connect(self.show_question)
        layout.addWidget(question_btn)
        
        self.setLayout(layout)
    
    def show_info(self):
        """显示信息框"""
        QMessageBox.information(
            self,                    # 父窗口
            "信息",                  # 标题
            "操作成功完成！",        # 消息内容
            QMessageBox.Ok          # 按钮
        )
    
    def show_warning(self):
        """显示警告框"""
        QMessageBox.warning(
            self,
            "警告",
            "这是一个警告消息！",
            QMessageBox.Ok
        )
    
    def show_error(self):
        """显示错误框"""
        QMessageBox.critical(
            self,
            "错误",
            "发生了一个错误！",
            QMessageBox.Ok
        )
    
    def show_question(self):
        """显示问题框"""
        reply = QMessageBox.question(
            self,
            "确认",
            "确定要执行此操作吗？",
            QMessageBox.Yes | QMessageBox.No,
            QMessageBox.No  # 默认按钮
        )
        
        if reply == QMessageBox.Yes:
            print("用户选择了是")
        else:
            print("用户选择了否")

if __name__ == "__main__":
    app = QApplication([])
    window = MessageBoxExample()
    window.show()
    app.exec()
```

```python
def show_custom_message(self):
```

## 4.2 QInputDialog - 输入对话框

**QInputDialog** 用于快速获取用户的单值输入，支持文本、整数、浮点数、下拉选择等。

### 文本输入

```python
from PySide6.QtWidgets import QInputDialog

def get_text_input(self):
    """获取文本输入"""
```

## 4.3 QFileDialog - 文件对话框

### QFileDialog 简介

**QFileDialog** 用于打开或保存文件，支持文件过滤、多选等功能。

### 打开单个文件

```python
from PySide6.QtWidgets import QFileDialog

def open_single_file(self):
    """打开单个文件"""
    file_path, _ = QFileDialog.getOpenFileName(
        self,                        # 父窗口
        "打开文件",                  # 标题
        "",                          # 默认目录（空表示当前目录）
        "所有文件 (*);;文本文件 (*.txt);;Python文件 (*.py);;图片文件 (*.png *.jpg *.jpeg)"  # 过滤器
    )
    
    if file_path:
        print(f"选择的文件: {file_path}")
        
        # 读取文件内容
        with open(file_path, 'r', encoding='utf-8') as f:
            content = f.read()
            print(f"文件内容:\n{content[:200]}...")  # 显示前200字符
```

### 完整示例：文本编辑器

```python
from PySide6.QtWidgets import (
    QApplication, QMainWindow, QTextEdit, 
    QMenuBar, QToolBar, QAction, QFileDialog,
    QMessageBox
)
from PySide6.QtGui import QIcon
import sys

class SimpleTextEditor(QMainWindow):
    """简单文本编辑器"""
    
    def __init__(self):
        super().__init__()
        self.setWindowTitle("简单文本编辑器")
        self.setGeometry(100, 100, 800, 600)
        
        self.current_file = None
        
        # 创建文本编辑器
        self.text_edit = QTextEdit()
        self.setCentralWidget(self.text_edit)
        
        # 创建菜单
        self.create_menu()
        
        # 创建工具栏
        self.create_toolbar()
        
        # 状态栏
        self.statusBar().showMessage("就绪")
    
    def create_menu(self):
        """创建菜单栏"""
        menubar = self.menuBar()
        
        # 文件菜单
        file_menu = menubar.addMenu("文件(&F)")
        
        # 新建
        new_action = QAction("新建(&N)", self)
        new_action.setShortcut("Ctrl+N")
        new_action.triggered.connect(self.new_file)
        file_menu.addAction(new_action)
        
        # 打开
        open_action = QAction("打开(&O)", self)
        open_action.setShortcut("Ctrl+O")
        open_action.triggered.connect(self.open_file)
        file_menu.addAction(open_action)
        
        # 保存
        save_action = QAction("保存(&S)", self)
        save_action.setShortcut("Ctrl+S")
        save_action.triggered.connect(self.save_file)
        file_menu.addAction(save_action)
        
        file_menu.addSeparator()
        
        # 退出
        exit_action = QAction("退出(&X)", self)
        exit_action.setShortcut("Ctrl+Q")
        exit_action.triggered.connect(self.close)
        file_menu.addAction(exit_action)
    
    def create_toolbar(self):
        """创建工具栏"""
        toolbar = QToolBar("主工具栏")
        self.addToolBar(toolbar)
        
        # 新建按钮
        new_btn = toolbar.addAction("📄 新建")
        new_btn.triggered.connect(self.new_file)
        
        # 打开按钮
        open_btn = toolbar.addAction("📂 打开")
        open_btn.triggered.connect(self.open_file)
        
        # 保存按钮
        save_btn = toolbar.addAction("💾 保存")
        save_btn.triggered.connect(self.save_file)
    
    def new_file(self):
        """新建文件"""
        if self.maybe_save():
            self.text_edit.clear()
            self.current_file = None
            self.setWindowTitle("简单文本编辑器 - 未命名")
            self.statusBar().showMessage("新建文件")
    
    def open_file(self):
        """打开文件"""
        if not self.maybe_save():
            return
        
        file_path, _ = QFileDialog.getOpenFileName(
            self,
            "打开文件",
            "",
            "文本文件 (*.txt);;所有文件 (*)"
        )
        
        if file_path:
            try:
                with open(file_path, 'r', encoding='utf-8') as f:
                    content = f.read()
                
                self.text_edit.setPlainText(content)
                self.current_file = file_path
                self.setWindowTitle(f"简单文本编辑器 - {file_path}")
                self.statusBar().showMessage(f"已打开: {file_path}")
                
            except Exception as e:
                QMessageBox.critical(self, "错误", f"无法打开文件:\n{str(e)}")
    
    def save_file(self):
        """保存文件"""
        if self.current_file:
            self.save_to_file(self.current_file)
        else:
            self.save_as()
    
    def save_as(self):
        """另存为"""
        file_path, _ = QFileDialog.getSaveFileName(
            self,
            "保存文件",
            "",
            "文本文件 (*.txt);;所有文件 (*)"
        )
        
        if file_path:
            if not file_path.endswith('.txt'):
                file_path += '.txt'
            self.save_to_file(file_path)
    
    def save_to_file(self, file_path):
        """保存到文件"""
        try:
            content = self.text_edit.toPlainText()
            with open(file_path, 'w', encoding='utf-8') as f:
                f.write(content)
            
            self.current_file = file_path
            self.setWindowTitle(f"简单文本编辑器 - {file_path}")
            self.statusBar().showMessage(f"已保存: {file_path}")
            
            QMessageBox.information(self, "成功", "文件已保存！")
            
        except Exception as e:
            QMessageBox.critical(self, "错误", f"无法保存文件:\n{str(e)}")
    
    def maybe_save(self):
        """检查是否需要保存"""
        if self.text_edit.document().isModified():
            reply = QMessageBox.question(
                self,
                "确认",
                "文件已修改，是否保存？",
                QMessageBox.Save | QMessageBox.Discard | QMessageBox.Cancel
            )
            
            if reply == QMessageBox.Save:
                self.save_file()
                return True
            elif reply == QMessageBox.Cancel:
                return False
        
        return True

if __name__ == "__main__":
    app = QApplication(sys.argv)
    editor = SimpleTextEditor()
    editor.show()
    sys.exit(app.exec())
```

```python
class LoginDialog(QDialog):
    """登录对话框"""
    
    def __init__(self, parent=None):
        super().__init__(parent)
        self.setWindowTitle("用户登录")
        self.init_ui()
    
    def init_ui(self):
        layout = QFormLayout()
        
        self.username_input = QLineEdit()
        self.password_input = QLineEdit()
        self.password_input.setEchoMode(QLineEdit.Password)
        
        layout.addRow("用户名:", self.username_input)
        layout.addRow("密码:", self.password_input)
        
        btn_layout = QHBoxLayout()
        login_btn = QPushButton("登录")
        login_btn.clicked.connect(self.validate_login)
        cancel_btn = QPushButton("取消")
        cancel_btn.clicked.connect(self.reject)
        
        btn_layout.addWidget(login_btn)
        btn_layout.addWidget(cancel_btn)
        layout.addRow(btn_layout)
        
        self.setLayout(layout)
    
    def validate_login(self):
        """验证登录"""
        username = self.username_input.text()
        password = self.password_input.text()
        
        # 简单验证（实际应该连接数据库）
        if username == "admin" and password == "123456":
            self.accept()
        else:
            QMessageBox.warning(self, "错误", "用户名或密码错误！")
    
    def get_credentials(self):
        """获取凭证"""
        return {
            'username': self.username_input.text(),
            'password': self.password_input.text()
        }

# 使用登录对话框
```

```python
from PySide6.QtWidgets import (
    QApplication, QMainWindow, QWidget, QVBoxLayout,
    QPushButton, QDialog, QTabWidget, QFormLayout,
    QLineEdit, QComboBox, QCheckBox, QSpinBox,
    QColorDialog, QFontDialog, QMessageBox,
    QHBoxLayout, QLabel
)
from PySide6.QtGui import QColor, QFont
from PySide6.QtCore import Qt

class SettingsDialog(QDialog):
    """设置对话框"""
    
    def __init__(self, parent=None):
        super().__init__(parent)
        self.setWindowTitle("设置")
        self.setGeometry(200, 200, 500, 400)
        
        # 保存原始设置
        self.original_settings = {}
        
        self.init_ui()
        self.load_settings()
    
    def init_ui(self):
        """初始化界面"""
        main_layout = QVBoxLayout()
        
        # 创建标签页
        tab_widget = QTabWidget()
        
        # 常规设置
        general_tab = self.create_general_tab()
        tab_widget.addTab(general_tab, "常规")
        
        # 外观设置
        appearance_tab = self.create_appearance_tab()
        tab_widget.addTab(appearance_tab, "外观")
        
        # 高级设置
        advanced_tab = self.create_advanced_tab()
        tab_widget.addTab(advanced_tab, "高级")
        
        main_layout.addWidget(tab_widget)
        
        # 按钮
        btn_layout = QHBoxLayout()
        btn_layout.addStretch()
        
        ok_btn = QPushButton("确定")
        ok_btn.clicked.connect(self.save_and_close)
        
        cancel_btn = QPushButton("取消")
        cancel_btn.clicked.connect(self.reject)
        
        apply_btn = QPushButton("应用")
        apply_btn.clicked.connect(self.apply_settings)
        
        btn_layout.addWidget(apply_btn)
        btn_layout.addWidget(cancel_btn)
        btn_layout.addWidget(ok_btn)
        
        main_layout.addLayout(btn_layout)
        
        self.setLayout(main_layout)
    
    def create_general_tab(self):
        """创建常规设置标签页"""
        widget = QWidget()
        layout = QFormLayout()
        
        # 语言选择
        self.language_combo = QComboBox()
        self.language_combo.addItems(["简体中文", "English", "日本語"])
        layout.addRow("语言:", self.language_combo)
        
        # 自动保存
        self.autosave_check = QCheckBox("启用自动保存")
        layout.addRow("", self.autosave_check)
        
        # 自动保存间隔
        self.autosave_interval = QSpinBox()
        self.autosave_interval.setRange(1, 60)
        self.autosave_interval.setValue(5)
        self.autosave_interval.setSuffix(" 分钟")
        layout.addRow("自动保存间隔:", self.autosave_interval)
        
        widget.setLayout(layout)
        return widget
    
    def create_appearance_tab(self):
        """创建外观设置标签页"""
        widget = QWidget()
        layout = QFormLayout()
        
        # 主题颜色
        color_btn = QPushButton("选择主题颜色")
        color_btn.clicked.connect(self.select_theme_color)
        self.color_preview = QLabel("████")
        self.color_preview.setStyleSheet("background-color: #4CAF50; min-width: 50px;")
        color_layout = QHBoxLayout()
        color_layout.addWidget(color_btn)
        color_layout.addWidget(self.color_preview)
        layout.addRow("主题颜色:", color_layout)
        
        # 字体
        font_btn = QPushButton("选择字体")
        font_btn.clicked.connect(self.select_font)
        self.font_label = QLabel("Arial, 12pt")
        font_layout = QHBoxLayout()
        font_layout.addWidget(font_btn)
        font_layout.addWidget(self.font_label)
        layout.addRow("字体:", font_layout)
        
        widget.setLayout(layout)
        return widget
    
    def create_advanced_tab(self):
        """创建高级设置标签页"""
        widget = QWidget()
        layout = QFormLayout()
        
        # 缓存大小
        cache_spin = QSpinBox()
        cache_spin.setRange(0, 10000)
        cache_spin.setValue(100)
        cache_spin.setSuffix(" MB")
        layout.addRow("缓存大小:", cache_spin)
        
        # 调试模式
        debug_check = QCheckBox("启用调试模式")
        layout.addRow("", debug_check)
        
        widget.setLayout(layout)
        return widget
    
    def select_theme_color(self):
        """选择主题颜色"""
        color = QColorDialog.getColor(QColor("#4CAF50"), self)
        if color.isValid():
            self.color_preview.setStyleSheet(
                f"background-color: {color.name()}; min-width: 50px;"
            )
    
    def select_font(self):
        """选择字体"""
        font, ok = QFontDialog.getFont(QFont("Arial", 12), self)
        if ok:
            self.font_label.setText(f"{font.family()}, {font.pointSize()}pt")
    
    def load_settings(self):
        """加载设置"""
        # 这里可以从配置文件加载
        pass
    
    def save_and_close(self):
        """保存并关闭"""
        self.apply_settings()
        self.accept()
    
    def apply_settings(self):
        """应用设置"""
        settings = {
            'language': self.language_combo.currentText(),
            'autosave': self.autosave_check.isChecked(),
            'autosave_interval': self.autosave_interval.value(),
        }
        
        print("应用设置:")
        for key, value in settings.items():
            print(f"  {key}: {value}")
        
        QMessageBox.information(self, "成功", "设置已应用！")

class MainWindow(QMainWindow):
    def __init__(self):
        super().__init__()
        self.setWindowTitle("设置对话框示例")
        self.setGeometry(100, 100, 400, 300)
        
        central_widget = QWidget()
        self.setCentralWidget(central_widget)
        layout = QVBoxLayout()
        
        settings_btn = QPushButton("⚙️ 打开设置")
        settings_btn.clicked.connect(self.open_settings)
        settings_btn.setStyleSheet("""
            QPushButton {
                padding: 15px 30px;
                font-size: 16px;
                background-color: #2196F3;
                color: white;
                border: none;
                border-radius: 5px;
            }
            QPushButton:hover {
                background-color: #1976D2;
            }
        """)
        layout.addWidget(settings_btn)
        
        central_widget.setLayout(layout)
    
    def open_settings(self):
        """打开设置对话框"""
        dialog = SettingsDialog(self)
        dialog.exec()

if __name__ == "__main__":
    app = QApplication([])
    window = MainWindow()
    window.show()
    app.exec()
```
