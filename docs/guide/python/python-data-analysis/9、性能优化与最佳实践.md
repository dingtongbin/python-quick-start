## 9.1 向量化操作 vs 循环

### 为什么向量化更快？

NumPy 和 Pandas 的向量化操作底层使用 C 语言实现，避免了 Python 循环的解释器开销，能够实现并行计算。

### 性能对比示例

```python
import numpy as np
import pandas as pd
import time

# 创建测试数据
np.random.seed(42)
n = 1000000
data = np.random.rand(n)
df = pd.DataFrame({'values': data})

print(f"测试数据量: {n:,} 条\n")

# 方法1: Python 循环（慢）
start_time = time.time()
result_loop = []
for value in data:
    result_loop.append(value * 2 + 1)
loop_time = time.time() - start_time
print(f"Python循环耗时: {loop_time:.4f}秒")

# 方法2: NumPy 向量化（快）
start_time = time.time()
result_vectorized = data * 2 + 1
vectorized_time = time.time() - start_time
print(f"NumPy向量化耗时: {vectorized_time:.4f}秒")

# 方法3: Pandas apply（中等）
start_time = time.time()
result_apply = df['values'].apply(lambda x: x * 2 + 1)
apply_time = time.time() - start_time
print(f"Pandas apply耗时: {apply_time:.4f}秒")

print(f"\n性能提升:")
print(f"向量化 vs 循环: {loop_time / vectorized_time:.1f}x 更快")
print(f"向量化 vs apply: {apply_time / vectorized_time:.1f}x 更快")

# 验证结果一致性
assert np.allclose(result_loop, result_vectorized)
assert np.allclose(result_vectorized, result_apply)
print("\n✓ 所有方法结果一致")
```

### 实际案例：数据转换

```python
# 创建示例数据
np.random.seed(42)
n = 500000
df_large = pd.DataFrame({
    'price': np.random.uniform(10, 1000, n),
    'quantity': np.random.randint(1, 100, n),
    'category': np.random.choice(['A', 'B', 'C', 'D'], n)
})

# 任务：计算销售额并分类

# ❌ 慢速方法：使用循环
def calculate_sales_slow(df):
    sales = []
    for idx, row in df.iterrows():
        sale = row['price'] * row['quantity']
        if sale > 10000:
            level = '高'
        elif sale > 5000:
            level = '中'
        else:
            level = '低'
        sales.append(level)
    return sales

start_time = time.time()
df_large['sales_level_slow'] = calculate_sales_slow(df_large)
slow_time = time.time() - start_time
print(f"循环方法耗时: {slow_time:.4f}秒")

# ✅ 快速方法：向量化操作
start_time = time.time()
df_large['total_sales'] = df_large['price'] * df_large['quantity']
df_large['sales_level_fast'] = np.where(
    df_large['total_sales'] > 10000, '高',
    np.where(df_large['total_sales'] > 5000, '中', '低')
)
fast_time = time.time() - start_time
print(f"向量化方法耗时: {fast_time:.4f}秒")

print(f"\n性能提升: {slow_time / fast_time:.1f}x 更快")

# 清理临时列
df_large = df_large.drop(columns=['sales_level_slow'])
```

### 批量优化函数

```python
def optimize_dataframe(df):
    """自动优化 DataFrame 的数据类型"""
    df_optimized = df.copy()
    
    for col in df_optimized.columns:
        col_type = df_optimized[col].dtype
        
        # 优化整数
        if col_type == 'int64':
            df_optimized[col] = pd.to_numeric(df_optimized[col], downcast='integer')
        
        # 优化浮点数
        elif col_type == 'float64':
            df_optimized[col] = pd.to_numeric(df_optimized[col], downcast='float')
        
        # 优化对象类型（字符串）
        elif col_type == 'object':
            num_unique = df_optimized[col].nunique()
            num_total = len(df_optimized[col])
            if num_unique / num_total < 0.5:  # 重复率高，使用 category
                df_optimized[col] = df_optimized[col].astype('category')
    
    return df_optimized

# 测试
df_large_test = pd.DataFrame({
    'int_values': np.random.randint(0, 1000, 100000),
    'float_values': np.random.rand(100000),
    'categories': np.random.choice(['A', 'B', 'C', 'D', 'E'], 100000),
    'descriptions': [f'item_{i}' for i in range(100000)]
})

print(f"优化前内存: {df_large_test.memory_usage(deep=True).sum() / 1024**2:.2f} MB")
df_optimized_test = optimize_dataframe(df_large_test)
print(f"优化后内存: {df_optimized_test.memory_usage(deep=True).sum() / 1024**2:.2f} MB")

saved = (df_large_test.memory_usage(deep=True).sum() - 
         df_optimized_test.memory_usage(deep=True).sum()) / 1024**2
print(f"节省: {saved:.2f} MB ({saved / (df_large_test.memory_usage(deep=True).sum() / 1024**2) * 100:.1f}%)")
```

## 9.3 大文件处理：分块读取

### 为什么要分块读取？

当文件太大无法一次性加载到内存时，可以使用分块读取的方式逐块处理。

### 分块读取 CSV

```python
# 创建大型 CSV 文件用于演示
n_rows = 2000000
df_large_csv = pd.DataFrame({
    'id': range(n_rows),
    'value1': np.random.rand(n_rows),
    'value2': np.random.randint(0, 100, n_rows),
    'category': np.random.choice(['A', 'B', 'C', 'D'], n_rows)
})

large_file = 'large_data.csv'
df_large_csv.to_csv(large_file, index=False)
print(f"已创建大型CSV文件: {os.path.getsize(large_file) / 1024 / 1024:.2f} MB")

# 方法1: 分块读取并处理
chunk_size = 100000
chunks = []

print("\n=== 分块读取处理 ===\n")
for i, chunk in enumerate(pd.read_csv(large_file, chunksize=chunk_size)):
    # 对每个块进行处理
    processed_chunk = chunk[chunk['value2'] > 50]  # 示例：过滤
    chunks.append(processed_chunk)
    print(f"处理第 {i+1} 块: {len(chunk)} 行 -> 过滤后 {len(processed_chunk)} 行")

# 合并结果
result_df = pd.concat(chunks, ignore_index=True)
print(f"\n最终结果: {len(result_df):,} 行")

# 方法2: 分块聚合（更省内存）
print("\n=== 分块聚合 ===\n")
agg_result = {
    'count': 0,
    'sum_value1': 0,
    'sum_value2': 0
}

for i, chunk in enumerate(pd.read_csv(large_file, chunksize=chunk_size)):
    agg_result['count'] += len(chunk)
    agg_result['sum_value1'] += chunk['value1'].sum()
    agg_result['sum_value2'] += chunk['value2'].sum()
    print(f"处理第 {i+1} 块")

agg_result['mean_value1'] = agg_result['sum_value1'] / agg_result['count']
agg_result['mean_value2'] = agg_result['sum_value2'] / agg_result['count']

print(f"\n聚合结果:")
print(f"总行数: {agg_result['count']:,}")
print(f"value1均值: {agg_result['mean_value1']:.4f}")
print(f"value2均值: {agg_result['mean_value2']:.4f}")

# 清理
del df_large_csv, chunks, result_df
gc.collect()
```

```python
def read_excel_in_chunks(filepath, sheet_name=0, chunk_size=10000):
    """分块读取Excel文件"""
    # 先读取全部数据（如果内存允许）
    df_full = pd.read_excel(filepath, sheet_name=sheet_name)
    
    # 然后分块处理
    for i in range(0, len(df_full), chunk_size):
        chunk = df_full.iloc[i:i+chunk_size]
```

```python
# data_utils.py
"""数据处理工具模块"""

import pandas as pd
import numpy as np

def load_data(filepath, **kwargs):
    """通用数据加载函数"""
    if filepath.endswith('.csv'):
        return pd.read_csv(filepath, **kwargs)
    elif filepath.endswith('.xlsx'):
        return pd.read_excel(filepath, **kwargs)
    else:
        raise ValueError(f"Unsupported file format: {filepath}")

def clean_data(df, drop_na=True, remove_duplicates=True):
    """数据清洗"""
    if drop_na:
        df = df.dropna()
    if remove_duplicates:
        df = df.drop_duplicates()
    return df

def summarize_data(df):
    """数据摘要"""
    return {
        'shape': df.shape,
        'columns': df.columns.tolist(),
        'dtypes': df.dtypes.to_dict(),
        'missing': df.isnull().sum().to_dict(),
        'describe': df.describe().to_dict()
    }
```

## 实战案例：性能优化前后对比

```python
import time
import pandas as pd
import numpy as np

print("=" * 60)
print("性能优化实战对比")
print("=" * 60)

# 创建测试数据
np.random.seed(42)

df_test['level_new'] = np.where(

)
time_new = time.time() - start
print(f"  优化后: {time_new:.4f}秒")
print(f"  提升: {time_old/time_new:.1f}x\n")
```

## 本章小结

**性能优化要点：**
1. **优先使用向量化操作**：避免 Python 循环，使用 NumPy/Pandas 内置函数
