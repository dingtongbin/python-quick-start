## 8.1 案例一：销售数据分析

### 项目背景

某电商公司需要分析过去一年的销售数据，了解销售趋势、产品表现、地区差异等，为决策提供数据支持。

### 数据准备

```python
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns

# 设置中文显示
plt.rcParams['font.sans-serif'] = ['SimHei']
plt.rcParams['axes.unicode_minus'] = False
sns.set_theme(style="whitegrid")

# 生成模拟销售数据
np.random.seed(42)
n_records = 5000

# 创建日期范围
dates = pd.date_range('2024-01-01', periods=365, freq='D')

# 生成数据
data = {
    '订单ID': range(1, n_records + 1),
    '订单日期': np.random.choice(dates, n_records),
    '产品类别': np.random.choice(['电子产品', '服装', '食品', '家居', '图书'], 
                                  n_records, p=[0.25, 0.25, 0.20, 0.20, 0.10]),
    '产品名称': np.random.choice([
        '手机', '电脑', '平板', '耳机', 'T恤', '衬衫', '裤子', '外套',
        '零食', '饮料', '保健品', '厨具', '床上用品', '装饰品', '小说', '教材'
    ], n_records),
    '单价': np.random.uniform(10, 5000, n_records),
    '数量': np.random.randint(1, 10, n_records),
    '城市': np.random.choice(['北京', '上海', '广州', '深圳', '杭州', '成都'], n_records),
    '客户等级': np.random.choice(['普通会员', '银卡会员', '金卡会员', '钻石会员'], 
                                  n_records, p=[0.4, 0.3, 0.2, 0.1])
}

df_sales = pd.DataFrame(data)

# 计算销售额
df_sales['销售额'] = df_sales['单价'] * df_sales['数量']

# 提取时间特征
df_sales['年份'] = df_sales['订单日期'].dt.year
df_sales['月份'] = df_sales['订单日期'].dt.month
df_sales['季度'] = df_sales['订单日期'].dt.quarter
df_sales['星期'] = df_sales['订单日期'].dt.day_name()

print("=== 数据概览 ===\n")
print(f"总记录数: {len(df_sales):,}")
print(f"时间范围: {df_sales['订单日期'].min()} 至 {df_sales['订单日期'].max()}")
print(f"\n前5条记录:\n{df_sales.head()}")
print(f"\n数据类型:\n{df_sales.dtypes}")
```

## 8.2 案例二：股票数据分析

### 项目背景

分析某只股票的历史价格数据，计算技术指标，评估投资风险和收益。

### 数据准备

```python
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import matplotlib.dates as mdates

plt.rcParams['font.sans-serif'] = ['SimHei']
plt.rcParams['axes.unicode_minus'] = False

# 生成模拟股票数据
np.random.seed(42)
n_days = 252  # 一年交易日

dates = pd.date_range('2024-01-01', periods=n_days, freq='B')  # 工作日

# 模拟股价走势（随机游走）
initial_price = 100
returns = np.random.normal(0.0005, 0.02, n_days)  # 日收益率
prices = initial_price * np.cumprod(1 + returns)

# 创建DataFrame
df_stock = pd.DataFrame({
    '日期': dates,
    '开盘价': prices * (1 + np.random.uniform(-0.01, 0.01, n_days)),
    '最高价': prices * (1 + np.abs(np.random.normal(0, 0.015, n_days))),
    '最低价': prices * (1 - np.abs(np.random.normal(0, 0.015, n_days))),
    '收盘价': prices,
    '成交量': np.random.randint(1000000, 5000000, n_days)
})

# 确保最高价 >= 其他价格，最低价 <= 其他价格
df_stock['最高价'] = df_stock[['开盘价', '最高价', '收盘价']].max(axis=1)
df_stock['最低价'] = df_stock[['开盘价', '最低价', '收盘价']].min(axis=1)

df_stock = df_stock.set_index('日期')

print("=== 股票数据概览 ===\n")
print(df_stock.head(10))
print(f"\n数据形状: {df_stock.shape}")
print(f"时间范围: {df_stock.index.min()} 至 {df_stock.index.max()}")
```

### 1. 基本价格分析

```python
# 基本统计
print("\n=== 价格统计 ===\n")
print(df_stock['收盘价'].describe().round(2))

# 计算涨跌幅
df_stock['涨跌幅'] = df_stock['收盘价'].pct_change() * 100
```

```python
rfm_date = df_activity['日期'].max()

# Recency: 最近一次活跃距今天数
recency = df_activity.groupby('用户ID')['日期'].max()
recency = (rfm_date - recency).dt.days

# Frequency: 活跃天数
frequency = df_activity.groupby('用户ID')['日期'].nunique()

# Monetary: 购买次数
monetary = df_activity[df_activity['是否购买'] == 1].groupby('用户ID').size()

# 创建RFM DataFrame
rfm_df = pd.DataFrame({
    'Recency': recency,
    'Frequency': frequency,
    'Monetary': monetary
})

# RFM 评分（简单分位数方法）
rfm_df['R_Score'] = pd.qcut(rfm_df['Recency'], q=3, labels=[3, 2, 1]).astype(int)
rfm_df['F_Score'] = pd.qcut(rfm_df['Frequency'].rank(method='first'), q=3, labels=[1, 2, 3]).astype(int)
rfm_df['M_Score'] = pd.qcut(rfm_df['Monetary'].rank(method='first'), q=3, labels=[1, 2, 3]).astype(int)

# 总分
rfm_df['RFM_Score'] = rfm_df['R_Score'] + rfm_df['F_Score'] + rfm_df['M_Score']

# 用户分群
def segment_customer(row):

        return '重要价值客户'
    elif row['F_Score'] >= 2 and row['M_Score'] >= 2:

    elif row['R_Score'] >= 2:

    else:
        return '一般客户'
```

```python
def create_formatted_excel(df_summary, title, filename):
    """创建格式化的Excel报表"""
    wb = Workbook()
    ws = wb.active
    ws.title = title
    
    # 定义样式
    header_font = Font(name='微软雅黑', size=12, bold=True, color='FFFFFF')
    header_fill = PatternFill(start_color='4472C4', end_color='4472C4', fill_type='solid')
    header_alignment = Alignment(horizontal='center', vertical='center')
    
    cell_font = Font(name='微软雅黑', size=11)
    cell_alignment = Alignment(horizontal='center', vertical='center')
    
    border = Border(
        left=Side(style='thin'),
        right=Side(style='thin'),
        top=Side(style='thin'),
        bottom=Side(style='thin')
    )
    
    # 写入标题
    ws.merge_cells('A1:G1')
    ws['A1'] = title
    ws['A1'].font = Font(name='微软雅黑', size=16, bold=True)
    ws['A1'].alignment = Alignment(horizontal='center', vertical='center')
    
    # 写入表头
    headers = list(df_summary.columns)
    for col_idx, header in enumerate(headers, 1):
        cell = ws.cell(row=3, column=col_idx, value=header)
        cell.font = header_font
        cell.fill = header_fill
        cell.alignment = header_alignment
        cell.border = border
    
    # 写入数据
    for row_idx, row in enumerate(df_summary.itertuples(index=False), 4):
        for col_idx, value in enumerate(row, 1):
            cell = ws.cell(row=row_idx, column=col_idx, value=value)
            cell.font = cell_font
            cell.alignment = cell_alignment
            cell.border = border
            
            # 数字格式
            if isinstance(value, (int, float)):
                cell.number_format = '#,##0.00'
    
    # 调整列宽
    for column in ws.columns:
        max_length = 0
        column_letter = column[0].column_letter
        for cell in column:
            try:
                if len(str(cell.value)) > max_length:
                    max_length = len(str(cell.value))
            except:
                pass
        adjusted_width = min(max_length + 4, 30)
        ws.column_dimensions[column_letter].width = adjusted_width
    
    # 保存
    wb.save(filename)
    print(f"已保存: {filename}")

# 生成各个汇总报表
create_formatted_excel(monthly_summary, '月度销售汇总',
```
