# 第七章：统计分析基础

统计分析是数据分析的核心，帮助我们理解数据特征、发现规律、验证假设。本章将学习描述性统计、相关性分析、假设检验等基础统计方法。

## 7.1 描述性统计

### 什么是描述性统计？

描述性统计用于总结和描述数据的主要特征，包括：

1. **集中趋势**：数据的"中心"位置
2. **离散程度**：数据的分散程度
3. **分布形态**：数据的分布形状

### 集中趋势度量

#### 均值（Mean）

均值是所有数据的算术平均，是最常用的集中趋势度量。

```python
import numpy as np
import pandas as pd

# 准备数据
data = [85, 92, 78, 95, 88, 90, 82, 87, 91, 86]

# 计算均值
mean_value = np.mean(data)
print(f"均值: {mean_value}")  # 87.4

# Pandas 中计算
df = pd.DataFrame({'成绩': data})
print(f"Pandas均值: {df['成绩'].mean()}")  # 87.4
```

**均值的优缺点：**
- ✅ 优点：利用了所有数据信息，数学性质好
- ❌ 缺点：对异常值敏感

```python
# 异常值对均值的影响
data_with_outlier = [85, 92, 78, 95, 88, 90, 82, 87, 91, 200]
print(f"含异常值的均值: {np.mean(data_with_outlier)}")  # 108.8（被拉高）
```

#### 中位数（Median）

中位数是将数据排序后位于中间位置的值。

```python
# 计算中位数
median_value = np.median(data)
print(f"中位数: {median_value}")  # 87.5

# Pandas 中计算
print(f"Pandas中位数: {df['成绩'].median()}")  # 87.5

# 中位数不受异常值影响
print(f"含异常值的中位数: {np.median(data_with_outlier)}")  # 89.0（稳定）
```

**中位数的特点：**
- ✅ 优点：对异常值不敏感，稳健性好
- ❌ 缺点：没有利用所有数据信息

#### 众数（Mode）

众数是数据中出现频率最高的值。

```python
from scipy import stats

# 计算众数
data_repeated = [85, 92, 78, 85, 88, 90, 85, 87, 91, 85]
mode_result = stats.mode(data_repeated)
print(f"众数: {mode_result.mode[0]}")  # 85
print(f"出现次数: {mode_result.count[0]}")  # 4

# Pandas 中计算
df_repeated = pd.DataFrame({'成绩': data_repeated})
print(f"Pandas众数:\n{df_repeated['成绩'].mode()}")  # 85
```

**众数的特点：**
- ✅ 优点：适用于分类数据，不受极端值影响
- ❌ 缺点：可能不存在或有多个众数

#### 三者比较

```python
import matplotlib.pyplot as plt
import seaborn as sns

# 创建偏态分布数据
np.random.seed(42)
skewed_data = np.random.exponential(scale=2, size=1000) + 5

# 计算三种集中趋势
mean_val = np.mean(skewed_data)
median_val = np.median(skewed_data)
mode_val = stats.mode(skewed_data.round(1)).mode[0]

print(f"均值: {mean_val:.2f}")
print(f"中位数: {median_val:.2f}")
print(f"众数: {mode_val:.2f}")

# 可视化
fig, ax = plt.subplots(figsize=(12, 6))
sns.histplot(skewed_data, bins=30, kde=True, ax=ax, color='steelblue', alpha=0.6)

# 标注三条线
ax.axvline(mean_val, color='red', linestyle='--', linewidth=2, label=f'均值: {mean_val:.2f}')
ax.axvline(median_val, color='green', linestyle='--', linewidth=2, label=f'中位数: {median_val:.2f}')
ax.axvline(mode_val, color='orange', linestyle='--', linewidth=2, label=f'众数: {mode_val:.2f}')

ax.set_title('偏态分布中的均值、中位数、众数', fontsize=16, fontweight='bold')
ax.set_xlabel('数值')
ax.set_ylabel('频数')
ax.legend(fontsize=12)
ax.grid(True, alpha=0.3)

plt.tight_layout()
plt.show()
```

### 离散程度度量

#### 极差（Range）

极差是最大值与最小值的差。

```python
data = [85, 92, 78, 95, 88, 90, 82, 87, 91, 86]

# 计算极差
range_value = np.max(data) - np.min(data)
print(f"极差: {range_value}")  # 17

# Pandas
print(f"Pandas极差: {df['成绩'].max() - df['成绩'].min()}")  # 17
```

#### 方差（Variance）

方差是每个数据与均值差的平方的平均值。

```python
# 计算方差
variance = np.var(data, ddof=1)  # ddof=1 表示样本方差
print(f"方差: {variance:.2f}")  # 28.71

# Pandas
print(f"Pandas方差: {df['成绩'].var():.2f}")  # 28.71

# 总体方差 vs 样本方差
print(f"总体方差 (ddof=0): {np.var(data, ddof=0):.2f}")  # 25.84
print(f"样本方差 (ddof=1): {np.var(data, ddof=1):.2f}")  # 28.71
```

**为什么除以 n-1？**
- 样本方差除以 n-1 是为了无偏估计总体方差
- 这是贝塞尔校正（Bessel's correction）

#### 标准差（Standard Deviation）

标准差是方差的平方根，与原始数据单位一致。

```python
# 计算标准差
std_dev = np.std(data, ddof=1)
print(f"标准差: {std_dev:.2f}")  # 5.36

# Pandas
print(f"Pandas标准差: {df['成绩'].std():.2f}")  # 5.36
```

**标准差的意义：**
- 约 68% 的数据在均值 ± 1 个标准差范围内
- 约 95% 的数据在均值 ± 2 个标准差范围内
- 约 99.7% 的数据在均值 ± 3 个标准差范围内

```python
# 验证经验法则
np.random.seed(42)
normal_data = np.random.normal(loc=100, scale=15, size=10000)

mean = np.mean(normal_data)
std = np.std(normal_data)

within_1_std = np.sum((normal_data >= mean - std) & (normal_data <= mean + std)) / len(normal_data)
within_2_std = np.sum((normal_data >= mean - 2*std) & (normal_data <= mean + 2*std)) / len(normal_data)
within_3_std = np.sum((normal_data >= mean - 3*std) & (normal_data <= mean + 3*std)) / len(normal_data)

print(f"±1σ 范围内: {within_1_std*100:.2f}%")  # 约 68%
print(f"±2σ 范围内: {within_2_std*100:.2f}%")  # 约 95%
print(f"±3σ 范围内: {within_3_std*100:.2f}%")  # 约 99.7%
```

#### 四分位距（IQR）

四分位距是第三四分位数与第一四分位数的差。

```python
# 计算四分位数
Q1 = np.percentile(data, 25)  # 第一四分位数（25%）
Q2 = np.percentile(data, 50)  # 第二四分位数（中位数）
Q3 = np.percentile(data, 75)  # 第三四分位数（75%）

print(f"Q1 (25%): {Q1}")
print(f"Q2 (50%): {Q2}")
print(f"Q3 (75%): {Q3}")

# 计算 IQR
IQR = Q3 - Q1
print(f"IQR: {IQR}")

# Pandas
print(f"\nPandas四分位数:")
print(df['成绩'].quantile([0.25, 0.5, 0.75]))
```

**IQR 的应用：检测异常值**

```python
# 使用 IQR 检测异常值
lower_bound = Q1 - 1.5 * IQR
upper_bound = Q3 + 1.5 * IQR

print(f"正常范围: [{lower_bound}, {upper_bound}]")

outliers = [x for x in data if x < lower_bound or x > upper_bound]
print(f"异常值: {outliers}")
```

### 分布形态度量

#### 偏度（Skewness）

偏度衡量数据分布的不对称程度。

```python
from scipy import stats

# 计算偏度
skewness = stats.skew(data)
print(f"偏度: {skewness:.2f}")

# 解释偏度
if abs(skewness) < 0.5:
    print("分布基本对称")
elif abs(skewness) < 1:
    print("分布中等偏斜")
else:
    print("分布高度偏斜")

if skewness > 0:
    print("右偏（正偏）：长尾在右侧")
elif skewness < 0:
    print("左偏（负偏）：长尾在左侧")
```

**偏度的含义：**
- **偏度 = 0**：对称分布（如正态分布）
- **偏度 > 0**：右偏分布（均值 > 中位数）
- **偏度 < 0**：左偏分布（均值 < 中位数）

```python
# 不同偏度的可视化
fig, axes = plt.subplots(1, 3, figsize=(18, 5))

# 左偏分布
left_skewed = np.random.beta(5, 2, 1000) * 100
axes[0].hist(left_skewed, bins=30, color='steelblue', edgecolor='black', alpha=0.7)
axes[0].axvline(np.mean(left_skewed), color='red', linestyle='--', label=f'均值: {np.mean(left_skewed):.1f}')
axes[0].axvline(np.median(left_skewed), color='green', linestyle='--', label=f'中位数: {np.median(left_skewed):.1f}')
axes[0].set_title(f'左偏分布 (偏度: {stats.skew(left_skewed):.2f})')
axes[0].legend()
axes[0].grid(True, alpha=0.3)

# 对称分布
symmetric = np.random.normal(100, 15, 1000)
axes[1].hist(symmetric, bins=30, color='coral', edgecolor='black', alpha=0.7)
axes[1].axvline(np.mean(symmetric), color='red', linestyle='--', label=f'均值: {np.mean(symmetric):.1f}')
axes[1].axvline(np.median(symmetric), color='green', linestyle='--', label=f'中位数: {np.median(symmetric):.1f}')
axes[1].set_title(f'对称分布 (偏度: {stats.skew(symmetric):.2f})')
axes[1].legend()
axes[1].grid(True, alpha=0.3)

# 右偏分布
right_skewed = np.random.exponential(2, 1000) * 20
axes[2].hist(right_skewed, bins=30, color='lightgreen', edgecolor='black', alpha=0.7)
axes[2].axvline(np.mean(right_skewed), color='red', linestyle='--', label=f'均值: {np.mean(right_skewed):.1f}')
axes[2].axvline(np.median(right_skewed), color='green', linestyle='--', label=f'中位数: {np.median(right_skewed):.1f}')
axes[2].set_title(f'右偏分布 (偏度: {stats.skew(right_skewed):.2f})')
axes[2].legend()
axes[2].grid(True, alpha=0.3)

plt.suptitle('不同偏度的分布对比', fontsize=16, fontweight='bold')
plt.tight_layout()
plt.show()
```

#### 峰度（Kurtosis）

峰度衡量数据分布的尖锐程度和尾部厚度。

```python
# 计算峰度
kurtosis = stats.kurtosis(data)
print(f"峰度: {kurtosis:.2f}")

# 解释峰度（超额峰度）
if kurtosis > 0:
    print("尖峰厚尾：比正态分布更尖锐，尾部更厚")
elif kurtosis < 0:
    print("平峰薄尾：比正态分布更平坦，尾部更薄")
else:
    print("与正态分布相似")
```

**峰度的含义：**
- **峰度 = 0**：与正态分布相同（中等峰值和尾部）
- **峰度 > 0**：尖峰厚尾（更多极端值）
- **峰度 < 0**：平峰薄尾（更少极端值）

```python
# 不同峰度的可视化
fig, axes = plt.subplots(1, 3, figsize=(18, 5))

# 高峰度（尖峰厚尾）
high_kurt = np.random.laplace(0, 1, 10000)
axes[0].hist(high_kurt, bins=50, density=True, color='steelblue', edgecolor='black', alpha=0.7)
axes[0].set_title(f'高峰度 (峰度: {stats.kurtosis(high_kurt):.2f})')
axes[0].grid(True, alpha=0.3)

# 正态分布（基准）
normal = np.random.normal(0, 1, 10000)
axes[1].hist(normal, bins=50, density=True, color='coral', edgecolor='black', alpha=0.7)
axes[1].set_title(f'正态分布 (峰度: {stats.kurtosis(normal):.2f})')
axes[1].grid(True, alpha=0.3)

# 低峰度（平峰薄尾）
low_kurt = np.random.uniform(-3, 3, 10000)
axes[2].hist(low_kurt, bins=50, density=True, color='lightgreen', edgecolor='black', alpha=0.7)
axes[2].set_title(f'低峰度 (峰度: {stats.kurtosis(low_kurt):.2f})')
axes[2].grid(True, alpha=0.3)

plt.suptitle('不同峰度的分布对比', fontsize=16, fontweight='bold')
plt.tight_layout()
plt.show()
```

## 7.2 相关性分析

### 什么是相关性？

相关性衡量两个变量之间的线性关系强度和方向。

### 相关系数类型

#### Pearson 相关系数

Pearson 相关系数衡量两个连续变量之间的线性相关性。

```python
from scipy import stats

# 准备数据
np.random.seed(42)
n = 100
x = np.random.rand(n) * 100
y = 2 * x + np.random.randn(n) * 20  # 强正相关

# 计算 Pearson 相关系数
pearson_corr, p_value = stats.pearsonr(x, y)
print(f"Pearson相关系数: {pearson_corr:.3f}")
print(f"P值: {p_value:.6f}")

# 解释相关系数
if abs(pearson_corr) >= 0.8:
    strength = "强相关"
elif abs(pearson_corr) >= 0.5:
    strength = "中等相关"
elif abs(pearson_corr) >= 0.3:
    strength = "弱相关"
else:
    strength = "几乎无关"

direction = "正相关" if pearson_corr > 0 else "负相关"
print(f"结论: {strength}，{direction}")
```

**Pearson 相关系数的特点：**
- 取值范围：[-1, 1]
- **r = 1**：完全正相关
- **r = -1**：完全负相关
- **r = 0**：无线性相关
- 只衡量线性关系，对非线性关系不敏感
- 要求数据近似正态分布

#### Spearman 等级相关系数

Spearman 相关系数基于数据的秩次，适用于非线性单调关系。

```python
# 计算 Spearman 相关系数
spearman_corr, p_value = stats.spearmanr(x, y)
print(f"Spearman相关系数: {spearman_corr:.3f}")
print(f"P值: {p_value:.6f}")

# 非线性但单调的关系
x_nonlinear = np.linspace(1, 10, 100)
y_nonlinear = x_nonlinear ** 2 + np.random.randn(100) * 5

pearson_nl, _ = stats.pearsonr(x_nonlinear, y_nonlinear)
spearman_nl, _ = stats.spearmanr(x_nonlinear, y_nonlinear)

print(f"\n非线性关系:")
print(f"Pearson: {pearson_nl:.3f}")
print(f"Spearman: {spearman_nl:.3f}")
```

**Spearman 的特点：**
- 基于秩次，不要求正态分布
- 能捕捉单调非线性关系
- 对异常值不敏感

#### Kendall 等级相关系数

Kendall 相关系数也基于秩次，适合小样本数据。

```python
kendall_corr, p_value = stats.kendalltau(x, y)
print(f"Kendall相关系数: {kendall_corr:.3f}")
print(f"P值: {p_value:.6f}")
```

### 实际案例：销售数据分析

```python
# 模拟销售数据
np.random.seed(42)
n = 200

sales_df = pd.DataFrame({
    '月份': range(1, n+1),
    '广告费用': np.random.uniform(10, 50, n),
    '促销活动': np.random.randint(0, 3, n),
    '价格折扣': np.random.uniform(0, 0.3, n),
    '季节因子': np.sin(np.arange(n) * 2 * np.pi / 12) * 0.3 + 1,
})

# 生成销售额（与多个因素相关）
sales_df['销售额'] = (
    100 + 
    sales_df['广告费用'] * 5 + 
    sales_df['促销活动'] * 30 + 
    sales_df['价格折扣'] * (-100) + 
    sales_df['季节因子'] * 50 +
    np.random.normal(0, 20, n)
)

# 计算相关性
sales_corr = sales_df.drop('月份', axis=1).corr()

# 可视化
fig, ax = plt.subplots(figsize=(10, 8))
heatmap = sns.heatmap(sales_corr, annot=True, fmt='.3f', cmap='RdBu_r',
                      center=0, square=True, linewidths=1, linecolor='white',
                      cbar_kws={'shrink': 0.8}, ax=ax, vmin=-1, vmax=1)

ax.set_title('销售影响因素相关性分析', fontsize=16, fontweight='bold', pad=20)
plt.xticks(rotation=45, ha='right')
plt.yticks(rotation=0)
plt.tight_layout()
plt.show()

# 找出与销售额最相关的因素
sales_correlations = sales_corr['销售额'].drop('销售额').abs().sort_values(ascending=False)
print("=== 与销售额的相关性（按绝对值排序）===\n")
for factor, corr in sales_correlations.items():
    original_corr = sales_corr.loc[factor, '销售额']
    print(f"{factor}: {original_corr:.3f}")
```

## 7.3 假设检验简介

### 什么是假设检验？

假设检验是统计推断的重要方法，用于判断样本数据是否支持某个假设。

**基本步骤：**
1. 建立原假设（H₀）和备择假设（H₁）
2. 选择显著性水平（α，通常为 0.05）
3. 计算检验统计量和 P 值
4. 做出决策：如果 P < α，拒绝原假设

### 单样本 t 检验

检验样本均值是否等于某个特定值。

```python
from scipy import stats

# 示例：检验学生平均成绩是否为 80 分
np.random.seed(42)
scores = np.random.normal(82, 10, 50)  # 实际均值 82

# 进行单样本 t 检验
t_stat, p_value = stats.ttest_1samp(scores, 80)

print(f"样本均值: {np.mean(scores):.2f}")
print(f"t 统计量: {t_stat:.3f}")
print(f"P 值: {p_value:.4f}")

# 做出决策
alpha = 0.05
if p_value < alpha:
    print(f"\n结论: P值 ({p_value:.4f}) < α ({alpha})，拒绝原假设")
    print("学生平均成绩显著不等于 80 分")
else:
    print(f"\n结论: P值 ({p_value:.4f}) >= α ({alpha})，不能拒绝原假设")
    print("没有足够证据表明学生平均成绩不等于 80 分")
```

检验两个独立样本的均值是否有显著差异。

## 7.4 异常值检测

### 为什么需要检测异常值？

异常值可能由以下原因产生：
- 数据录入错误
- 测量误差
- 真实的极端情况
- 罕见事件

异常值会影响统计分析结果，需要识别和处理。

### IQR 方法

基于四分位距检测异常值。

```python
def detect_outliers_iqr(data, column):
    """使用 IQR 方法检测异常值"""
    Q1 = data[column].quantile(0.25)
    Q3 = data[column].quantile(0.75)
    IQR = Q3 - Q1
    
    lower_bound = Q1 - 1.5 * IQR
    upper_bound = Q3 + 1.5 * IQR
    
    outliers = data[(data[column] < lower_bound) | (data[column] > upper_bound)]
    
    return {
        'lower_bound': lower_bound,
        'upper_bound': upper_bound,
        'outliers': outliers,
        'outlier_count': len(outliers)
    }

# 示例
np.random.seed(42)
df_outlier = pd.DataFrame({
    '销售额': np.concatenate([
        np.random.normal(1000, 200, 95),  # 正常数据
        [3000, 3500, 4000, 2500, 2800]    # 异常值
    ])
})

result = detect_outliers_iqr(df_outlier, '销售额')

print(f"正常范围: [{result['lower_bound']:.2f}, {result['upper_bound']:.2f}]")
print(f"异常值数量: {result['outlier_count']}")
print(f"\n异常值:\n{result['outliers']}")

# 可视化
fig, ax = plt.subplots(figsize=(10, 6))
sns.boxplot(y=df_outlier['销售额'], ax=ax, color='steelblue')
ax.axhline(result['lower_bound'], color='red', linestyle='--', 
           label=f"下界: {result['lower_bound']:.0f}")
ax.axhline(result['upper_bound'], color='red', linestyle='--', 
           label=f"上界: {result['upper_bound']:.0f}")
ax.set_title('IQR 方法检测异常值', fontsize=16, fontweight='bold')
ax.set_ylabel('销售额')
ax.legend()
ax.grid(True, alpha=0.3, axis='y')
plt.tight_layout()
plt.show()
```

### Z-Score 方法

基于标准分数检测异常值。

```python
def detect_outliers_zscore(data, column, threshold=3):
    """使用 Z-Score 方法检测异常值"""
    mean = data[column].mean()
    std = data[column].std()
    
    z_scores = np.abs((data[column] - mean) / std)
    outliers = data[z_scores > threshold]
    
    return {
        'mean': mean,
        'std': std,
        'z_scores': z_scores,
        'outliers': outliers,
        'outlier_count': len(outliers)
    }

# 示例
result_z = detect_outliers_zscore(df_outlier, '销售额', threshold=2.5)

print(f"均值: {result_z['mean']:.2f}")
print(f"标准差: {result_z['std']:.2f}")
print(f"异常值数量: {result_z['outlier_count']}")
print(f"\n异常值:\n{result_z['outliers']}")

# 可视化 Z-Score 分布
fig, ax = plt.subplots(figsize=(10, 6))
ax.hist(result_z['z_scores'], bins=20, color='steelblue', 
        edgecolor='black', alpha=0.7)
ax.axvline(2.5, color='red', linestyle='--', linewidth=2, 
           label='阈值: 2.5')
ax.set_title('Z-Score 分布', fontsize=16, fontweight='bold')
ax.set_xlabel('Z-Score')
ax.set_ylabel('频数')
ax.legend()
ax.grid(True, alpha=0.3)
plt.tight_layout()
plt.show()
```

```python
# 处理异常值的几种方法
df_clean = df_outlier.copy()

# 方法1：删除异常值
Q1 = df_clean['销售额'].quantile(0.25)
Q3 = df_clean['销售额'].quantile(0.75)
IQR = Q3 - Q1
lower = Q1 - 1.5 * IQR
upper = Q3 + 1.5 * IQR

df_removed = df_clean[(df_clean['销售额'] >= lower) & 
                      (df_clean['销售额'] <= upper)]
print(f"删除异常值后: {len(df_removed)} 条记录")

median_val = df_clean['销售额'].median()
```
