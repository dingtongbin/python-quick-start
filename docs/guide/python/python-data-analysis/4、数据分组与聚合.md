# 第四章：数据分组与聚合

分组聚合是数据分析中最常用的操作之一，用于对数据进行分类统计和汇总。本章将深入讲解 Pandas 的 GroupBy 机制。

---

## 4.1 GroupBy 机制详解

### 什么是 GroupBy？

GroupBy 遵循"拆分-应用-合并"（Split-Apply-Combine）模式：

1. **拆分**：根据条件将数据分成多个组
2. **应用**：对每个组应用函数（求和、均值等）
3. **合并**：将结果合并成新的数据结构

### 基本语法

```python
import pandas as pd

# 创建示例数据
df = pd.DataFrame({
    '部门': ['技术部', '销售部', '技术部', '人事部', '销售部', '技术部'],
    '姓名': ['张三', '李四', '王五', '赵六', '钱七', '孙八'],
    '工资': [8000, 9000, 8500, 7500, 9500, 8200],
    '年龄': [25, 30, 28, 35, 32, 27]
})

print("原始数据:\n", df)

# 按部门分组
grouped = df.groupby('部门')

# 查看分组
for name, group in grouped:
    print(f"\n部门: {name}")
    print(group)
```

### 分组后的操作

```python
# 单列分组
grouped = df.groupby('部门')

# 计算每组的均值
print("各部门平均工资:\n", grouped['工资'].mean())

# 计算每组的总和
print("\n各部门工资总和:\n", grouped['工资'].sum())

# 计算每组的数量
print("\n各部门人数:\n", grouped['姓名'].count())

# 多个统计量
print("\n各部门工资统计:\n")
print(grouped['工资'].agg(['mean', 'sum', 'count']))
```

## 4.2 单列分组与多列分组

### 单列分组

```python
# 按单个列分组
df = pd.DataFrame({
    '城市': ['北京', '上海', '北京', '广州', '上海', '北京'],
    '产品': ['A', 'B', 'A', 'C', 'B', 'C'],
    '销售额': [100, 200, 150, 300, 250, 180]
})

# 按城市分组
city_group = df.groupby('城市')

print("各城市销售统计:\n")
print(city_group['销售额'].agg(['sum', 'mean', 'count']))
```

### 多列分组

```python
# 按多个列分组
multi_group = df.groupby(['城市', '产品'])

print("各城市各产品销售统计:\n")
print(multi_group['销售额'].sum())

# 重置索引，使结果更易读
result = multi_group['销售额'].sum().reset_index()
print("\n重置索引后:\n", result)
```

### 多级索引操作

```python
# 多级索引的结果
result = df.groupby(['城市', '产品'])['销售额'].sum()
print("多级索引结果:\n", result)

# 访问特定组
print("\n北京产品A的销售额:", result['北京']['A'])

# 转换为 DataFrame
result_df = result.reset_index()
print("\n转换后:\n", result_df)
```

## 4.3 聚合函数

### 内置聚合函数

```python
df = pd.DataFrame({
    '部门': ['技术部', '销售部', '技术部', '人事部', '销售部'],
    '工资': [8000, 9000, 8500, 7500, 9500],
    '年龄': [25, 30, 28, 35, 32]
})

grouped = df.groupby('部门')

# 常用聚合函数
print("最大值:\n", grouped.max())
print("\n最小值:\n", grouped.min())
print("\n均值:\n", grouped.mean())
print("\n中位数:\n", grouped.median())
print("\n标准差:\n", grouped.std())
print("\n计数:\n", grouped.count())
print("\n总和:\n", grouped.sum())
```

### agg 方法 - 自定义聚合

```python
# 对单列使用多个聚合函数
result = grouped['工资'].agg(['sum', 'mean', 'max', 'min'])
print("工资多维度统计:\n", result)

# 重命名聚合列
result = grouped['工资'].agg(
    总工资='sum',
    平均工资='mean',
    最高工资='max',
    最低工资='min'
)
print("\n重命名后:\n", result)

# 对不同列使用不同聚合函数
result = df.groupby('部门').agg(
    总工资=('工资', 'sum'),
    平均工资=('工资', 'mean'),
    最大年龄=('年龄', 'max'),
    平均年龄=('年龄', 'mean')
)
print("\n多列聚合:\n", result)
```

### 自定义聚合函数

```python
# 定义自定义函数
def 极差(x):
    return x.max() - x.min()

# 使用自定义函数
result = grouped['工资'].agg(['mean', 'std', 极差])
print("包含极差的统计:\n", result)

# 使用 lambda 函数
result = grouped['工资'].agg(
    平均值='mean',
    范围=lambda x: x.max() - x.min()
)
print("\n使用lambda:\n", result)
```

## 4.4 自定义聚合函数

### apply 方法

```python
df = pd.DataFrame({
    '部门': ['技术部', '销售部', '技术部', '人事部', '销售部', '技术部'],
    '姓名': ['张三', '李四', '王五', '赵六', '钱七', '孙八'],
    '工资': [8000, 9000, 8500, 7500, 9500, 8200]
})

# 获取每个部门工资最高的员工信息
def 最高工资员工(group):
    return group.loc[group['工资'].idxmax()]

result = df.groupby('部门').apply(最高工资员工)
print("各部门最高工资员工:\n", result)
```

### transform 方法

```python
# 计算每个部门的平均工资，并添加到原数据
df['部门平均工资'] = df.groupby('部门')['工资'].transform('mean')
print("添加部门平均工资:\n", df)

# 计算与部门平均工资的差异
df['工资差异'] = df['工资'] - df['部门平均工资']
print("\n工资差异:\n", df)
```

### filter 方法

```python
# 筛选出人数大于2的部门
def 人数大于2(group):
    return len(group) > 2

filtered = df.groupby('部门').filter(人数大于2)
print("人数大于2的部门:\n", filtered)
```

## 4.5 透视表：pivot_table

### 基本透视表

```python
df = pd.DataFrame({
    '日期': ['2024-01', '2024-01', '2024-02', '2024-02', '2024-03', '2024-03'],
    '产品': ['A', 'B', 'A', 'B', 'A', 'B'],
    '地区': ['华东', '华北', '华东', '华北', '华东', '华北'],
    '销售额': [100, 200, 150, 250, 180, 220]
})

# 创建透视表
pivot = pd.pivot_table(df, 
                       values='销售额',      # 值
                       index='产品',          # 行
                       columns='日期',        # 列
                       aggfunc='sum')         # 聚合函数

print("透视表:\n", pivot)
```

### 高级透视表

```python
# 多值、多行列
pivot = pd.pivot_table(df,
                       values=['销售额'],
                       index=['产品', '地区'],
                       columns=['日期'],
                       aggfunc={'销售额': ['sum', 'mean']},
                       fill_value=0,           # 填充缺失值
                       margins=True,           # 添加总计
                       margins_name='总计')     # 总计名称

print("高级透视表:\n", pivot)
```

### 交叉表：crosstab

```python
# 创建交叉表
cross = pd.crosstab(index=df['产品'],
                    columns=df['地区'],
                    values=df['销售额'],
                    aggfunc='sum')

print("交叉表:\n", cross)

# 归一化交叉表（百分比）
cross_norm = pd.crosstab(index=df['产品'],
                         columns=df['地区'],
                         normalize='index')  # 按行归一化

print("\n归一化交叉表:\n", cross_norm)
```

## 4.6 实战案例

### 案例：电商销售数据分析

```python
import pandas as pd

# 模拟电商数据
data = {
    '订单日期': ['2024-01-01', '2024-01-01', '2024-01-02', '2024-01-02', 
                '2024-01-03', '2024-01-03', '2024-01-04', '2024-01-04'],
    '产品类别': ['电子产品', '服装', '电子产品', '食品', 
                '服装', '电子产品', '食品', '服装'],
    '产品名称': ['手机', 'T恤', '电脑', '零食', 
                '裤子', '平板', '饮料', '衬衫'],
    '单价': [3999, 99, 5999, 29, 199, 2999, 15, 149],
    '数量': [2, 5, 1, 10, 3, 2, 20, 4],
    '城市': ['北京', '上海', '北京', '广州', '上海', '北京', '广州', '上海']
}

df = pd.DataFrame(data)

# 计算销售额
df['销售额'] = df['单价'] * df['数量']

print("=== 原始数据 ===\n", df.head())

# 1. 按产品类别统计
category_stats = df.groupby('产品类别').agg(
    订单数=('产品名称', 'count'),
    总销售额=('销售额', 'sum'),
    平均单价=('单价', 'mean'),
    总销量=('数量', 'sum')
).round(2)

print("\n=== 产品类别统计 ===\n", category_stats)

# 2. 按城市和产品类别的透视表
city_category_pivot = pd.pivot_table(df,
                                     values='销售额',
                                     index='城市',
                                     columns='产品类别',
                                     aggfunc='sum',
                                     fill_value=0)

print("\n=== 城市-类别销售透视表 ===\n", city_category_pivot)

# 3. 找出每个城市的畅销产品
def 畅销产品(group):
    return group.loc[group['销售额'].idxmax()]

best_sellers = df.groupby('城市').apply(畅销产品)[['城市', '产品名称', '销售额']]
print("\n=== 各城市畅销产品 ===\n", best_sellers)

# 4. 每日销售趋势
daily_sales = df.groupby('订单日期')['销售额'].sum().reset_index()
print("\n=== 每日销售趋势 ===\n", daily_sales)
```

import pandas as pd
