## 10.2 数据管道构建

### 什么是数据管道？

数据管道是一系列数据处理步骤的自动化流程，从数据获取、清洗、转换到输出，形成可重复执行的工作流。

### 简单数据管道

```python
class DataPipeline:
    """简单数据管道"""
    
    def __init__(self):
        self.data = None
        self.steps = []
    
    def load(self, filepath, **kwargs):
        """加载数据"""
        print(f"步骤1: 加载数据 from {filepath}")
        if filepath.endswith('.csv'):
            self.data = pd.read_csv(filepath, **kwargs)
        elif filepath.endswith('.xlsx'):
            self.data = pd.read_excel(filepath, **kwargs)
        else:
            raise ValueError("Unsupported file format")
        print(f"  ✓ 加载完成: {len(self.data)} 行")
        return self
    
    def clean(self, drop_na=True, remove_duplicates=True):
        """清洗数据"""
        print("步骤2: 清洗数据")
        initial_rows = len(self.data)
        
        if drop_na:
            self.data = self.data.dropna()
            print(f"  - 删除缺失值: {initial_rows - len(self.data)} 行")
            initial_rows = len(self.data)
        
        if remove_duplicates:
            self.data = self.data.drop_duplicates()
            print(f"  - 删除重复值: {initial_rows - len(self.data)} 行")
        
        print(f"  ✓ 清洗完成: {len(self.data)} 行")
        return self
    
    def transform(self, func, **kwargs):
        """转换数据"""
        print("步骤3: 转换数据")
        self.data = func(self.data, **kwargs)
        print(f"  ✓ 转换完成")
        return self
    
    def aggregate(self, groupby_col, agg_dict):
        """聚合数据"""
        print(f"步骤4: 按 {groupby_col} 聚合")
        self.data = self.data.groupby(groupby_col).agg(agg_dict)
        print(f"  ✓ 聚合完成: {len(self.data)} 组")
        return self
    
    def save(self, filepath, **kwargs):
        """保存数据"""
        print(f"步骤5: 保存数据 to {filepath}")
        if filepath.endswith('.csv'):
            self.data.to_csv(filepath, index=False, **kwargs)
        elif filepath.endswith('.xlsx'):
            self.data.to_excel(filepath, index=False, **kwargs)
        print(f"  ✓ 保存完成")
        return self
    
    def run(self):
        """执行管道"""
        print("\n" + "="*60)
        print("数据管道执行开始")
        print("="*60 + "\n")
        return self

# 使用示例
def add_features(df):
    """添加特征"""
    df['total'] = df['price'] * df['quantity']
    df['category_level'] = df['category'].map({'A': 1, 'B': 2, 'C': 3})
    return df

# 构建和执行管道
pipeline = DataPipeline()
result = (pipeline
    .run()
    .load('data.csv')
    .clean(drop_na=True, remove_duplicates=True)
    .transform(add_features)
    .aggregate('category', {'total': 'sum', 'quantity': 'mean'})
    .save('output.csv'))

print("\n" + "="*60)
print("数据管道执行完成")
print("="*60)
```

```python
from prefect import flow, task
import pandas as pd

@task
def extract_data(filepath):
    """提取数据"""
    print(f"提取数据: {filepath}")
    df = pd.read_csv(filepath)
    return df

@task
def transform_data(df):
    """转换数据"""
    print("转换数据")
    df = df.dropna()
    df['total'] = df['price'] * df['quantity']
    return df

@task
def load_data(df, output_path):
    """加载数据"""
    print(f"保存数据: {output_path}")
    df.to_csv(output_path, index=False)

@flow
def etl_pipeline(input_path, output_path):
    """ETL 流程"""
    df = extract_data(input_path)
    df_clean = transform_data(df)
    load_data(df_clean, output_path)
```

```python

query = "SELECT * FROM users WHERE age > 25"
df = pd.read_sql_query(query, engine)

# 方法2: 使用参数化查询（防止 SQL 注入）
age_threshold = 25
query = text("SELECT * FROM users WHERE age > :age")
df = pd.read_sql_query(query, engine, params={'age': age_threshold})

# 方法3: 读取整个表
df = pd.read_sql_table('users', engine)

# 方法4: 分块读取大表
chunks = pd.read_sql_query(query, engine, chunksize=10000)
for i, chunk in enumerate(chunks):
    print(f"处理第 {i+1} 块: {len(chunk)} 行")
```

### 写入数据

```python

df_to_save.to_sql(
```

### 事务管理

```python
from sqlalchemy.orm import sessionmaker

# 创建会话
Session = sessionmaker(bind=engine)
session = Session()

try:
    # 执行多个操作
    df1.to_sql('table1', engine, if_exists='append', index=False)
    df2.to_sql('table2', engine, if_exists='append', index=False)
    
    # 提交事务
    session.commit()
    print("事务提交成功")
    
except Exception as e:
    # 回滚事务
    session.rollback()
    print(f"事务失败，已回滚: {e}")
    
finally:
    session.close()
```

```python
import requests
from bs4 import BeautifulSoup
import pandas as pd

def scrape_website(url):
    """抓取网页数据"""
    headers = {
        'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36'
    }
    
    response = requests.get(url, headers=headers)
    response.encoding = 'utf-8'
    
    if response.status_code == 200:
        soup = BeautifulSoup(response.text, 'html.parser')
        return soup
    else:
        raise Exception(f"Request failed with status {response.status_code}")

# 示例：抓取新闻标题
def scrape_news():
    """抓取新闻数据"""
    url = "https://news.ycombinator.com/"
    soup = scrape_website(url)
    
    titles = []
    links = []
    
    for item in soup.select('.titleline > a'):
        titles.append(item.get_text())
        links.append(item.get('href'))
    
    df_news = pd.DataFrame({
        'title': titles,
        'link': links
    })
    
    return df_news
```

```python
from selenium import webdriver
from selenium.webdriver.common.by import By
from webdriver_manager.chrome import ChromeDriverManager
from selenium.webdriver.chrome.service import Service

def scrape_dynamic_page(url):
    """抓取动态网页"""
    options = webdriver.ChromeOptions()
    options.add_argument('--headless')  # 无头模式
    
    driver = webdriver.Chrome(service=Service(ChromeDriverManager().install()), 
                             options=options)
    
    try:
        driver.get(url)
        
        # 等待页面加载
        import time
        time.sleep(3)
        
        # 提取数据
        elements = driver.find_elements(By.CSS_SELECTOR, '.item-class')
        data = [elem.text for elem in elements]
        
        return data
        
    finally:
        driver.quit()
```

## 10.5 机器学习预备知识简介

### 什么是机器学习？

机器学习是让计算机从数据中学习规律，并对新数据做出预测或决策的技术。

### 机器学习流程

```
数据收集 → 数据清洗 → 特征工程 → 模型训练 → 模型评估 → 部署应用
```

### 使用 Scikit-learn 入门

```python

import numpy as np

# 准备数据
```
