# 第六章：数据可视化

数据可视化是将数据转换为图形的过程，帮助我们更直观地理解数据、发现规律和趋势。本章将学习使用 Matplotlib 和 Seaborn 进行数据可视化。

## 6.1 Matplotlib 基础

### 什么是 Matplotlib？

Matplotlib 是 Python 最基础的绘图库，提供了类似 MATLAB 的绘图接口。它是许多高级可视化库（如 Seaborn）的基础。

**主要特点：**
- **功能强大**：支持几乎所有类型的二维图表
- **高度可定制**：可以控制图表的每个细节
- **跨平台**：支持 Windows、macOS、Linux
- **多种输出格式**：PNG、JPG、PDF、SVG 等
- **交互式支持**：支持缩放、平移等交互操作

### 安装与导入

```python
# 如果未安装
# pip install matplotlib

import matplotlib.pyplot as plt
import numpy as np
import pandas as pd

# 设置中文显示（Windows）
plt.rcParams['font.sans-serif'] = ['SimHei']  # 用来正常显示中文标签
plt.rcParams['axes.unicode_minus'] = False  # 用来正常显示负号

# macOS 使用
# plt.rcParams['font.sans-serif'] = ['Arial Unicode MS']

# Linux 使用
# plt.rcParams['font.sans-serif'] = ['WenQuanYi Micro Hei']
```

### 基本绘图流程

Matplotlib 绘图遵循以下基本流程：

1. **准备数据**：准备好要绘制的 x 和 y 数据
2. **创建图形**：使用 `plt.figure()` 创建画布
3. **绘制图表**：使用 `plt.plot()` 等方法绘制
4. **添加元素**：添加标题、标签、图例等
5. **显示或保存**：使用 `plt.show()` 显示或 `plt.savefig()` 保存

```python
# 1. 准备数据
x = np.linspace(0, 10, 100)  # 0到10之间100个点
y = np.sin(x)

# 2. 创建图形
plt.figure(figsize=(10, 6))  # 设置图形大小：宽10英寸，高6英寸

# 3. 绘制图表
plt.plot(x, y, label='sin(x)', color='blue', linewidth=2)

# 4. 添加元素
plt.title('正弦函数曲线', fontsize=16, fontweight='bold')  # 标题
plt.xlabel('X轴', fontsize=12)  # X轴标签
plt.ylabel('Y轴', fontsize=12)  # Y轴标签
plt.legend(loc='upper right', fontsize=10)  # 图例
plt.grid(True, alpha=0.3)  # 网格线

# 5. 显示图表
plt.tight_layout()  # 自动调整布局
plt.show()
```

### Figure 和 Axes 对象

Matplotlib 有两种主要的绘图方式：

**方式一：pyplot 接口（简单）**

```python
plt.plot([1, 2, 3], [4, 5, 6])
plt.title('标题')
plt.show()
```

**方式二：面向对象接口（推荐，更灵活）**

```python
# 创建 figure 和 axes 对象
fig, ax = plt.subplots(figsize=(10, 6))

# 通过 axes 对象绘图
ax.plot([1, 2, 3], [4, 5, 6])
ax.set_title('标题')
ax.set_xlabel('X轴')
ax.set_ylabel('Y轴')

plt.show()
```

**推荐使用面向对象接口**，特别是在创建复杂图表或多子图时。

### 图形大小和分辨率

```python
# 设置图形大小（单位：英寸）
plt.figure(figsize=(12, 8))

# 保存时设置分辨率（DPI）
plt.savefig('chart.png', dpi=300, bbox_inches='tight')
# dpi=300 适合打印，dpi=150 适合屏幕显示
# bbox_inches='tight' 去除多余白边
```

## 6.2 常用图表类型

### 折线图（Line Chart）

折线图用于展示数据随时间或其他连续变量的变化趋势。

```python
import matplotlib.pyplot as plt
import numpy as np

# 准备数据
x = np.linspace(0, 10, 100)
y1 = np.sin(x)
y2 = np.cos(x)

# 创建图形
fig, ax = plt.subplots(figsize=(12, 6))

# 绘制多条折线
ax.plot(x, y1, label='sin(x)', color='steelblue', linewidth=2, linestyle='-')
ax.plot(x, y2, label='cos(x)', color='coral', linewidth=2, linestyle='--')

# 添加标题和标签
ax.set_title('三角函数曲线对比', fontsize=16, fontweight='bold', pad=20)
ax.set_xlabel('X轴', fontsize=12)
ax.set_ylabel('Y轴', fontsize=12)

# 添加图例
ax.legend(loc='upper right', fontsize=11, framealpha=0.9)

# 添加网格
ax.grid(True, alpha=0.3, linestyle='--')

# 设置坐标轴范围
ax.set_xlim(0, 10)
ax.set_ylim(-1.5, 1.5)

# 添加参考线
ax.axhline(y=0, color='black', linewidth=0.5)
ax.axvline(x=5, color='gray', linewidth=0.5, linestyle=':')

plt.tight_layout()
plt.savefig('line_chart.png', dpi=300, bbox_inches='tight')
plt.show()
```

**实际案例：股票价格走势**

```python
import pandas as pd

# 模拟股票数据

ax.set_ylabel('价格（元）', fontsize=12)
ax.legend(loc='best', fontsize=11)

plt.show()
```

### 柱状图（Bar Chart）

柱状图用于比较不同类别的数据。

```python
import matplotlib.pyplot as plt
import numpy as np

# 准备数据
categories = ['电子产品', '服装', '食品', '家居', '图书']
sales = [50000, 35000, 28000, 22000, 15000]
colors = ['#FF6B6B', '#4ECDC4', '#45B7D1', '#FFA07A', '#98D8C8']

# 创建图形
fig, ax = plt.subplots(figsize=(12, 6))

# 绘制柱状图
bars = ax.bar(categories, sales, color=colors, edgecolor='black', linewidth=0.5)

# 在柱子上方添加数值标签
for bar, value in zip(bars, sales):
    height = bar.get_height()
    ax.text(bar.get_x() + bar.get_width()/2., height,
            f'{value:,}',
            ha='center', va='bottom', fontsize=10, fontweight='bold')

# 添加标题和标签
ax.set_title('各类产品销售额对比', fontsize=16, fontweight='bold', pad=20)
ax.set_xlabel('产品类别', fontsize=12)
ax.set_ylabel('销售额（元）', fontsize=12)

# 添加网格
ax.yaxis.grid(True, alpha=0.3, linestyle='--')
ax.set_axisbelow(True)  # 网格在柱子下方

plt.tight_layout()
plt.show()
```

### 散点图（Scatter Plot）

散点图用于展示两个变量之间的关系和相关性。

```python
import matplotlib.pyplot as plt
import numpy as np

# 准备数据
np.random.seed(42)
n = 200
x = np.random.rand(n) * 100  # 广告投入
y = 50 + 2 * x + np.random.randn(n) * 50  # 销售额

# 创建图形
fig, ax = plt.subplots(figsize=(12, 8))

# 绘制散点图
scatter = ax.scatter(x, y, c=y, cmap='viridis', alpha=0.6, 
                     edgecolors='black', linewidth=0.5, s=100)

# 添加颜色条
cbar = plt.colorbar(scatter, ax=ax)
cbar.set_label('销售额', fontsize=11)

# 添加标题和标签
ax.set_title('广告投入与销售额关系', fontsize=16, fontweight='bold', pad=20)
ax.set_xlabel('广告投入（万元）', fontsize=12)
ax.set_ylabel('销售额（万元）', fontsize=12)

# 添加趋势线
z = np.polyfit(x, y, 1)
p = np.poly1d(z)
x_line = np.linspace(min(x), max(x), 100)
ax.plot(x_line, p(x_line), "r--", linewidth=2, label=f'趋势线 (y={z[0]:.2f}x+{z[1]:.2f})')

ax.legend(fontsize=11)
ax.grid(True, alpha=0.3, linestyle='--')

plt.tight_layout()
plt.show()
```

### 直方图（Histogram）

直方图用于展示数据的分布情况。

```python
import matplotlib.pyplot as plt
import numpy as np

# 准备数据
np.random.seed(42)
data = np.random.normal(loc=100, scale=15, size=1000)  # 正态分布

# 创建图形
fig, ax = plt.subplots(figsize=(12, 6))

# 绘制直方图
n, bins, patches = ax.hist(data, bins=30, color='steelblue', 
                           edgecolor='black', linewidth=0.5, alpha=0.7)

# 添加理论正态分布曲线
from scipy import stats
x = np.linspace(min(data), max(data), 100)
pdf = stats.norm.pdf(x, np.mean(data), np.std(data))
# 缩放到直方图的高度
scale_factor = len(data) * (max(data) - min(data)) / 30
ax.plot(x, pdf * scale_factor, 'r-', linewidth=2, label='正态分布拟合')

# 添加统计信息
stats_text = f'均值: {np.mean(data):.2f}\n标准差: {np.std(data):.2f}\n样本数: {len(data)}'
ax.text(0.95, 0.95, stats_text, transform=ax.transAxes,
        fontsize=11, verticalalignment='top', horizontalalignment='right',
        bbox=dict(boxstyle='round,pad=0.5', facecolor='yellow', alpha=0.5))

ax.set_title('数据分布直方图', fontsize=16, fontweight='bold', pad=20)
ax.set_xlabel('数值', fontsize=12)
ax.set_ylabel('频数', fontsize=12)
ax.legend(fontsize=11)
ax.grid(True, alpha=0.3, linestyle='--', axis='y')
ax.set_axisbelow(True)

plt.tight_layout()
plt.show()
```

### 饼图（Pie Chart）

饼图用于展示各部分占总体的比例。

```python
import matplotlib.pyplot as plt

# 准备数据
categories = ['电子产品', '服装', '食品', '家居', '图书']
sales = [50000, 35000, 28000, 22000, 15000]
colors = ['#FF6B6B', '#4ECDC4', '#45B7D1', '#FFA07A', '#98D8C8']

# 计算百分比
total = sum(sales)
percentages = [s/total*100 for s in sales]

# 创建图形
fig, ax = plt.subplots(figsize=(10, 8))

# 绘制饼图
wedges, texts, autotexts = ax.pie(sales, labels=categories, autopct='%1.1f%%',
                                    colors=colors, startangle=90,
                                    pctdistance=0.85, wedgeprops=dict(width=0.4, 
                                    edgecolor='white', linewidth=2))

# 设置字体样式
for text in texts:
    text.set_fontsize(11)
for autotext in autotexts:
    autotext.set_fontsize(10)
    autotext.set_fontweight('bold')
    autotext.set_color('white')

# 添加中心文字
centre_circle = plt.Circle((0, 0), 0.60, fc='white')
fig.gca().add_artist(centre_circle)
ax.text(0, 0, f'总销售额\n{total:,}', ha='center', va='center', 
        fontsize=14, fontweight='bold')

ax.set_title('产品销售占比', fontsize=16, fontweight='bold', pad=20)

plt.tight_layout()
plt.show()
```

## 6.3 Seaborn 统计可视化

### 什么是 Seaborn？

Seaborn 是基于 Matplotlib 的高级统计可视化库，提供了更美观的默认样式和更简单的 API。

**主要优势：**
- **美观的默认样式**：无需过多调整即可生成专业图表
- **统计功能集成**：内置统计分析和置信区间
- **简化复杂图表**：轻松创建热力图、配对图等
- **与 Pandas 完美集成**：直接传入 DataFrame

### 安装与基本设置

```python
# 安装
# pip install seaborn

import seaborn as sns
import matplotlib.pyplot as plt
import pandas as pd
import numpy as np

# 设置主题
sns.set_theme(style="whitegrid")  # 白色网格背景
# 其他主题：darkgrid, white, dark, ticks

# 设置调色板
sns.set_palette("husl")

# 设置中文字体
plt.rcParams['font.sans-serif'] = ['SimHei']
plt.rcParams['axes.unicode_minus'] = False
```

### 美化样式

```python
# 不同的主题风格
styles = ['darkgrid', 'whitegrid', 'dark', 'white', 'ticks']

fig, axes = plt.subplots(2, 3, figsize=(15, 10))
axes = axes.flatten()

x = np.linspace(0, 10, 100)
y = np.sin(x)

for i, style in enumerate(styles):
    sns.set_theme(style=style)
    ax = axes[i]
    ax.plot(x, y)
    ax.set_title(f'Style: {style}', fontsize=12)

# 恢复默认主题
sns.set_theme(style="whitegrid")
plt.tight_layout()
plt.show()
```

### 箱线图（Box Plot）

```python
import seaborn as sns
import matplotlib.pyplot as plt

fig, ax = plt.subplots(figsize=(12, 6))
```

### 热力图（Heatmap）

热力图用于展示矩阵数据或相关性。

```python
import seaborn as sns
import matplotlib.pyplot as plt
import numpy as np

# 准备相关性矩阵
np.random.seed(42)
df = pd.DataFrame({
    '变量A': np.random.rand(100),
    '变量B': np.random.rand(100),
    '变量C': np.random.rand(100),
    '变量D': np.random.rand(100),
    '变量E': np.random.rand(100)
})

# 计算相关性矩阵
corr_matrix = df.corr()

# 创建图形
fig, ax = plt.subplots(figsize=(10, 8))

# 绘制热力图
```

## 实战案例：综合数据分析可视化

```python
import matplotlib.pyplot as plt
import seaborn as sns
import pandas as pd
import numpy as np

# 设置样式
sns.set_theme(style="whitegrid")
plt.rcParams['font.sans-serif'] = ['SimHei']
plt.rcParams['axes.unicode_minus'] = False

# 创建综合销售数据
np.random.seed(42)
n = 365
dates = pd.date_range('2024-01-01', periods=n, freq='D')

df = pd.DataFrame({
    '日期': dates,
    '销售额': np.random.randint(1000, 5000, n) + np.arange(n) * 5,
    '订单数': np.random.randint(50, 200, n),
    '客单价': np.random.randint(50, 150, n),
    '类别': np.random.choice(['电子产品', '服装', '食品', '家居'], n)
})

# 创建大型仪表板
fig = plt.figure(figsize=(20, 12))
gs = fig.add_gridspec(3, 3, hspace=0.3, wspace=0.3)

# 1. 销售趋势图
ax1 = fig.add_subplot(gs[0, :2])
df['7日平均'] = df['销售额'].rolling(window=7).mean()
ax1.plot(df['日期'], df['销售额'], alpha=0.3, label='日销售额', color='steelblue')
ax1.plot(df['日期'], df['7日平均'], linewidth=2, label='7日移动平均', color='coral')
ax1.set_title('销售趋势分析', fontsize=14, fontweight='bold')
ax1.set_xlabel('日期')
ax1.set_ylabel('销售额（元）')
ax1.legend()
ax1.grid(True, alpha=0.3)

# 2. 类别占比饼图
ax2 = fig.add_subplot(gs[0, 2])
category_sales = df.groupby('类别')['销售额'].sum()
colors = ['#FF6B6B', '#4ECDC4', '#45B7D1', '#FFA07A']
wedges, texts, autotexts = ax2.pie(category_sales, labels=category_sales.index, 
                                    autopct='%1.1f%%', colors=colors, startangle=90)
for autotext in autotexts:
    autotext.set_fontsize(9)
    autotext.set_color('white')
ax2.set_title('销售类别占比', fontsize=14, fontweight='bold')

# 3. 月度销售柱状图
ax3 = fig.add_subplot(gs[1, :])
df['月份'] = df['日期'].dt.month
monthly_sales = df.groupby('月份')['销售额'].sum().reset_index()
bars = ax3.bar(monthly_sales['月份'], monthly_sales['销售额'], 
               color='steelblue', edgecolor='black', linewidth=0.5)
for bar, val in zip(bars, monthly_sales['销售额']):
    ax3.text(bar.get_x() + bar.get_width()/2., bar.get_height(),
             f'{int(val):,}', ha='center', va='bottom', fontsize=9)
ax3.set_title('月度销售汇总', fontsize=14, fontweight='bold')
ax3.set_xlabel('月份')
ax3.set_ylabel('销售额（元）')
ax3.set_xticks(range(1, 13))
ax3.grid(True, alpha=0.3, axis='y')
ax3.set_axisbelow(True)

# 4. 销售额分布直方图
ax4 = fig.add_subplot(gs[2, 0])
ax4.hist(df['销售额'], bins=30, color='steelblue', edgecolor='black', alpha=0.7)
ax4.set_title('销售额分布', fontsize=12, fontweight='bold')
ax4.set_xlabel('销售额')
ax4.set_ylabel('频数')
ax4.grid(True, alpha=0.3, axis='y')

# 5. 订单数与销售额散点图
ax5 = fig.add_subplot(gs[2, 1])
scatter = ax5.scatter(df['订单数'], df['销售额'], c=df['客单价'], 
                      cmap='viridis', alpha=0.6, edgecolors='black', s=50)
plt.colorbar(scatter, ax=ax5, label='客单价')
ax5.set_title('订单数 vs 销售额', fontsize=12, fontweight='bold')
ax5.set_xlabel('订单数')
ax5.set_ylabel('销售额')
ax5.grid(True, alpha=0.3)

# 6. 类别销售箱线图
ax6 = fig.add_subplot(gs[2, 2])
sns.boxplot(x='类别', y='销售额', data=df, palette='Set2', ax=ax6)
ax6.set_title('各类别销售分布', fontsize=12, fontweight='bold')
ax6.set_xlabel('类别')
ax6.set_ylabel('销售额')
ax6.tick_params(axis='x', rotation=45)
ax6.grid(True, alpha=0.3, axis='y')

fig.suptitle('2024年销售数据分析仪表板', fontsize=18, fontweight='bold', y=0.98)
plt.savefig('dashboard.png', dpi=300, bbox_inches='tight')
plt.show()

print("=== 数据统计摘要 ===")
print(f"总销售额: {df['销售额'].sum():,.0f} 元")
print(f"平均日销售额: {df['销售额'].mean():,.0f} 元")
print(f"总订单数: {df['订单数'].sum():,} 单")
print(f"平均客单价: {df['客单价'].mean():,.0f} 元")
print(f"\n各类别销售:")
print(category_sales.apply(lambda x: f"{x:,.0f} 元").to_string())
```
