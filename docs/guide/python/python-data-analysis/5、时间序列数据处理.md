# 第五章：时间序列数据处理

时间序列数据是按时间顺序排列的数据，在金融、气象、销售等领域广泛应用。本章将学习如何使用 Pandas 处理时间序列数据。

---

## 5.1 日期时间基础：datetime 模块

### Python 标准库 datetime

```python
from datetime import datetime, date, time, timedelta

# 创建日期时间对象
now = datetime.now()
print("当前时间:", now)

today = date.today()
print("今天:", today)

# 创建指定日期
dt = datetime(2024, 1, 15, 10, 30, 0)
print("指定时间:", dt)

# 日期时间属性
print("\n年:", dt.year)
print("月:", dt.month)
print("日:", dt.day)
print("时:", dt.hour)
print("分:", dt.minute)
print("秒:", dt.second)
print("星期:", dt.weekday())  # 0=周一, 6=周日
```

### 日期格式化

```python
from datetime import datetime

dt = datetime(2024, 1, 15, 10, 30, 0)

# 格式化为字符串
formatted = dt.strftime("%Y-%m-%d %H:%M:%S")
print("格式化:", formatted)  # 2024-01-15 10:30:00

formatted2 = dt.strftime("%Y年%m月%d日")
print("中文格式:", formatted2)  # 2024年01月15日

# 常用格式代码
# %Y - 四位年份
# %m - 月份（01-12）
# %d - 日期（01-31）
# %H - 小时（00-23）
# %M - 分钟（00-59）
# %S - 秒（00-59）
# %A - 星期全称
# %B - 月份全称
```

### 字符串解析为日期

```python
from datetime import datetime

# 字符串转日期
date_str = "2024-01-15"
dt = datetime.strptime(date_str, "%Y-%m-%d")
print("解析结果:", dt)

date_str2 = "15/01/2024 10:30"
dt2 = datetime.strptime(date_str2, "%d/%m/%Y %H:%M")
print("解析结果2:", dt2)
```

### 日期运算

```python
from datetime import datetime, timedelta

dt = datetime(2024, 1, 15)

# 加法
next_day = dt + timedelta(days=1)
print("明天:", next_day)

next_week = dt + timedelta(weeks=1)
print("下周:", next_week)

# 减法
yesterday = dt - timedelta(days=1)
print("昨天:", yesterday)

# 计算日期差
delta = next_day - dt
print("相差天数:", delta.days)
```

## 5.2 Pandas 时间类型

### Timestamp

```python
import pandas as pd

# 创建 Timestamp
ts = pd.Timestamp('2024-01-15')
print("Timestamp:", ts)

ts2 = pd.Timestamp('2024-01-15 10:30:00')
print("带时间:", ts2)

# 属性
print("\n年:", ts.year)
print("月:", ts.month)
print("日:", ts.day)
print("星期:", ts.dayofweek)  # 0=周一
print("一年中的第几天:", ts.dayofyear)
```

### DatetimeIndex

```python
# 创建日期范围
dates = pd.date_range('2024-01-01', periods=10, freq='D')
print("日期范围:\n", dates)

# 常用频率
# D - 天
# W - 周
# M - 月末
# MS - 月初
# H - 小时
# T/min - 分钟
# S - 秒

# 按月生成
monthly_dates = pd.date_range('2024-01-01', periods=12, freq='MS')
print("\n月度日期:\n", monthly_dates)

# 按小时生成
hourly_dates = pd.date_range('2024-01-01', periods=24, freq='H')
print("\n小时日期:\n", hourly_dates)
```

### 创建时间序列 DataFrame

```python
# 创建时间序列数据
dates = pd.date_range('2024-01-01', periods=30, freq='D')
df = pd.DataFrame({
    '日期': dates,
    '销售额': range(100, 130),
    '访问量': range(1000, 1030)
})

# 设置日期为索引
df = df.set_index('日期')

```

## 5.3 字符串转日期：pd.to_datetime

### 基本转换

```python
import pandas as pd

# 字符串列表转日期
dates_str = ['2024-01-01', '2024-01-02', '2024-01-03']
dates = pd.to_datetime(dates_str)
print("转换结果:", dates)

# DataFrame 中的列转换
df = pd.DataFrame({
    '日期': ['2024-01-01', '2024-01-02', '2024-01-03'],
    '值': [100, 200, 300]
})

df['日期'] = pd.to_datetime(df['日期'])
print("\n转换后类型:", df['日期'].dtype)
```

### 处理不同格式

```python
# 不同格式的日期
dates_mixed = ['01/15/2024', '2024-01-16', 'Jan 17, 2024']
dates = pd.to_datetime(dates_mixed)
print("混合格式转换:", dates)

# 指定格式
dates_custom = ['20240115', '20240116', '20240117']
dates = pd.to_datetime(dates_custom, format='%Y%m%d')
print("自定义格式:", dates)

# 处理错误
dates_with_error = ['2024-01-01', 'invalid', '2024-01-03']
dates = pd.to_datetime(dates_with_error, errors='coerce')  # 错误转为 NaT
print("含错误处理:", dates)
```

### 从多列组合日期

```python
df = pd.DataFrame({
    '年': [2024, 2024, 2024],
    '月': [1, 2, 3],
    '日': [15, 20, 25]
})

df['日期'] = pd.to_datetime(df[['年', '月', '日']])
print("组合日期:\n", df)
```

## 5.4 日期提取

### 提取日期组件

```python
import pandas as pd

# 创建示例数据
dates = pd.date_range('2024-01-01', periods=10, freq='D')
df = pd.DataFrame({'日期': dates, '值': range(10)})

# 提取各种组件
df['年'] = df['日期'].dt.year
df['月'] = df['日期'].dt.month
df['日'] = df['日期'].dt.day
df['星期'] = df['日期'].dt.dayofweek  # 0=周一
df['星期名'] = df['日期'].dt.day_name()
df['月份名'] = df['日期'].dt.month_name()
df['一年中的第几天'] = df['日期'].dt.dayofyear
df['季度'] = df['日期'].dt.quarter

print("日期组件提取:\n", df)
```

### 工作日判断

```python
# 判断是否为工作日
df['是否工作日'] = df['日期'].dt.weekday < 5  # 0-4为工作日

# 获取 week of year
df['年第几周'] = df['日期'].dt.isocalendar().week

print("\n包含工作日信息:\n", df)
```

## 5.5 时间重采样：resample

### 降采样（高频到低频）

```python
import pandas as pd
import numpy as np

# 创建日度数据
dates = pd.date_range('2024-01-01', periods=90, freq='D')
df = pd.DataFrame({
    '日期': dates,
    '销售额': np.random.randint(100, 500, 90)
})
df = df.set_index('日期')

print("原始日度数据:\n", df.head())

# 重采样为月度
monthly = df.resample('M').sum()
print("\n月度汇总:\n", monthly.head())

# 重采样为周度
weekly = df.resample('W').mean()
print("\n周度平均:\n", weekly.head())

# 重采样为季度
quarterly = df.resample('Q').agg({
    '销售额': ['sum', 'mean', 'std']
})
print("\n季度统计:\n", quarterly.head())
```

### 升采样（低频到高频）

```python
# 创建月度数据
dates = pd.date_range('2024-01-01', periods=12, freq='MS')
df_monthly = pd.DataFrame({
    '月份': dates,
    '销售额': [1000, 1200, 1100, 1300, 1400, 1500, 
               1600, 1700, 1500, 1400, 1300, 1200]
})
df_monthly = df_monthly.set_index('月份')

print("月度数据:\n", df_monthly.head())

# 重采样为日度（向前填充）
daily_ffill = df_monthly.resample('D').ffill()
print("\n日度数据（向前填充）:\n", daily_ffill.head(10))

# 重采样为日度（线性插值）
daily_interp = df_monthly.resample('D').interpolate()
print("\n日度数据（插值）:\n", daily_interp.head(10))
```

### 常用重采样频率

```python
# 频率别名
# D - 天
# W - 周
# M - 月末
# MS - 月初
# Q - 季末
# QS - 季初
# Y - 年末
# YS - 年初
# H - 小时
# T/min - 分钟
# S - 秒

# 复合频率
# 2D - 每2天
# 3H - 每3小时
# W-MON - 每周星期一
```

## 5.6 时间窗口计算：rolling

### 移动平均

```python
import pandas as pd
import numpy as np

# 创建示例数据
dates = pd.date_range('2024-01-01', periods=30, freq='D')
df = pd.DataFrame({
    '日期': dates,
    '价格': np.random.randint(100, 200, 30)
})
df = df.set_index('日期')

# 计算7日移动平均
df['7日平均'] = df['价格'].rolling(window=7).mean()

# 计算30日移动平均
df['30日平均'] = df['价格'].rolling(window=30).min_periods(1).mean()

print("移动平均:\n", df.head(15))
```

### 其他窗口函数

```python
# 移动总和
df['7日总和'] = df['价格'].rolling(window=7).sum()

# 移动标准差
df['7日标准差'] = df['价格'].rolling(window=7).std()

# 移动最大值
df['7日最大'] = df['价格'].rolling(window=7).max()

# 移动最小值
df['7日最小'] = df['价格'].rolling(window=7).min()

print("\n窗口统计:\n", df.head(15))
```

### 指数加权移动平均

```python
# EMA（更重视近期数据）
df['EMA_7'] = df['价格'].ewm(span=7, adjust=False).mean()
print("指数加权平均:\n", df.head(15))
```

## 5.7 时区处理

### 添加时区

```python
import pandas as pd

# 创建无时区的时间
dates = pd.date_range('2024-01-01', periods=5, freq='D')
print("无时区:", dates)

# 添加时区
dates_utc = dates.tz_localize('UTC')
print("\nUTC时区:", dates_utc)

dates_beijing = dates.tz_localize('Asia/Shanghai')
print("\n北京时区:", dates_beijing)
```

### 时区转换

```python
# UTC 转北京时间
dates_utc = pd.date_range('2024-01-01', periods=5, freq='D', tz='UTC')
dates_beijing = dates_utc.tz_convert('Asia/Shanghai')

print("UTC时间:", dates_utc)
print("\n北京时间:", dates_beijing)

# 北京转纽约
dates_ny = dates_beijing.tz_convert('America/New_York')
print("\n纽约时间:", dates_ny)
```

## 实战案例：股票数据分析

```python
import pandas as pd
import numpy as np

# 模拟股票数据
np.random.seed(42)
dates = pd.date_range('2024-01-01', periods=100, freq='D')
prices = 100 + np.cumsum(np.random.randn(100) * 2)

df = pd.DataFrame({
    '日期': dates,
    '收盘价': prices
})
df = df.set_index('日期')

# 1. 计算收益率
df['日收益率'] = df['收盘价'].pct_change()

# 2. 计算移动平均
df['MA5'] = df['收盘价'].rolling(window=5).mean()
df['MA20'] = df['收盘价'].rolling(window=20).mean()

# 3. 计算波动率（20日）
df['波动率'] = df['日收益率'].rolling(window=20).std()

# 4. 月度汇总
monthly = df.resample('M').agg({
    '收盘价': ['first', 'last', 'max', 'min'],
    '日收益率': ['mean', 'std']
})

print("=== 股票数据分析 ===\n")
print("最新数据:\n", df.tail())
print("\n月度汇总:\n", monthly.head())
print(f"\n总收益率: {(df['收盘价'].iloc[-1] / df['收盘价'].iloc[0] - 1) * 100:.2f}%")
print(f"平均日收益率: {df['日收益率'].mean() * 100:.4f}%")
print(f"年化波动率: {df['波动率'].iloc[-1] * np.sqrt(252) * 100:.2f}%")
```

import pandas as pd
