本章将学习使用 asyncio 和 aiohttp 构建高性能的异步爬虫，大幅提升爬取效率。

## 5.1 同步 vs 异步

### 同步爬虫的问题

```python
import requests
import time

urls = ['https://example.com/page1', 'https://example.com/page2', ...]

start = time.time()

# 串行请求 - 很慢！
for url in urls:
    response = requests.get(url)
    process(response)

end = time.time()
print(f"耗时: {end - start}秒")  # 假设每个请求 1 秒，100 个 URL 需要 100 秒
```

### 异步爬虫的优势

```python
import aiohttp
import asyncio
import time

async def fetch(session, url):
    async with session.get(url) as response:
        return await response.text()

async def main():
    urls = ['https://example.com/page1', ...]
    
    start = time.time()
    
    async with aiohttp.ClientSession() as session:
        tasks = [fetch(session, url) for url in urls]
        results = await asyncio.gather(*tasks)
    
    end = time.time()
    print(f"耗时: {end - start}秒")  # 可能只需 2-3 秒！

asyncio.run(main())
```

**性能对比：**
- 同步：100 个 URL × 1 秒 = 100 秒
- 异步：并发执行 ≈ 2-3 秒
- **提升 30-50 倍！**

## 5.2 asyncio 基础

### 协程（Coroutine）

```python
import asyncio

# 定义协程函数
async def say_hello():
    print("Hello")
    await asyncio.sleep(1)  # 模拟异步操作
    print("World")

# 运行协程
asyncio.run(say_hello())
```

### 任务（Task）

```python
async def main():
    # 创建任务
    task1 = asyncio.create_task(say_hello())
    task2 = asyncio.create_task(say_hello())
    
    # 等待完成
    await task1
    await task2

asyncio.run(main())
```

### 并发执行

```python
async def main():
    tasks = [say_hello() for _ in range(10)]
    
    # 并发执行所有任务
    await asyncio.gather(*tasks)

asyncio.run(main())
```

### 获取返回值

```python
async def fetch_data(url):
    await asyncio.sleep(1)
    return f"Data from {url}"

async def main():
    results = await asyncio.gather(
        fetch_data('url1'),
        fetch_data('url2'),
        fetch_data('url3')
    )
    print(results)  # ['Data from url1', 'Data from url2', 'Data from url3']

asyncio.run(main())
```

## 5.4 基本使用

### GET 请求

```python
import aiohttp
import asyncio

async def fetch_url(url):
    async with aiohttp.ClientSession() as session:
        async with session.get(url) as response:
            print(f"Status: {response.status}")
            text = await response.text()
            return text

async def main():
    html = await fetch_url('https://example.com')
    print(html[:200])

asyncio.run(main())
```

### POST 请求

```python
async def post_data(url, data):
    async with aiohttp.ClientSession() as session:
        async with session.post(url, json=data) as response:
            return await response.json()

async def main():
    result = await post_data(
        'https://api.example.com/data',
        {'key': 'value'}
    )
    print(result)

asyncio.run(main())
```

### 设置请求头

```python
headers = {
    'User-Agent': 'Mozilla/5.0',
    'Accept': 'text/html',
}

async with aiohttp.ClientSession(headers=headers) as session:
    async with session.get(url) as response:
        ...
```

### 超时设置

```python
timeout = aiohttp.ClientTimeout(total=30)

async with aiohttp.ClientSession(timeout=timeout) as session:
    async with session.get(url) as response:
        ...
```

## 5.5 并发控制

### Semaphore（信号量）

```python
import asyncio
import aiohttp

async def fetch_with_limit(session, url, semaphore):
    async with semaphore:  # 限制并发数
        async with session.get(url) as response:
            return await response.text()

async def main():
    urls = [f'https://example.com/page{i}' for i in range(100)]
    
    # 最多同时 10 个请求
    semaphore = asyncio.Semaphore(10)
    
    async with aiohttp.ClientSession() as session:
        tasks = [
            fetch_with_limit(session, url, semaphore)
            for url in urls
        ]
        results = await asyncio.gather(*tasks)
    
    print(f"完成 {len(results)} 个请求")

asyncio.run(main())
```

### 批量处理

```python
async def process_batch(urls, batch_size=10):
    """分批处理"""
    results = []
    
    async with aiohttp.ClientSession() as session:
        for i in range(0, len(urls), batch_size):
            batch = urls[i:i + batch_size]
            
            tasks = [session.get(url) for url in batch]
            responses = await asyncio.gather(*tasks)
            
            for response in responses:
                results.append(await response.text())
            
            print(f"完成批次 {i // batch_size + 1}")
    
    return results
```

## 5.6 错误处理

### 异常捕获

```python
async def safe_fetch(session, url):
    try:
        async with session.get(url, timeout=10) as response:
            if response.status == 200:
                return await response.text()
            else:
                print(f"HTTP {response.status}: {url}")
                return None
    
    except aiohttp.ClientError as e:
        print(f"客户端错误 {url}: {e}")
        return None
    
    except asyncio.TimeoutError:
        print(f"超时 {url}")
        return None
    
    except Exception as e:
        print(f"未知错误 {url}: {e}")
        return None
```

### 重试机制

```python
async def fetch_with_retry(session, url, max_retries=3):
    for attempt in range(max_retries):
        try:
            async with session.get(url, timeout=10) as response:
                if response.status == 200:
                    return await response.text()
                elif response.status == 429:  # 频率限制
                    wait_time = 2 ** attempt
                    print(f"被限流，等待 {wait_time} 秒")
                    await asyncio.sleep(wait_time)
                else:
                    print(f"HTTP {response.status}")
                    return None
        
        except (aiohttp.ClientError, asyncio.TimeoutError) as e:
            if attempt == max_retries - 1:
                print(f"重试失败: {url}")
                return None
            
            wait_time = 2 ** attempt
            print(f"重试 {attempt + 1}/{max_retries}: {url}")
            await asyncio.sleep(wait_time)
    
    return None
```

## 5.7 实战：异步新闻爬虫

### 完整实现

```python
import aiohttp
import asyncio
from bs4 import BeautifulSoup
import json
import time
from typing import List, Dict

class AsyncNewsScraper:
    """异步新闻爬虫"""
    
    def __init__(self, max_concurrent=10):
        self.max_concurrent = max_concurrent
        self.semaphore = asyncio.Semaphore(max_concurrent)
        self.results = []
    
    async def fetch_page(self, session: aiohttp.ClientSession, url: str) -> str:
        """获取页面内容"""
        async with self.semaphore:
            try:
                headers = {
                    'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) '
                                 'AppleWebKit/537.36'
                }
                
                async with session.get(
                    url, 
                    headers=headers,
                    timeout=aiohttp.ClientTimeout(total=30)
                ) as response:
                    if response.status == 200:
                        return await response.text()
                    else:
                        print(f"HTTP {response.status}: {url}")
                        return None
            
            except Exception as e:
                print(f"请求失败 {url}: {e}")
                return None
    
    def parse_news_list(self, html: str) -> List[Dict]:
        """解析新闻列表"""
        if not html:
            return []
        
        soup = BeautifulSoup(html, 'lxml')
        news_list = []
        
        for article in soup.select('article.news-item'):
            title_tag = article.select_one('h2.title a')
            if not title_tag:
                continue
            
            title = title_tag.get_text(strip=True)
            url = title_tag['href']
            
            summary_tag = article.select_one('p.summary')
            summary = summary_tag.get_text(strip=True) if summary_tag else ''
            
            time_tag = article.select_one('time')
            publish_time = time_tag.get_text(strip=True) if time_tag else ''
            
            news_list.append({
                'title': title,
                'url': url,
                'summary': summary,
                'publish_time': publish_time
            })
        
        return news_list
    
    async def scrape_page(self, session: aiohttp.ClientSession, url: str) -> List[Dict]:
        """爬取单页"""
        html = await self.fetch_page(session, url)
        return self.parse_news_list(html)
    
    async def scrape_multiple_pages(self, base_url: str, pages: int = 5) -> List[Dict]:
        """爬取多页"""
        urls = [
            f"{base_url}/page/{i}" if i > 1 else base_url
            for i in range(1, pages + 1)
        ]
        
        async with aiohttp.ClientSession() as session:
            tasks = [self.scrape_page(session, url) for url in urls]
            results = await asyncio.gather(*tasks)
        
        # 合并结果
        all_news = []
        for news_list in results:
            all_news.extend(news_list)
        
        return all_news
    
    async def scrape_detail(self, session: aiohttp.ClientSession, url: str) -> Dict:
        """爬取详情"""
        html = await self.fetch_page(session, url)
        if not html:
            return {}
        
        soup = BeautifulSoup(html, 'lxml')
        
        title = soup.select_one('h1.article-title')
        title = title.get_text(strip=True) if title else ''
        
        content_div = soup.select_one('div.article-content')
        if content_div:
            paragraphs = content_div.find_all('p')
            content = '\n'.join([p.get_text(strip=True) for p in paragraphs])
        else:
            content = ''
        
        return {
            'title': title,
            'content': content,
            'url': url
        }
    
    async def scrape_details(self, news_list: List[Dict], max_concurrent=5) -> List[Dict]:
        """批量爬取详情"""
        semaphore = asyncio.Semaphore(max_concurrent)
        
        async def fetch_detail(url):
            async with semaphore:
                async with aiohttp.ClientSession() as session:
                    return await self.scrape_detail(session, url)
        
        tasks = [fetch_detail(news['url']) for news in news_list[:20]]  # 限制数量
        details = await asyncio.gather(*tasks)
        
        return [d for d in details if d]
    
    def save_to_json(self, data: List[Dict], filename: str = 'news.json'):
        """保存到 JSON"""
        with open(filename, 'w', encoding='utf-8') as f:
            json.dump(data, f, ensure_ascii=False, indent=2)
        print(f"数据已保存到 {filename}")

async def main():
    scraper = AsyncNewsScraper(max_concurrent=10)
    
    start = time.time()
    
    # 爬取列表
    print("开始爬取新闻列表...")
    news_list = await scraper.scrape_multiple_pages(
        'https://news.example.com',
        pages=5
    )
    print(f"获取 {len(news_list)} 条新闻\n")
    
    # 爬取详情
    print("开始爬取新闻详情...")
    details = await scraper.scrape_details(news_list)
    print(f"获取 {len(details)} 条详情\n")
    
    # 保存
    scraper.save_to_json(news_list, 'news_list.json')
    scraper.save_to_json(details, 'news_details.json')
    
    end = time.time()
    print(f"总耗时: {end - start:.2f} 秒")

if __name__ == '__main__':
    asyncio.run(main())
```

## 5.8 生产者-消费者模式

```python
import asyncio
import aiohttp
from asyncio import Queue

async def producer(queue: Queue, urls: list):
    """生产者：生成 URL"""
    for url in urls:
        await queue.put(url)
        print(f"生产: {url}")
    
    # 添加结束标记
    for _ in range(3):  # 消费者数量
        await queue.put(None)

async def consumer(queue: Queue, session: aiohttp.ClientSession, consumer_id: int):
    """消费者：处理 URL"""
    while True:
        url = await queue.get()
        
        if url is None:
            break
        
        try:
            async with session.get(url) as response:
                text = await response.text()
                print(f"消费者 {consumer_id}: 完成 {url}")
        except Exception as e:
            print(f"消费者 {consumer_id}: 错误 {e}")
        
        queue.task_done()

async def main():
    urls = [f'https://example.com/page{i}' for i in range(100)]
    queue = Queue(maxsize=20)
    
    async with aiohttp.ClientSession() as session:
        # 启动消费者
        consumers = [
            asyncio.create_task(consumer(queue, session, i))
            for i in range(3)
        ]
        
        # 启动生产者
        await producer(queue, urls)
        
        # 等待队列处理完毕
        await queue.join()
        
        # 等待消费者完成
        await asyncio.gather(*consumers)

asyncio.run(main())
```

## 5.9 性能对比测试

```python
import requests
import aiohttp
import asyncio
import time

# 同步版本
def sync_crawler(urls):
    start = time.time()
    for url in urls:
        requests.get(url)
    end = time.time()
    return end - start

# 异步版本
async def async_crawler(urls):
    start = time.time()
    
    async with aiohttp.ClientSession() as session:
        tasks = [session.get(url) for url in urls]
        await asyncio.gather(*tasks)
    
    end = time.time()
    return end - start

# 测试
urls = [f'https://httpbin.org/delay/1' for _ in range(20)]

sync_time = sync_crawler(urls)
print(f"同步耗时: {sync_time:.2f} 秒")

async_time = asyncio.run(async_crawler(urls))
print(f"异步耗时: {async_time:.2f} 秒")

print(f"加速比: {sync_time / async_time:.2f}x")
```
