## 13.2 微博爬虫

### Weibo Spider

```python
import scrapy
import json
from datetime import datetime

class WeiboSpider(scrapy.Spider):
    name = 'weibo'
    
    def __init__(self, user_id='1234567890', *args, **kwargs):
        super().__init__(*args, **kwargs)
        self.user_id = user_id
        self.base_url = f'https://m.weibo.cn/api/container/getIndex?type=uid&value={user_id}'
    
    def start_requests(self):
        yield scrapy.Request(
            self.base_url,
            callback=self.parse_user_info,
            headers={'User-Agent': 'Mozilla/5.0 ...'}
        )
    
    def parse_user_info(self, response):
        """解析用户信息"""
        data = json.loads(response.text)
        
        if data.get('ok') == 1:
            containerid = data['data']['tabsInfo']['tabs'][1]['containerid']
            
            # 获取用户信息
            userinfo = data['data']['userInfo']
            
            yield {
                'type': 'user',
                'user_id': self.user_id,
                'nickname': userinfo.get('screen_name'),
                'followers': userinfo.get('followers_count'),
                'following': userinfo.get('follow_count'),
                'description': userinfo.get('description'),
            }
            
            # 获取微博列表
            yield scrapy.Request(
                f'{self.base_url}&containerid={containerid}',
                callback=self.parse_weibos
            )
    
    def parse_weibos(self, response):
        """解析微博"""
        data = json.loads(response.text)
        
        if data.get('ok') == 1:
            cards = data['data']['cards']
            
            for card in cards:
                if card.get('card_type') == 9:
                    mblog = card.get('mblog')
                    
                    if mblog:
                        yield {
                            'type': 'weibo',
                            'weibo_id': mblog.get('id'),
                            'text': mblog.get('text'),
                            'created_at': mblog.get('created_at'),
                            'reposts_count': mblog.get('reposts_count', 0),
                            'comments_count': mblog.get('comments_count', 0),
                            'attitudes_count': mblog.get('attitudes_count', 0),
                        }
            
            # 分页
            page_info = data['data'].get('cardlistInfo', {})
            since_id = page_info.get('since_id')
            
            if since_id:
                yield scrapy.Request(
                    f'{response.url}&since_id={since_id}',
                    callback=self.parse_weibos
                )
```

```python
import tweepy

class TwitterCrawler:
    def __init__(self, bearer_token):
        self.client = tweepy.Client(bearer_token=bearer_token)
    
    def get_user_tweets(self, user_id, max_results=100):
        """获取用户推文"""
        tweets = self.client.get_users_tweets(
            id=user_id,
            max_results=max_results,
            tweet_fields=['created_at', 'public_metrics', 'text']
        )
        
        results = []
        if tweets.data:
            for tweet in tweets.data:
                results.append({
                    'id': tweet.id,
                    'text': tweet.text,
                    'created_at': tweet.created_at,
                    'retweet_count': tweet.public_metrics['retweet_count'],
                    'like_count': tweet.public_metrics['like_count'],
                })
        
        return results
    
    def search_tweets(self, query, max_results=100):
        """搜索推文"""
        tweets = self.client.search_recent_tweets(
            query=query,
            max_results=max_results,
            tweet_fields=['created_at', 'author_id', 'public_metrics']
        )
        
        results = []
        if tweets.data:
            for tweet in tweets.data:
                results.append({
                    'id': tweet.id,
                    'text': tweet.text,
                    'author_id': tweet.author_id,
                    'created_at': tweet.created_at,
                })
        
        return results
```

```python
from snownlp import SnowNLP

def analyze_sentiment(text):
    """情感分析"""
    s = SnowNLP(text)
    
    return {
        'sentiment': s.sentiments,  # 0-1，越接近 1 越积极
        'keywords': s.keywords(5),  # 关键词
        'summary': s.summary(3),    # 摘要
    }

# 使用
text = "这个产品非常好用，我很满意！"
result = analyze_sentiment(text)
print(f"情感得分: {result['sentiment']}")
```

## 13.5 关系网络分析

### 使用 NetworkX

```python
import networkx as nx
import matplotlib.pyplot as plt

class SocialNetworkAnalyzer:
    def __init__(self):
        self.graph = nx.DiGraph()
    
    def add_relationship(self, user1, user2, relationship='follow'):
        """添加关系"""
        self.graph.add_edge(user1, user2, type=relationship)
    
    def visualize_network(self, filename='network.png'):
        """可视化网络"""
        plt.figure(figsize=(12, 12))
        
        pos = nx.spring_layout(self.graph)
        nx.draw_networkx_nodes(self.graph, pos, node_size=500)
        nx.draw_networkx_edges(self.graph, pos, arrowstyle='->')
        nx.draw_networkx_labels(self.graph, pos, font_size=8)
        
        plt.axis('off')
        plt.savefig(filename, dpi=300, bbox_inches='tight')
        plt.show()
    
    def find_influencers(self, top_n=10):
        """查找意见领袖"""
        # 基于 PageRank
        pagerank = nx.pagerank(self.graph)
        
        # 排序
        sorted_users = sorted(
            pagerank.items(),
            key=lambda x: x[1],
            reverse=True
        )[:top_n]
        
        return sorted_users
    
    def detect_communities(self):
        """检测社区"""
        communities = nx.community.girvan_newman(self.graph)
        top_level_communities = next(communities)
        
        return [list(c) for c in top_level_communities]
```

```python
from wordcloud import WordCloud
import jieba

def generate_wordcloud(texts, filename='wordcloud.png'):
    """生成词云"""
    # 中文分词
    text = ' '.join([jieba.cut(t) for t in texts])
    
    # 生成词云

        font_path='simhei.ttf',  # 中文字体
        width=800,
        height=600,
        background_color='white'
    ).generate(text)
    
    # 保存
    wc.to_file(filename)
```

### 时间序列分析

```python
import pandas as pd
import matplotlib.pyplot as plt

def plot_activity_timeline(posts):
    """绘制活动时间线"""
    df = pd.DataFrame(posts)
    df['created_at'] = pd.to_datetime(df['created_at'])
    df.set_index('created_at', inplace=True)
    
    # 按小时统计
    hourly = df.resample('H').size()
    
    plt.figure(figsize=(15, 6))
    plt.plot(hourly.index, hourly.values)
    plt.title('Posting Activity Over Time')
    plt.xlabel('Time')
    plt.ylabel('Number of Posts')
    plt.grid(True)
    plt.savefig('activity_timeline.png')
    plt.show()
```

1. **遵守平台条款** - 阅读并遵守服务条款
