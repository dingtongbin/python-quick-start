## 10.1 CSV 文件存储

### 基本写入

```python
import csv

data = [
    {'name': 'Alice', 'age': 25, 'city': 'Beijing'},
    {'name': 'Bob', 'age': 30, 'city': 'Shanghai'},
]

with open('data.csv', 'w', newline='', encoding='utf-8-sig') as f:
    writer = csv.DictWriter(f, fieldnames=['name', 'age', 'city'])
    writer.writeheader()
    writer.writerows(data)
```

### 追加模式

```python
def append_to_csv(data, filename='data.csv'):
    """追加数据到 CSV"""
    file_exists = os.path.exists(filename)
    
    with open(filename, 'a', newline='', encoding='utf-8-sig') as f:
        writer = csv.DictWriter(f, fieldnames=['name', 'age', 'city'])
        
        if not file_exists:
            writer.writeheader()
        
        writer.writerows(data)
```

### 读取 CSV

```python
def read_csv(filename='data.csv'):
    """读取 CSV"""
    data = []
    
    with open(filename, 'r', encoding='utf-8-sig') as f:
        reader = csv.DictReader(f)
        for row in reader:
            data.append(dict(row))
    
    return data
```

```python
data = [
```

```python
# 每行一个 JSON 对象（适合大数据）
with open('data.jl', 'w', encoding='utf-8') as f:
    for item in data:
        f.write(json.dumps(item, ensure_ascii=False) + '\n')

# 读取
def read_jsonl(filename):
    data = []
    with open(filename, 'r', encoding='utf-8') as f:
        for line in f:
            data.append(json.loads(line))
    return data
```

## 10.3 MySQL 数据库

### 安装

```bash
pip install pymysql==1.1.0
```

### 基本操作

```python
import pymysql

class MySQLStorage:
    def __init__(self, host='localhost', user='root', 
                 password='password', database='crawler_db'):
        self.connection = pymysql.connect(
            host=host,
            user=user,
            password=password,
            database=database,
            charset='utf8mb4',
            cursorclass=pymysql.cursors.DictCursor
        )
        self.cursor = self.connection.cursor()
    
    def create_table(self):
        """创建表"""
        sql = """
        CREATE TABLE IF NOT EXISTS news (
            id INT AUTO_INCREMENT PRIMARY KEY,
            title VARCHAR(500),
            url VARCHAR(1000) UNIQUE,
            author VARCHAR(100),
            publish_time DATETIME,
            content TEXT,
            created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
        ) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4
        """
        self.cursor.execute(sql)
        self.connection.commit()
    
    def insert_item(self, item):
        """插入数据"""
        sql = """
        INSERT INTO news (title, url, author, publish_time, content)
        VALUES (%s, %s, %s, %s, %s)
        ON DUPLICATE KEY UPDATE title=VALUES(title)
        """
        
        self.cursor.execute(sql, (
            item.get('title'),
            item.get('url'),
            item.get('author'),
            item.get('publish_time'),
            item.get('content')
        ))
        self.connection.commit()
    
    def insert_many(self, items):
        """批量插入"""
        sql = """
        INSERT INTO news (title, url, author, publish_time, content)
        VALUES (%s, %s, %s, %s, %s)
        ON DUPLICATE KEY UPDATE title=VALUES(title)
        """
        
        values = [
            (
                item.get('title'),
                item.get('url'),
                item.get('author'),
                item.get('publish_time'),
                item.get('content')
            )
            for item in items
        ]
        
        self.cursor.executemany(sql, values)
        self.connection.commit()
    
    def query(self, sql, params=None):
        """查询"""
        self.cursor.execute(sql, params or ())
        return self.cursor.fetchall()
    
    def close(self):
        """关闭连接"""
        self.cursor.close()
        self.connection.close()

# 使用
storage = MySQLStorage()
storage.create_table()

item = {
    'title': '新闻标题',
    'url': 'http://example.com/news/1',
    'author': '作者',
    'publish_time': '2024-01-15 10:00:00',
    'content': '新闻内容'
}

storage.insert_item(item)
storage.close()
```

## 10.4 MongoDB 文档数据库

### 安装

```bash
pip install pymongo==4.6.0
```

### 基本操作

```python
from pymongo import MongoClient
from pymongo.errors import DuplicateKeyError

class MongoStorage:
    def __init__(self, uri='mongodb://localhost:27017/', 
                 database='crawler_db'):
        self.client = MongoClient(uri)
        self.db = self.client[database]
    
    def get_collection(self, collection_name):
        """获取集合"""
        return self.db[collection_name]
    
    def insert_one(self, collection_name, document):
        """插入单个文档"""
        collection = self.get_collection(collection_name)
        try:
            result = collection.insert_one(document)
            return result.inserted_id
        except DuplicateKeyError:
            print("文档已存在")
            return None
    
    def insert_many(self, collection_name, documents):
        """批量插入"""
        collection = self.get_collection(collection_name)
        result = collection.insert_many(documents, ordered=False)
        return result.inserted_ids
    
    def find(self, collection_name, filter_dict=None):
        """查询"""
        collection = self.get_collection(collection_name)
        return list(collection.find(filter_dict or {}))
    
    def update_one(self, collection_name, filter_dict, update_dict):
        """更新"""
        collection = self.get_collection(collection_name)
        return collection.update_one(filter_dict, {'$set': update_dict})
    
    def delete_one(self, collection_name, filter_dict):
        """删除"""
        collection = self.get_collection(collection_name)
        return collection.delete_one(filter_dict)
    
    def create_index(self, collection_name, field, unique=False):
        """创建索引"""
        collection = self.get_collection(collection_name)
        collection.create_index(field, unique=unique)
    
    def close(self):
        """关闭连接"""
        self.client.close()

# 使用
storage = MongoStorage()

# 创建索引
storage.create_index('news', 'url', unique=True)

# 插入数据
item = {
    'title': '新闻标题',
    'url': 'http://example.com/news/1',
    'author': '作者',
    'publish_time': '2024-01-15',
    'tags': ['科技', 'AI']
}

storage.insert_one('news', item)

# 查询
results = storage.find('news', {'author': '作者'})

storage.close()
```

## 10.5 Redis 缓存

### 安装

```bash
pip install redis==5.0.8
```

### 基本操作

```python
import redis
import json

class RedisStorage:
    def __init__(self, host='localhost', port=6379, db=0):
        self.redis = redis.Redis(
            host=host,
            port=port,
            db=db,
            decode_responses=True
        )
    
    # 字符串
    def set_string(self, key, value, expire=None):
        """设置字符串"""
        if expire:
            self.redis.setex(key, expire, value)
        else:
            self.redis.set(key, value)
    
    def get_string(self, key):
        """获取字符串"""
        return self.redis.get(key)
    
    # 哈希
    def set_hash(self, name, key, value):
        """设置哈希字段"""
        self.redis.hset(name, key, value)
    
    def get_hash(self, name, key=None):
        """获取哈希"""
        if key:
            return self.redis.hget(name, key)
        return self.redis.hgetall(name)
    
    # 列表
    def push_list(self, name, *values):
        """推入列表"""
        self.redis.rpush(name, *values)
    
    def pop_list(self, name):
        """弹出列表"""
        return self.redis.lpop(name)
    
    def get_list(self, name, start=0, end=-1):
        """获取列表"""
        return self.redis.lrange(name, start, end)
    
    # 集合
    def add_set(self, name, *values):
        """添加到集合"""
        self.redis.sadd(name, *values)
    
    def is_member(self, name, value):
        """检查成员"""
        return self.redis.sismember(name, value)
    
    # 有序集合
    def add_zset(self, name, mapping):
        """添加到有序集合"""
        self.redis.zadd(name, mapping)
    
    def get_zset(self, name, start=0, end=-1):
        """获取有序集合"""
        return self.redis.zrange(name, start, end, withscores=True)
    
    # 去重队列
    def add_to_seen(self, url):
        """添加到已访问"""
        self.redis.sadd('seen_urls', url)
    
    def is_seen(self, url):
        """检查是否已访问"""
        return self.redis.sismember('seen_urls', url)
    
    # 任务队列
    def push_task(self, task):
        """推送任务"""
        self.redis.lpush('task_queue', json.dumps(task))
    
    def pop_task(self):
        """弹出任务"""
        task = self.redis.rpop('task_queue')
        return json.loads(task) if task else None

# 使用
storage = RedisStorage()

# 缓存
storage.set_string('cache:key', 'value', expire=3600)
value = storage.get_string('cache:key')

# 去重
if not storage.is_seen('http://example.com'):
    storage.add_to_seen('http://example.com')
    # 爬取...

# 队列
storage.push_task({'url': 'http://example.com'})
task = storage.pop_task()
```

```python
from elasticsearch import Elasticsearch

class ESStorage:
    def __init__(self, hosts=['localhost:9200']):
        self.es = Elasticsearch(hosts)
    
    def create_index(self, index_name, mappings):
        """创建索引"""
        if not self.es.indices.exists(index=index_name):
            self.es.indices.create(index=index_name, mappings=mappings)
    
    def insert_document(self, index_name, doc_id, document):
        """插入文档"""
        self.es.index(index=index_name, id=doc_id, document=document)
    
    def search(self, index_name, query):
        """搜索"""
        result = self.es.search(index=index_name, body=query)
        return result['hits']['hits']
    
    def delete_index(self, index_name):
        """删除索引"""
        self.es.indices.delete(index=index_name)

# 使用
storage = ESStorage()

# 创建索引
mappings = {
    "properties": {
        "title": {"type": "text"},
        "content": {"type": "text"},
        "publish_time": {"type": "date"}
    }
}
storage.create_index('news', mappings)

# 插入文档
doc = {
    'title': '新闻标题',
    'content': '新闻内容',
    'publish_time': '2024-01-15'
}
storage.insert_document('news', '1', doc)

# 搜索
query = {
    "query": {
        "match": {
            "title": "新闻"
        }
    }
}
results = storage.search('news', query)
```

## 10.7 Scrapy Pipeline 集成

### MySQL Pipeline

```python
import pymysql

class MySQLPipeline:
    def open_spider(self, spider):
        self.connection = pymysql.connect(
            host='localhost',
            user='root',
            password='password',
            database='crawler_db',
            charset='utf8mb4'
        )
        self.cursor = self.connection.cursor()
    
    def close_spider(self, spider):
        self.connection.commit()
        self.cursor.close()
        self.connection.close()
    
    def process_item(self, item, spider):
        sql = """
        INSERT INTO news (title, url, author)
        VALUES (%s, %s, %s)
        ON DUPLICATE KEY UPDATE title=VALUES(title)
        """
        
        self.cursor.execute(sql, (
            item.get('title'),
            item.get('url'),
            item.get('author')
        ))
        
        return item
```

### MongoDB Pipeline

```python
from pymongo import MongoClient

class MongoPipeline:
    def open_spider(self, spider):
        self.client = MongoClient('mongodb://localhost:27017/')
        self.db = self.client['crawler_db']
    
    def close_spider(self, spider):
```

```python
import redis
import json

class RedisPipeline:
```

## 10.8 数据备份

### 定期备份

```python
import shutil
from datetime import datetime

def backup_database(db_path, backup_dir='backups'):
    """备份数据库"""
    import os
    
    os.makedirs(backup_dir, exist_ok=True)
    
    timestamp = datetime.now().strftime('%Y%m%d_%H%M%S')
    backup_path = os.path.join(backup_dir, f'db_backup_{timestamp}.db')
    
    shutil.copy2(db_path, backup_path)
    
    print(f"备份完成: {backup_path}")
    
    # 清理旧备份（保留最近 7 天）
    cleanup_old_backups(backup_dir, days=7)

def cleanup_old_backups(backup_dir, days=7):
    """清理旧备份"""
    import os
    from datetime import datetime, timedelta
    
    cutoff = datetime.now() - timedelta(days=days)
    
    for filename in os.listdir(backup_dir):
        filepath = os.path.join(backup_dir, filename)
        if os.path.getmtime(filepath) < cutoff.timestamp():
            os.remove(filepath)
            print(f"删除旧备份: {filename}")
```
