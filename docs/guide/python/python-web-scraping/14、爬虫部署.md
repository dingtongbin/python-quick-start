## 14.1 Docker 容器化

### Dockerfile

```dockerfile
FROM python:3.10-slim

# 设置工作目录
WORKDIR /app

# 安装系统依赖
RUN apt-get update && apt-get install -y \
    gcc \
    default-libmysqlclient-dev \
    pkg-config \
    && rm -rf /var/lib/apt/lists/*

# 复制依赖文件
COPY requirements.txt .

# 安装 Python 依赖
RUN pip install --no-cache-dir -r requirements.txt

# 复制代码
COPY . .

# 创建日志目录
RUN mkdir -p /app/logs

# 运行爬虫
CMD ["scrapy", "crawl", "news"]
```

### docker-compose.yml

```yaml
version: '3.8'

services:
  crawler:
    build: .
    environment:
      - DB_HOST=mysql
      - REDIS_HOST=redis
    depends_on:
      - mysql
      - redis
    volumes:
      - ./logs:/app/logs
    restart: unless-stopped

  mysql:
    image: mysql:8.0
    environment:
      MYSQL_ROOT_PASSWORD: password
      MYSQL_DATABASE: crawler_db
    volumes:
      - mysql_data:/var/lib/mysql
    ports:
      - "3306:3306"

  redis:
    image: redis:7-alpine
    ports:
      - "6379:6379"
    volumes:
      - redis_data:/data

volumes:
  mysql_data:
  redis_data:
```

## 14.2 Celery 任务调度

### 配置 Celery

```python
# tasks.py
from celery import Celery
import subprocess
import logging

logger = logging.getLogger(__name__)

app = Celery(
    'crawler_tasks',
    broker='redis://redis:6379/0',
    backend='redis://redis:6379/1'
)

@app.task(bind=True, max_retries=3)
def run_spider(self, spider_name):
    """运行爬虫任务"""
    try:
        logger.info(f"开始运行爬虫: {spider_name}")
        
        cmd = f'scrapy crawl {spider_name}'
        result = subprocess.run(
            cmd,
            shell=True,
            capture_output=True,
            text=True,
            timeout=3600  # 1 小时超时
        )
        
        if result.returncode == 0:
            logger.info(f"爬虫完成: {spider_name}")
            return {'status': 'success'}
        else:
            logger.error(f"爬虫失败: {result.stderr}")
            raise Exception(result.stderr)
    
    except Exception as e:
        logger.error(f"任务失败: {e}")
        raise self.retry(exc=e, countdown=60 * 5)  # 5 分钟后重试

@app.task
def schedule_crawls():
    """定时调度爬虫"""
    spiders = ['sina', 'netease', 'jd']
    
    for spider in spiders:
        run_spider.delay(spider)
    
    return f'Scheduled {len(spiders)} crawls'
```

### Celery Beat 配置

```python
# celeryconfig.py
from celery.schedules import crontab

beat_schedule = {
    'hourly-crawl': {
        'task': 'tasks.schedule_crawls',
        'schedule': crontab(minute=0),
    },
    'daily-report': {
        'task': 'tasks.generate_report',
        'schedule': crontab(hour=0, minute=0),
    },
}

timezone = 'Asia/Shanghai'
```

### 启动服务

```bash
# Worker
celery -A tasks worker --loglevel=info --concurrency=4

# Beat
celery -A tasks beat --loglevel=info

# Flower (监控界面)
celery -A tasks flower --port=5555
```

## 14.3 监控告警

### Prometheus + Grafana

#### 安装

```bash
pip install prometheus-client
```

#### 指标收集

```python
# metrics.py
from prometheus_client import Counter, Histogram, Gauge
import time

# 指标定义
ITEMS_SCRAPED = Counter(
    'items_scraped_total',
    'Total items scraped',
    ['spider']
)

REQUEST_DURATION = Histogram(
    'request_duration_seconds',
    'Request duration',
    ['spider']
)

ACTIVE_SPIDERS = Gauge(
    'active_spiders',
    'Number of active spiders'
)

ERRORS_TOTAL = Counter(
    'errors_total',
    'Total errors',
    ['spider', 'error_type']
)

# 中间件
class MetricsMiddleware:
    def process_request(self, request, spider):
        request.meta['start_time'] = time.time()
    
    def process_response(self, request, response, spider):
        duration = time.time() - request.meta.get('start_time', time.time())
        REQUEST_DURATION.labels(spider=spider.name).observe(duration)
        return response
    
    def process_exception(self, request, exception, spider):
        ERRORS_TOTAL.labels(
            spider=spider.name,
            error_type=type(exception).__name__
        ).inc()
```

#### Grafana 仪表板

配置数据源和仪表板，可视化监控指标。

#### 结构化日志

```python
import logging
import json

class JSONFormatter(logging.Formatter):
    def format(self, record):
        log_data = {
            'timestamp': self.formatTime(record),
            'level': record.levelname,
            'message': record.getMessage(),
            'spider': getattr(record, 'spider', None),
            'url': getattr(record, 'url', None),
        }
        return json.dumps(log_data)

# 配置
handler = logging.FileHandler('crawler.log')
handler.setFormatter(JSONFormatter())
logger.addHandler(handler)
```

## 14.7 备份与恢复

### 自动备份脚本

```python
#!/usr/bin/env python3
import subprocess
from datetime import datetime

def backup_database():
    """备份数据库"""
    timestamp = datetime.now().strftime('%Y%m%d_%H%M%S')
    filename = f'backup_{timestamp}.sql'
    
    cmd = f'mysqldump -u root -ppassword crawler_db > /backups/{filename}'
    subprocess.run(cmd, shell=True)
    
    # 压缩
    subprocess.run(f'gzip /backups/{filename}', shell=True)
    
    # 清理旧备份（保留 30 天）
    subprocess.run('find /backups -mtime +30 -delete', shell=True)

if __name__ == '__main__':
    backup_database()
```

### Cron 定时任务

```bash
# 每天凌晨 2 点备份
0 2 * * * /usr/bin/python3 /app/backup.py >> /var/log/backup.log 2>&1
```
