### spiders/sina_spider.py

```python
import scrapy
from datetime import datetime
from news_crawler.items import NewsItem

class SinaSpider(scrapy.Spider):
    name = 'sina'
    allowed_domains = ['news.sina.com.cn']
    start_urls = ['https://news.sina.com.cn/']
    
    custom_settings = {
        'DOWNLOAD_DELAY': 2,
        'RANDOMIZE_DOWNLOAD_DELAY': True,
    }
    
    def parse(self, response):
        # 提取新闻链接
        for link in response.css('a[href*="/news/"]'):
            url = link.css('::attr(href)').get()
            
            if url:
                yield response.follow(
                    url,
                    callback=self.parse_detail,
                    meta={'source': '新浪新闻'}
                )
        
        # 分页
        next_page = response.css('a.next::attr(href)').get()
        if next_page:
            yield response.follow(next_page, self.parse)
    
    def parse_detail(self, response):
        item = NewsItem()
        
        item['source'] = response.meta.get('source')
        item['url'] = response.url
        
        # 标题
        item['title'] = response.css('h1.main-title::text').get()
        
        # 作者
        item['author'] = response.css('span.source::text').get()
        
        # 时间
        time_text = response.css('span.date::text').get()
        if time_text:
            try:
                item['publish_time'] = datetime.strptime(
                    time_text.strip(), 
                    '%Y年%m月%d日 %H:%M'
                )
            except:
                item['publish_time'] = None
        
        # 内容
        paragraphs = response.css('div.article p::text').getall()
        item['content'] = '\n'.join([p.strip() for p in paragraphs if p.strip()])
        
        # 摘要
        item['summary'] = response.css('meta[name="description"]::attr(content)').get()
        
        # 分类
        item['category'] = response.css('a.category::text').get()
        
        # 标签
        item['tags'] = response.css('span.tag::text').getall()
        
        # 图片
        item['images'] = response.css('img::attr(src)').getall()
        
        # 爬取时间
        item['crawl_time'] = datetime.now()
        
        yield item
```

### pipelines.py

```python
import json
import pymysql
import redis
from datetime import datetime
from scrapy.exceptions import DropItem

class DuplicatePipeline:
    """去重 Pipeline"""
    
    def __init__(self):
        self.redis_client = redis.Redis(host='localhost', port=6379, db=0)
    
    def process_item(self, item, spider):
        url = item.get('url')
        
        if self.redis_client.sismember('seen_urls', url):
            raise DropItem(f"重复: {url}")
        
        self.redis_client.sadd('seen_urls', url)
        return item
    
    def close_spider(self, spider):
        self.redis_client.close()

class MySQLPipeline:
    """MySQL Pipeline"""
    
    def open_spider(self, spider):
        self.connection = pymysql.connect(
            host='localhost',
            user='root',
            password='password',
            database='news_db',
            charset='utf8mb4'
        )
        self.cursor = self.connection.cursor()
        self.create_table()
    
    def create_table(self):
        sql = """
        CREATE TABLE IF NOT EXISTS news (
            id INT AUTO_INCREMENT PRIMARY KEY,
            source VARCHAR(50),
            title VARCHAR(500),
            url VARCHAR(1000) UNIQUE,
            author VARCHAR(100),
            publish_time DATETIME,
            content TEXT,
            summary TEXT,
            category VARCHAR(100),
            tags JSON,
            images JSON,
            crawl_time DATETIME,
            created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
            INDEX idx_publish_time (publish_time),
            INDEX idx_category (category)
        ) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4
        """
        self.cursor.execute(sql)
        self.connection.commit()
    
    def process_item(self, item, spider):
        sql = """
        INSERT INTO news 
        (source, title, url, author, publish_time, content, 
         summary, category, tags, images, crawl_time)
        VALUES (%s, %s, %s, %s, %s, %s, %s, %s, %s, %s, %s)
        ON DUPLICATE KEY UPDATE title=VALUES(title)
        """
        
        self.cursor.execute(sql, (
            item.get('source'),
            item.get('title'),
            item.get('url'),
            item.get('author'),
            item.get('publish_time'),
            item.get('content'),
            item.get('summary'),
            item.get('category'),
            json.dumps(item.get('tags', []), ensure_ascii=False),
            json.dumps(item.get('images', []), ensure_ascii=False),
            item.get('crawl_time')
        ))
        
        self.connection.commit()
        return item
    
    def close_spider(self, spider):
        self.cursor.close()
        self.connection.close()

class StatsPipeline:
    """统计 Pipeline"""
    
    def open_spider(self, spider):
        self.count = 0
        self.start_time = datetime.now()
    
    def process_item(self, item, spider):
        self.count += 1
        
        if self.count % 100 == 0:
            elapsed = (datetime.now() - self.start_time).total_seconds()
            speed = self.count / elapsed if elapsed > 0 else 0
            spider.logger.info(f"已爬取 {self.count} 条，速度: {speed:.2f} 条/秒")
        
        return item
    
    def close_spider(self, spider):
        elapsed = (datetime.now() - self.start_time).total_seconds()
        spider.logger.info(f"完成！共爬取 {self.count} 条，耗时 {elapsed:.2f} 秒")
```

#### tasks.py

```python
from celery import Celery
import subprocess

app = Celery('tasks', broker='redis://localhost:6379/0')

@app.task
def run_spider(spider_name):
    """运行爬虫"""
    cmd = f'scrapy crawl {spider_name}'
    result = subprocess.run(cmd, shell=True, capture_output=True, text=True)
    
    if result.returncode == 0:
        return {'status': 'success', 'output': result.stdout}
    else:
        return {'status': 'failed', 'error': result.stderr}

@app.task
def run_all_spiders():
    """运行所有爬虫"""
    spiders = ['sina', 'netease', 'sohu']
    results = []
    
    for spider in spiders:
        result = run_spider.delay(spider)
        results.append(result)
    
    return results
```

#### 定时配置

```python
# celeryconfig.py
from celery.schedules import crontab

beat_schedule = {
    'run-every-hour': {
        'task': 'tasks.run_all_spiders',
        'schedule': crontab(minute=0),  # 每小时执行
    },
}
```

#### 启动

```bash
# 启动 worker
celery -A tasks worker --loglevel=info

# 启动 beat（定时调度）
celery -A tasks beat --loglevel=info
```
