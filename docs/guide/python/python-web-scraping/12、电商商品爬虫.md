### spiders/jd_spider.py

```python
import scrapy
import json
from datetime import datetime
from news_crawler.items import ProductItem

class JDSpider(scrapy.Spider):
    name = 'jd'
    allowed_domains = ['jd.com']
    
    def __init__(self, keyword='手机', *args, **kwargs):
        super().__init__(*args, **kwargs)
        self.keyword = keyword
        self.start_urls = [
            f'https://search.jd.com/Search?keyword={keyword}&page=1'
        ]
    
    def parse(self, response):
        # 提取商品列表
        for item in response.css('li.gl-item'):
            product = ProductItem()
            
            # 商品 ID
            product['product_id'] = item.css('::attr(data-sku)').get()
            
            # 标题
            product['title'] = item.css('div.p-name a em::text').get()
            
            # 价格（需要异步加载）
            sku = product['product_id']
            if sku:
                yield scrapy.Request(
                    f'https://p.3.cn/prices/mgets?skuIds=J_{sku}',
                    callback=self.parse_price,
                    meta={'product': product}
                )
            
            # 图片
            product['images'] = [
                'https:' + img for img in 
                item.css('img::attr(data-lazy-img)').getall()
            ]
            
            # 详情链接
            product['detail_url'] = item.css('div.p-name a::attr(href)').get()
            
            # 卖家
            product['seller'] = item.css('div.p-shop a::text').get()
        
        # 分页
        next_page = response.css('a.pn-next::attr(href)').get()
        if next_page:
            yield response.follow(next_page, self.parse)
    
    def parse_price(self, response):
        """解析价格"""
        product = response.meta['product']
        
        try:
            data = json.loads(response.text)
            if data:
                product['price'] = data[0].get('p')
                product['original_price'] = data[0].get('m')
        except:
            pass
        
        yield product
```

### pipelines.py

```python
import pymysql
import json
from datetime import datetime

class ProductPipeline:
    def open_spider(self, spider):
        self.connection = pymysql.connect(
            host='localhost',
            user='root',
            password='password',
            database='ecommerce_db',
            charset='utf8mb4'
        )
        self.cursor = self.connection.cursor()
        self.create_tables()
    
    def create_tables(self):
        # 商品表
        self.cursor.execute("""
        CREATE TABLE IF NOT EXISTS products (
            id INT AUTO_INCREMENT PRIMARY KEY,
            source VARCHAR(50),
            product_id VARCHAR(50) UNIQUE,
            title VARCHAR(500),
            price DECIMAL(10, 2),
            original_price DECIMAL(10, 2),
            rating DECIMAL(3, 2),
            review_count INT,
            seller VARCHAR(200),
            brand VARCHAR(100),
            category VARCHAR(100),
            images JSON,
            detail_url VARCHAR(1000),
            crawl_time DATETIME,
            created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
            updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
        ) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4
        """)
        
        # 价格历史表
        self.cursor.execute("""
        CREATE TABLE IF NOT EXISTS price_history (
            id INT AUTO_INCREMENT PRIMARY KEY,
            product_id VARCHAR(50),
            price DECIMAL(10, 2),
            recorded_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
            INDEX idx_product (product_id),
            INDEX idx_time (recorded_at)
        ) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4
        """)
        
        self.connection.commit()
    
    def process_item(self, item, spider):
        # 插入或更新商品
        sql = """
        INSERT INTO products 
        (source, product_id, title, price, original_price, 
         seller, images, detail_url, crawl_time)
        VALUES (%s, %s, %s, %s, %s, %s, %s, %s, %s)
        ON DUPLICATE KEY UPDATE
            title=VALUES(title),
            price=VALUES(price),
            updated_at=NOW()
        """
        
        self.cursor.execute(sql, (
            item.get('source'),
            item.get('product_id'),
            item.get('title'),
            item.get('price'),
            item.get('original_price'),
            item.get('seller'),
            json.dumps(item.get('images', [])),
            item.get('detail_url'),
            item.get('crawl_time')
        ))
        
        # 记录价格历史
        if item.get('price'):
            self.cursor.execute("""
            INSERT INTO price_history (product_id, price)
            VALUES (%s, %s)
            """, (item.get('product_id'), item.get('price')))
        
        self.connection.commit()
        return item
    
    def close_spider(self, spider):
        self.cursor.close()
        self.connection.close()
```

    def check_price_drop(self, product_id, threshold=0.1):
        """检查价格下降"""
        cursor = self.connection.cursor()
        
        # 获取当前价格
        cursor.execute(
            "SELECT price FROM products WHERE product_id=%s",
            (product_id,)
        )
        current_price = cursor.fetchone()[0]
        
        # 获取历史最低价
        cursor.execute(
            "SELECT MIN(price) FROM price_history WHERE product_id=%s",
            (product_id,)
        )
        min_price = cursor.fetchone()[0]
        
        if min_price and current_price > min_price * (1 + threshold):
            drop_percent = (current_price - min_price) / min_price

                'product_id': product_id,

class CommentSpider(scrapy.Spider):
    name = 'comment'
    
    def parse_comments(self, response):
        """解析评论"""
        data = json.loads(response.text)
        
        for comment in data.get('comments', []):
            yield {
                'user': comment.get('nickname'),
                'rating': comment.get('score'),
                'content': comment.get('content'),
                'time': comment.get('creationTime'),
                'images': comment.get('images', [])
            }
        
        # 分页
        page = response.meta.get('page', 1)
        if page < 10:  # 限制页数
            next_page = page + 1
            yield scrapy.Request(
                f'{response.url}&page={next_page}',
                callback=self.parse_comments,
                meta={'page': next_page}

## 12.5 图片下载

### Images Pipeline

```python
from scrapy.pipelines.images import ImagesPipeline
from scrapy.exceptions import DropItem

class CustomImagesPipeline(ImagesPipeline):
    def get_media_requests(self, item, info):
        for image_url in item.get('images', []):
            yield scrapy.Request(image_url)
    
    def file_path(self, request, response=None, info=None):
        # 自定义文件名
        product_id = request.meta.get('product_id')
        image_name = request.url.split('/')[-1]
        return f'{product_id}/{image_name}'
    
    def item_completed(self, results, item, info):
        downloaded = [x for ok, x in results if ok]
        if not downloaded:
            raise DropItem("没有下载的图片")
        return item
```

import matplotlib.pyplot as plt
import pandas as pd

def plot_price_trend(product_id):
    """绘制价格趋势图"""
    conn = pymysql.connect(...)
    
    df = pd.read_sql(
        "SELECT recorded_at, price FROM price_history WHERE product_id=%s ORDER BY recorded_at",
